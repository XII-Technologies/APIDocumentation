72

95 XII_ASSERT_DEBUG(m_Remove != nullptr, "The property '{0}' has no setter function, thus it is read-only.", xiiAbstractProperty::GetPropertyName());

97 (static_cast<Class*>(pInstance)->*m_Remove)(uiIndex);

103 {
104 XII_ASSERT_DEBUG(m_Insert != nullptr && m_Remove != nullptr, "The property '{0}' has no remove and insert function, thus it is fixed-size.", xiiAbstractProperty::GetPropertyName());
105
106 while (uiCount < GetCount(pInstance))
107 {
108 Remove(pInstance, GetCount(pInstance) - 1);
109 }
110 while (uiCount > GetCount(pInstance))
111 {
112 RealType elem = RealType();
113 Insert(pInstance, GetCount(pInstance), &elem);
114 }
115 }
