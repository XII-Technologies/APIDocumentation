95 {

97  m_CenterAndRadius = mMat.TransformPosition(m_CenterAndRadius);
98 
99  xiiSimdFloat maxRadius = mMat.m_col0.Dot<3>(mMat.m_col0);
100  maxRadius = maxRadius.Max(mMat.m_col1.Dot<3>(mMat.m_col1));
101  maxRadius = maxRadius.Max(mMat.m_col2.Dot<3>(mMat.m_col2));
102  radius *= maxRadius.GetSqrt();
103 
104  m_CenterAndRadius.SetW(radius);
105 
106  xiiSimdVec4f newHalfExtents = mMat.m_col0.Abs() * m_BoxHalfExtents.x();
107  newHalfExtents += mMat.m_col1.Abs() * m_BoxHalfExtents.y();
108  newHalfExtents += mMat.m_col2.Abs() * m_BoxHalfExtents.z();
109 
110  m_BoxHalfExtents = newHalfExtents.CompMin(xiiSimdVec4f(radius));
