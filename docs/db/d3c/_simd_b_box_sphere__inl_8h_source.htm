100inline void xiiSimdBBoxSphere::ExpandToInclude(const xiiSimdBBoxSphere& rhs)
101{
102 xiiSimdBBox box = GetBox();
103 box.ExpandToInclude(rhs.GetBox());
104
105 xiiSimdVec4f center = box.GetCenter();
106 xiiSimdVec4f boxHalfExtents = center - box.m_Min;
107 xiiSimdFloat tmpRadius = boxHalfExtents.GetLength<3>();
108
109 const xiiSimdFloat fSphereRadiusA = (m_CenterAndRadius - center).GetLength<3>() + m_CenterAndRadius.w();
110 const xiiSimdFloat fSphereRadiusB = (rhs.m_CenterAndRadius - center).GetLength<3>() + rhs.m_CenterAndRadius.w();
111
112 m_CenterAndRadius = center;
113 m_CenterAndRadius.SetW(tmpRadius.Min(fSphereRadiusA.Max(fSphereRadiusB)));
114 m_BoxHalfExtents = boxHalfExtents;
115}

122XII_ALWAYS_INLINE void xiiSimdBBoxSphere::Transform(const xiiSimdMat4f& mMat)
123{
124 xiiSimdFloat radius = m_CenterAndRadius.w();
125 m_CenterAndRadius = mMat.TransformPosition(m_CenterAndRadius);
126
127 xiiSimdFloat maxRadius = mMat.m_col0.Dot<3>(mMat.m_col0);
128 maxRadius = maxRadius.Max(mMat.m_col1.Dot<3>(mMat.m_col1));
129 maxRadius = maxRadius.Max(mMat.m_col2.Dot<3>(mMat.m_col2));
130 radius *= maxRadius.GetSqrt();
131
132 m_CenterAndRadius.SetW(radius);
133
134 xiiSimdVec4f newHalfExtents = mMat.m_col0.Abs() * m_BoxHalfExtents.x();
135 newHalfExtents += mMat.m_col1.Abs() * m_BoxHalfExtents.y();
136 newHalfExtents += mMat.m_col2.Abs() * m_BoxHalfExtents.z();
137
138 m_BoxHalfExtents = newHalfExtents.CompMin(xiiSimdVec4f(radius));
139}
