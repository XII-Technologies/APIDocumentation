31template <typename T, xiiUInt16 Size>
32void xiiSmallArrayBase<T, Size>::CopyFrom(const xiiArrayPtr<const T>& other, xiiAllocatorBase* pAllocator)
33{
34 XII_ASSERT_DEV(other.GetCount() <= xiiSmallInvalidIndex, "Can't copy {} elements to small array. Maximum count is {}", other.GetCount(), xiiSmallInvalidIndex);
35
36 if (GetData() == other.GetPtr())
37 {
38 if (m_uiCount == other.GetCount())
39 return;
40
41 XII_ASSERT_DEV(m_uiCount > other.GetCount(), "Dangling array pointer. The given array pointer points to invalid memory.");
42
43 T* pElements = GetElementsPtr();
44 xiiMemoryUtils::Destruct(pElements + other.GetCount(), m_uiCount - other.GetCount());
45 m_uiCount = static_cast<xiiUInt16>(other.GetCount());
46 return;
47 }
48
49 const xiiUInt32 uiOldCount = m_uiCount;
50 const xiiUInt32 uiNewCount = other.GetCount();
51
52 if (uiNewCount > uiOldCount)
53 {
54 Reserve(static_cast<xiiUInt16>(uiNewCount), pAllocator);
55
56 T* pElements = GetElementsPtr();
57 xiiMemoryUtils::Copy(pElements, other.GetPtr(), uiOldCount);
58 xiiMemoryUtils::CopyConstructArray(pElements + uiOldCount, other.GetPtr() + uiOldCount, uiNewCount - uiOldCount);
59 }
60 else
61 {
62 T* pElements = GetElementsPtr();
63 xiiMemoryUtils::Copy(pElements, other.GetPtr(), uiNewCount);
64 xiiMemoryUtils::Destruct(pElements + uiNewCount, uiOldCount - uiNewCount);
65 }
66
67 m_uiCount = static_cast<xiiUInt16>(uiNewCount);
68}

70template <typename T, xiiUInt16 Size>
71void xiiSmallArrayBase<T, Size>::MoveFrom(xiiSmallArrayBase<T, Size>&& other, xiiAllocatorBase* pAllocator)
72{
73 Clear();
74
75 if (other.m_uiCapacity > Size)
76 {
77 if (m_uiCapacity > Size)
78 {
79 // only delete our own external storage
80 XII_DELETE_RAW_BUFFER(pAllocator, m_pElements);
81 }
82
83 m_uiCapacity = other.m_uiCapacity;
84 m_pElements = other.m_pElements;
85 }
86 else
87 {
88 xiiMemoryUtils::RelocateConstruct(GetElementsPtr(), other.GetElementsPtr(), other.m_uiCount);
89 }
90
91 m_uiCount = other.m_uiCount;
92 m_uiUserData = other.m_uiUserData;
93
94 // reset the other array to not reference the data anymore
95 other.m_pElements = nullptr;
96 other.m_uiCount = 0;
97 other.m_uiCapacity = 0;
98}

282template <typename T, xiiUInt16 Size>
283void xiiSmallArrayBase<T, Size>::RemoveAtAndSwap(xiiUInt32 uiIndex, xiiUInt16 uiNumElements /*= 1*/)
284{
285 XII_ASSERT_DEV(uiIndex + uiNumElements <= m_uiCount, "Out of bounds access. Array has {0} elements, trying to remove element at index {1}.", m_uiCount, uiIndex + uiNumElements - 1);
286
287 T* pElements = GetElementsPtr();
288
289 for (xiiUInt32 i = 0; i < uiNumElements; ++i)
290 {
291 m_uiCount--;
292
293 if (m_uiCount != uiIndex)
294 {
295 pElements[uiIndex] = std::move(pElements[m_uiCount]);
296 }
297 xiiMemoryUtils::Destruct(pElements + m_uiCount, 1);
298 ++uiIndex;
299 }

476template <typename T, xiiUInt16 Size>
477void xiiSmallArrayBase<T, Size>::Reserve(xiiUInt16 uiCapacity, xiiAllocatorBase* pAllocator)
478{
479 if (m_uiCapacity >= uiCapacity)
480 return;
481
482 const xiiUInt32 uiCurCap = static_cast<xiiUInt32>(m_uiCapacity);
483 xiiUInt32 uiNewCapacity = uiCurCap + (uiCurCap / 2);
484
485 uiNewCapacity = xiiMath::Max<xiiUInt32>(uiNewCapacity, uiCapacity);
486 uiNewCapacity = xiiMemoryUtils::AlignSize<xiiUInt32>(uiNewCapacity, CAPACITY_ALIGNMENT);
487 uiNewCapacity = xiiMath::Min<xiiUInt32>(uiNewCapacity, 0xFFFFu);
488
489 SetCapacity(static_cast<xiiUInt16>(uiNewCapacity), pAllocator);
490}

492template <typename T, xiiUInt16 Size>
493void xiiSmallArrayBase<T, Size>::Compact(xiiAllocatorBase* pAllocator)
494{
495 if (IsEmpty())
496 {
497 if (m_uiCapacity > Size)
498 {
499 // completely deallocate all data, if the array is empty.
500 XII_DELETE_RAW_BUFFER(pAllocator, m_pElements);
501 }
502
503 m_uiCapacity = Size;
504 m_pElements = nullptr;
505 }
506 else if (m_uiCapacity > Size)
507 {
508 xiiUInt32 uiNewCapacity = xiiMemoryUtils::AlignSize<xiiUInt32>(m_uiCount, CAPACITY_ALIGNMENT);
509 uiNewCapacity = xiiMath::Min<xiiUInt32>(uiNewCapacity, 0xFFFFu);
510
511 if (m_uiCapacity != uiNewCapacity)
512 SetCapacity(static_cast<xiiUInt16>(uiNewCapacity), pAllocator);
513 }
514}

538template <typename T, xiiUInt16 Size>
539void xiiSmallArrayBase<T, Size>::SetCapacity(xiiUInt16 uiCapacity, xiiAllocatorBase* pAllocator)
540{
541 if (m_uiCapacity > Size && uiCapacity > m_uiCapacity)
542 {
543 m_pElements = XII_EXTEND_RAW_BUFFER(pAllocator, m_pElements, m_uiCount, uiCapacity);
544 m_uiCapacity = uiCapacity;
545 }
546 else
547 {
548 // special case when migrating from in-place to external storage or shrinking
549 T* pOldElements = GetElementsPtr();
550
551 const xiiUInt32 uiOldCapacity = m_uiCapacity;
552 const xiiUInt32 uiNewCapacity = uiCapacity;
553 m_uiCapacity = xiiMath::Max(uiCapacity, Size);
554
555 if (uiNewCapacity > Size)
556 {
557 // new external storage
558 T* pNewElements = XII_NEW_RAW_BUFFER(pAllocator, T, uiCapacity);
559 xiiMemoryUtils::RelocateConstruct(pNewElements, pOldElements, m_uiCount);
560 m_pElements = pNewElements;
561 }
562 else
563 {
564 // Re-use inplace storage
565 xiiMemoryUtils::RelocateConstruct(GetElementsPtr(), pOldElements, m_uiCount);
566 }
567
568 if (uiOldCapacity > Size)
569 {
570 XII_DELETE_RAW_BUFFER(pAllocator, pOldElements);
571 }
572 }
573}
