141template <typename T, xiiUInt32 BlockSize, xiiBlockStorageType::Enum StorageType>
142T* xiiBlockStorage<T, BlockSize, StorageType>::Create()
143{
144 T* pNewObject = nullptr;
145 xiiUInt32 uiNewIndex = xiiInvalidIndex;
146
147 if (StorageType == xiiBlockStorageType::FreeList && m_uiFreelistStart != xiiInvalidIndex)
148 {
149 uiNewIndex = m_uiFreelistStart;
150
151 const xiiUInt32 uiBlockIndex = uiNewIndex / xiiDataBlock<T, BlockSize>::CAPACITY;
152 const xiiUInt32 uiInnerIndex = uiNewIndex - uiBlockIndex * xiiDataBlock<T, BlockSize>::CAPACITY;
153
154 pNewObject = &(m_Blocks[uiBlockIndex][uiInnerIndex]);
155
156 m_uiFreelistStart = *reinterpret_cast<xiiUInt32*>(pNewObject);
157 }
158 else
159 {
160 xiiDataBlock<T, BlockSize>* pBlock = nullptr;
161
162 if (m_Blocks.GetCount() > 0)
163 {
164 pBlock = &m_Blocks.PeekBack();
165 }
166
167 if (pBlock == nullptr || pBlock->IsFull())
168 {
169 m_Blocks.PushBack(m_pBlockAllocator->template AllocateBlock<T>());
170 pBlock = &m_Blocks.PeekBack();
171 }
172
173 pNewObject = pBlock->ReserveBack();
174 uiNewIndex = m_uiCount;
175
176 ++m_uiCount;
177 }
178
179 xiiMemoryUtils::Construct<SkipTrivialTypes>(pNewObject, 1);
180
181 if (StorageType == xiiBlockStorageType::FreeList)
182 {
183 m_UsedEntries.SetCount(m_uiCount);
184 m_UsedEntries.SetBit(uiNewIndex);
185 }
186
187 return pNewObject;
188}

223{

247XII_FORCE_INLINE void xiiBlockStorage<T, BlockSize, StorageType>::Delete(T* pObject, T*& out_pMovedObject, xiiTraitInt<xiiBlockStorageType::FreeList>)
248{
249 xiiUInt32 uiIndex = xiiInvalidIndex;
250 for (xiiUInt32 uiBlockIndex = 0; uiBlockIndex < m_Blocks.GetCount(); ++uiBlockIndex)
251 {
252 std::ptrdiff_t diff = pObject - m_Blocks[uiBlockIndex].m_pData;
253 if (diff >= 0 && diff < xiiDataBlock<T, BlockSize>::CAPACITY)
254 {
255 uiIndex = uiBlockIndex * xiiDataBlock<T, BlockSize>::CAPACITY + (xiiInt32)diff;

264 out_pMovedObject = pObject;
