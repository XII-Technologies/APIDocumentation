47 XII_FORCE_INLINE void GetClusterCornerPoints(const xiiCamera& camera, float fZf, float fZn, float fTanLeft, float fTanRight, float fTanBottom, float fTanTop, xiiInt32 x, xiiInt32 y, xiiInt32 z, xiiVec3* out_pCorners)
48 {
49 const xiiVec3& pos = camera.GetPosition();

51 const xiiVec3& dirRight = camera.GetDirRight();
52 const xiiVec3& dirUp = camera.GetDirUp();

54 const float fStartXf = fZf * fTanLeft;
55 const float fStartYf = fZf * fTanBottom;
56 const float fEndXf = fZf * fTanRight;
57 const float fEndYf = fZf * fTanTop;

61
62 float fXf = fStartXf + x * fStepXf;
63 float fYf = fStartYf + y * fStepYf;
64
65 out_pCorners[0] = pos + dirForward * fZf + dirRight * fXf - dirUp * fYf;
66 out_pCorners[1] = out_pCorners[0] + dirRight * fStepXf;
67 out_pCorners[2] = out_pCorners[0] - dirUp * fStepYf;
68 out_pCorners[3] = out_pCorners[2] + dirRight * fStepXf;
69
70 const float fStartXn = fZn * fTanLeft;
71 const float fStartYn = fZn * fTanBottom;
72 const float fEndXn = fZn * fTanRight;
73 const float fEndYn = fZn * fTanTop;
74
75 float fStepXn = (fEndXn - fStartXn) / NUM_CLUSTERS_X;
76 float fStepYn = (fEndYn - fStartYn) / NUM_CLUSTERS_Y;
77 float fXn = fStartXn + x * fStepXn;
78 float fYn = fStartYn + y * fStepYn;
79
80 out_pCorners[4] = pos + dirForward * fZn + dirRight * fXn - dirUp * fYn;
81 out_pCorners[5] = out_pCorners[4] + dirRight * fStepXn;
82 out_pCorners[6] = out_pCorners[4] - dirUp * fStepYn;
83 out_pCorners[7] = out_pCorners[6] + dirRight * fStepXn;

86 void FillClusterBoundingSpheres(const xiiCamera& camera, float fAspectRatio, xiiArrayPtr<xiiSimdBSphere> clusterBoundingSpheres)
87 {
88 XII_PROFILE_SCOPE("FillClusterBoundingSpheres");
89
91 if (camera.IsOrthographic())
92 return;
93
94 xiiMat4 mProj;
95 camera.GetProjectionMatrix(fAspectRatio, mProj);
96
97 xiiSimdVec4f stepScale;
98 xiiSimdVec4f tanLBLB;
99 {
100 xiiAngle fFovLeft;
101 xiiAngle fFovRight;
102 xiiAngle fFovBottom;
103 xiiAngle fFovTop;
104 xiiGraphicsUtils::ExtractPerspectiveMatrixFieldOfView(mProj, fFovLeft, fFovRight, fFovBottom, fFovTop);
105
106 const float fTanLeft = xiiMath::Tan(fFovLeft);
107 const float fTanRight = xiiMath::Tan(fFovRight);
108 const float fTanBottom = xiiMath::Tan(fFovBottom);
109 const float fTanTop = xiiMath::Tan(fFovTop);
110
111 float fStepXf = (fTanRight - fTanLeft) / NUM_CLUSTERS_X;
112 float fStepYf = (fTanTop - fTanBottom) / NUM_CLUSTERS_Y;
113
114 stepScale = xiiSimdVec4f(fStepXf, fStepYf, fStepXf, fStepYf);
115 tanLBLB = xiiSimdVec4f(fTanLeft, fTanBottom, fTanLeft, fTanBottom);
116 }
117
118 xiiSimdVec4f pos = xiiSimdConversion::ToVec3(camera.GetPosition());
119 xiiSimdVec4f dirForward = xiiSimdConversion::ToVec3(camera.GetDirForwards());
120 xiiSimdVec4f dirRight = xiiSimdConversion::ToVec3(camera.GetDirRight());
121 xiiSimdVec4f dirUp = xiiSimdConversion::ToVec3(camera.GetDirUp());
122
123
124 xiiSimdVec4f fZn = xiiSimdVec4f::MakeZero();
125 xiiSimdVec4f cc[8];
126
127 for (xiiInt32 z = 0; z < NUM_CLUSTERS_Z; z++)
128 {
129 xiiSimdVec4f fZf = xiiSimdVec4f(GetDepthFromSliceIndex(z));
130 xiiSimdVec4f zff_znn = fZf.GetCombined<xiiSwizzle::XXXX>(fZn);
131 xiiSimdVec4f steps = zff_znn.CompMul(stepScale);
132
133 xiiSimdVec4f depthF = pos + dirForward * fZf.x();
134 xiiSimdVec4f depthN = pos + dirForward * fZn.x();
135
136 xiiSimdVec4f startLBLB = zff_znn.CompMul(tanLBLB);
137
138 for (xiiInt32 y = 0; y < NUM_CLUSTERS_Y; y++)
139 {
140 for (xiiInt32 x = 0; x < NUM_CLUSTERS_X; x++)
141 {
142 xiiSimdVec4f xyxy = xiiSimdVec4i(x, y, x, y).ToFloat();
143 xiiSimdVec4f xfyf = startLBLB + (xyxy).CompMul(steps);
144
145 cc[0] = depthF + dirRight * xfyf.x() - dirUp * xfyf.y();
146 cc[1] = cc[0] + dirRight * steps.x();
147 cc[2] = cc[0] - dirUp * steps.y();
148 cc[3] = cc[2] + dirRight * steps.x();
149
150 cc[4] = depthN + dirRight * xfyf.z() - dirUp * xfyf.w();
151 cc[5] = cc[4] + dirRight * steps.z();
152 cc[6] = cc[4] - dirUp * steps.w();
153 cc[7] = cc[6] + dirRight * steps.z();
154
155 clusterBoundingSpheres[GetClusterIndexFromCoord(x, y, z)] = xiiSimdBSphere::MakeFromPoints(cc, 8);
156 }
157 }
158
159 fZn = fZf;
160 }
161 }

261 XII_FORCE_INLINE xiiSimdBBox GetScreenSpaceBounds(const xiiSimdBSphere& sphere, const xiiSimdMat4f& mViewMatrix, const xiiSimdMat4f& mProjectionMatrix)
262 {
263 xiiSimdVec4f viewSpaceCenter = mViewMatrix.TransformPosition(sphere.GetCenter());
264 xiiSimdFloat depth = viewSpaceCenter.z();
265 xiiSimdFloat radius = sphere.GetRadius();
266
267 xiiSimdVec4f mi;
268 xiiSimdVec4f ma;
269
270 if (viewSpaceCenter.GetLength<3>() > radius && depth > radius)
271 {
272 xiiSimdVec4f one = xiiSimdVec4f(1.0f);
273 xiiSimdVec4f oneNegOne = xiiSimdVec4f(1.0f, -1.0f, 1.0f, -1.0f);
274
275 xiiSimdVec4f pRadius = xiiSimdVec4f(radius / depth);
276 xiiSimdVec4f pRadius2 = pRadius.CompMul(pRadius);
277
278 xiiSimdVec4f xy = viewSpaceCenter / depth;
279 xiiSimdVec4f xxyy = xy.Get<xiiSwizzle::XXYY>();
280 xiiSimdVec4f nom = (pRadius2.CompMul(xxyy.CompMul(xxyy) - pRadius2 + one)).GetSqrt() - xxyy.CompMul(oneNegOne);
281 xiiSimdVec4f denom = pRadius2 - one;
282
283 xiiSimdVec4f projection = mProjectionMatrix.m_col0.GetCombined<xiiSwizzle::XXYY>(mProjectionMatrix.m_col1);
284 xiiSimdVec4f minXmaxX_minYmaxY = nom.CompDiv(denom).CompMul(oneNegOne).CompMul(projection);
285
286 mi = minXmaxX_minYmaxY.Get<xiiSwizzle::XZXX>();
287 ma = minXmaxX_minYmaxY.Get<xiiSwizzle::YWYY>();
288 }
289 else
290 {
291 mi = xiiSimdVec4f(-1.0f);
292 ma = xiiSimdVec4f(1.0f);
293 }
294
295 mi.SetZ(depth - radius);
296 ma.SetZ(depth + radius);
297
298 return xiiSimdBBox(mi, ma);
299 }

301 template <typename Cluster, typename IntersectionFunc>
302 XII_FORCE_INLINE void FillCluster(const xiiSimdBBox& screenSpaceBounds, xiiUInt32 uiBlockIndex, xiiUInt32 uiMask, Cluster* pClusters, IntersectionFunc func)
303 {
304 xiiSimdVec4f scale = xiiSimdVec4f(0.5f * NUM_CLUSTERS_X, -0.5f * NUM_CLUSTERS_Y, 1.0f, 1.0f);
305 xiiSimdVec4f bias = xiiSimdVec4f(0.5f * NUM_CLUSTERS_X, 0.5f * NUM_CLUSTERS_Y, 0.0f, 0.0f);
306
307 xiiSimdVec4f mi = xiiSimdVec4f::MulAdd(screenSpaceBounds.m_Min, scale, bias);
308 xiiSimdVec4f ma = xiiSimdVec4f::MulAdd(screenSpaceBounds.m_Max, scale, bias);
309
310 xiiSimdVec4i minXY_maxXY = xiiSimdVec4i::Truncate(mi.GetCombined<xiiSwizzle::XYXY>(ma));
311
312 xiiSimdVec4i maxClusterIndex = xiiSimdVec4i(NUM_CLUSTERS_X, NUM_CLUSTERS_Y, NUM_CLUSTERS_X, NUM_CLUSTERS_Y);
313 minXY_maxXY = minXY_maxXY.CompMin(maxClusterIndex - xiiSimdVec4i(1));
314 minXY_maxXY = minXY_maxXY.CompMax(xiiSimdVec4i::MakeZero());
315
316 xiiUInt32 xMin = minXY_maxXY.x();
317 xiiUInt32 yMin = minXY_maxXY.w();
318
319 xiiUInt32 xMax = minXY_maxXY.z();
320 xiiUInt32 yMax = minXY_maxXY.y();
321
322 xiiUInt32 zMin = GetSliceIndexFromDepth(screenSpaceBounds.m_Min.z());
323 xiiUInt32 zMax = GetSliceIndexFromDepth(screenSpaceBounds.m_Max.z());
324
325 for (xiiUInt32 z = zMin; z <= zMax; ++z)
326 {
327 for (xiiUInt32 y = yMin; y <= yMax; ++y)
328 {
329 for (xiiUInt32 x = xMin; x <= xMax; ++x)
330 {
331 xiiUInt32 uiClusterIndex = GetClusterIndexFromCoord(x, y, z);
332 if (func(uiClusterIndex))
333 {
334 pClusters[uiClusterIndex].m_BitMask[uiBlockIndex] |= uiMask;
335 }
336 }
337 }
338 }

360 template <typename Cluster>
361 void RasterizeSpotLight(const BoundingCone& spotLightCone, xiiUInt32 uiLightIndex, const xiiSimdMat4f& mViewMatrix, const xiiSimdMat4f& mProjectionMatrix, Cluster* pClusters, xiiSimdBSphere* pClusterBoundingSpheres)
362 {
363 xiiSimdVec4f position = spotLightCone.m_PositionAndRange;
364 xiiSimdFloat range = spotLightCone.m_PositionAndRange.w();
365 xiiSimdVec4f forwardDir = spotLightCone.m_ForwardDir;
366 xiiSimdFloat sinAngle = spotLightCone.m_SinCosAngle.x();
367 xiiSimdFloat cosAngle = spotLightCone.m_SinCosAngle.y();
368
369 // First calculate a bounding sphere around the cone to get min and max bounds
370 xiiSimdVec4f bSphereCenter;
371 xiiSimdFloat bSphereRadius;
372 if (sinAngle > 0.707107f) // sin(45)
373 {
374 bSphereCenter = position + forwardDir * cosAngle * range;
375 bSphereRadius = sinAngle * range;
376 }
377 else
378 {
379 bSphereRadius = range / (cosAngle + cosAngle);
380 bSphereCenter = position + forwardDir * bSphereRadius;
381 }
382
383 xiiSimdBSphere spotLightSphere(bSphereCenter, bSphereRadius);
384 xiiSimdBBox screenSpaceBounds = GetScreenSpaceBounds(spotLightSphere, mViewMatrix, mProjectionMatrix);
385
386 const xiiUInt32 uiBlockIndex = uiLightIndex / 32;
387 const xiiUInt32 uiMask = 1 << (uiLightIndex - uiBlockIndex * 32);
388
389 FillCluster(screenSpaceBounds, uiBlockIndex, uiMask, pClusters, [&](xiiUInt32 uiClusterIndex) {
390 xiiSimdBSphere clusterSphere = pClusterBoundingSpheres[uiClusterIndex];
391 xiiSimdFloat clusterRadius = clusterSphere.GetRadius();
392
393 xiiSimdVec4f toConePos = clusterSphere.m_CenterAndRadius - position;
394 xiiSimdFloat projected = forwardDir.Dot<3>(toConePos);
395 xiiSimdFloat distToConeSq = toConePos.Dot<3>(toConePos);
396 xiiSimdFloat distClosestP = cosAngle * (distToConeSq - projected * projected).GetSqrt() - projected * sinAngle;
397
398 bool angleCull = distClosestP > clusterRadius;
399 bool frontCull = projected > clusterRadius + range;
400 bool backCull = projected < -clusterRadius;
401
402 return !(angleCull || frontCull || backCull);
403 });
404 }

418 template <typename Cluster>
419 void RasterizeBox(const xiiTransform& transform, xiiUInt32 uiDecalIndex, const xiiSimdMat4f& mViewProjectionMatrix, Cluster* pClusters, xiiSimdBSphere* pClusterBoundingSpheres)
420 {
421 xiiSimdMat4f decalToWorld = xiiSimdConversion::ToTransform(transform).GetAsMat4();
422 xiiSimdMat4f worldToDecal = decalToWorld.GetInverse();
423
424 xiiVec3 corners[8];
425 xiiBoundingBox(xiiVec3(-1), xiiVec3(1)).GetCorners(corners);
426
427 xiiSimdMat4f decalToScreen = mViewProjectionMatrix * decalToWorld;
428 bool bInsideBox = false;
429 xiiSimdBBox screenSpaceBounds = xiiSimdBBox::MakeInvalid();
430
431 for (xiiUInt32 i = 0; i < 8; ++i)
432 {
433 xiiSimdVec4f corner = xiiSimdConversion::ToVec3(corners[i]);
434 xiiSimdVec4f screenSpaceCorner = decalToScreen.TransformPosition(corner);
435 xiiSimdFloat depth = screenSpaceCorner.w();
436 bInsideBox |= depth < xiiSimdFloat::MakeZero();
437
438 screenSpaceCorner /= depth;
439 screenSpaceCorner = screenSpaceCorner.GetCombined<xiiSwizzle::XYZW>(xiiSimdVec4f(depth));
440
441 screenSpaceBounds.m_Min = screenSpaceBounds.m_Min.CompMin(screenSpaceCorner);
442 screenSpaceBounds.m_Max = screenSpaceBounds.m_Max.CompMax(screenSpaceCorner);
443 }
444
445 if (bInsideBox)
446 {
447 screenSpaceBounds.m_Min = xiiSimdVec4f(-1.0f).GetCombined<xiiSwizzle::XYZW>(screenSpaceBounds.m_Min);
448 screenSpaceBounds.m_Max = xiiSimdVec4f(1.0f).GetCombined<xiiSwizzle::XYZW>(screenSpaceBounds.m_Max);
449 }
