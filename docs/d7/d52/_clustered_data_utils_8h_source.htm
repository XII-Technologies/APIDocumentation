86  void FillClusterBoundingSpheres(const xiiCamera& camera, float fAspectRatio, xiiArrayPtr<xiiSimdBSphere> clusterBoundingSpheres)
87  {
88  XII_PROFILE_SCOPE("FillClusterBoundingSpheres");
89 
91  if (camera.IsOrthographic())
92  return;
93 
94  xiiMat4 mProj;
95  camera.GetProjectionMatrix(fAspectRatio, mProj);
96 
97  xiiSimdVec4f stepScale;
98  xiiSimdVec4f tanLBLB;
99  {
100  xiiAngle fFovLeft;
101  xiiAngle fFovRight;
102  xiiAngle fFovBottom;
103  xiiAngle fFovTop;
104  xiiGraphicsUtils::ExtractPerspectiveMatrixFieldOfView(mProj, fFovLeft, fFovRight, fFovBottom, fFovTop);
105 
106  const float fTanLeft = xiiMath::Tan(fFovLeft);
107  const float fTanRight = xiiMath::Tan(fFovRight);
108  const float fTanBottom = xiiMath::Tan(fFovBottom);
109  const float fTanTop = xiiMath::Tan(fFovTop);
110 
111  float fStepXf = (fTanRight - fTanLeft) / NUM_CLUSTERS_X;
112  float fStepYf = (fTanTop - fTanBottom) / NUM_CLUSTERS_Y;
113 
114  stepScale = xiiSimdVec4f(fStepXf, fStepYf, fStepXf, fStepYf);
115  tanLBLB = xiiSimdVec4f(fTanLeft, fTanBottom, fTanLeft, fTanBottom);
116  }
117 
118  xiiSimdVec4f pos = xiiSimdConversion::ToVec3(camera.GetPosition());
119  xiiSimdVec4f dirForward = xiiSimdConversion::ToVec3(camera.GetDirForwards());
120  xiiSimdVec4f dirRight = xiiSimdConversion::ToVec3(camera.GetDirRight());
121  xiiSimdVec4f dirUp = xiiSimdConversion::ToVec3(camera.GetDirUp());
122 
123 
124  xiiSimdVec4f fZn = xiiSimdVec4f::ZeroVector();
125  xiiSimdVec4f cc[8];
126 
127  for (xiiInt32 z = 0; z < NUM_CLUSTERS_Z; z++)
128  {
129  xiiSimdVec4f fZf = xiiSimdVec4f(GetDepthFromSliceIndex(z));
130  xiiSimdVec4f zff_znn = fZf.GetCombined<xiiSwizzle::XXXX>(fZn);
131  xiiSimdVec4f steps = zff_znn.CompMul(stepScale);
132 
133  xiiSimdVec4f depthF = pos + dirForward * fZf.x();
134  xiiSimdVec4f depthN = pos + dirForward * fZn.x();
135 
136  xiiSimdVec4f startLBLB = zff_znn.CompMul(tanLBLB);
137 
138  for (xiiInt32 y = 0; y < NUM_CLUSTERS_Y; y++)
139  {
140  for (xiiInt32 x = 0; x < NUM_CLUSTERS_X; x++)
141  {
142  xiiSimdVec4f xyxy = xiiSimdVec4i(x, y, x, y).ToFloat();
143  xiiSimdVec4f xfyf = startLBLB + (xyxy).CompMul(steps);
144 
145  cc[0] = depthF + dirRight * xfyf.x() - dirUp * xfyf.y();
146  cc[1] = cc[0] + dirRight * steps.x();
147  cc[2] = cc[0] - dirUp * steps.y();
148  cc[3] = cc[2] + dirRight * steps.x();
149 
150  cc[4] = depthN + dirRight * xfyf.z() - dirUp * xfyf.w();
151  cc[5] = cc[4] + dirRight * steps.z();
152  cc[6] = cc[4] - dirUp * steps.w();
153  cc[7] = cc[6] + dirRight * steps.z();
154 
155  clusterBoundingSpheres[GetClusterIndexFromCoord(x, y, z)].SetFromPoints(cc, 8);
156  }
157  }
158 
159  fZn = fZf;
160  }
161  }

262  XII_FORCE_INLINE xiiSimdBBox GetScreenSpaceBounds(const xiiSimdBSphere& sphere, const xiiSimdMat4f& mViewMatrix, const xiiSimdMat4f& mProjectionMatrix)
263  {
264  xiiSimdVec4f viewSpaceCenter = mViewMatrix.TransformPosition(sphere.GetCenter());
265  xiiSimdFloat depth = viewSpaceCenter.z();
266  xiiSimdFloat radius = sphere.GetRadius();
267 
268  xiiSimdVec4f mi;
269  xiiSimdVec4f ma;
270 
271  if (viewSpaceCenter.GetLength<3>() > radius && depth > radius)
272  {
273  xiiSimdVec4f one = xiiSimdVec4f(1.0f);
274  xiiSimdVec4f oneNegOne = xiiSimdVec4f(1.0f, -1.0f, 1.0f, -1.0f);
275 
276  xiiSimdVec4f pRadius = xiiSimdVec4f(radius / depth);
277  xiiSimdVec4f pRadius2 = pRadius.CompMul(pRadius);
278 
279  xiiSimdVec4f xy = viewSpaceCenter / depth;
280  xiiSimdVec4f xxyy = xy.Get<xiiSwizzle::XXYY>();
281  xiiSimdVec4f nom = (pRadius2.CompMul(xxyy.CompMul(xxyy) - pRadius2 + one)).GetSqrt() - xxyy.CompMul(oneNegOne);
282  xiiSimdVec4f denom = pRadius2 - one;
283 
284  xiiSimdVec4f projection = mProjectionMatrix.m_col0.GetCombined<xiiSwizzle::XXYY>(mProjectionMatrix.m_col1);
285  xiiSimdVec4f minXmaxX_minYmaxY = nom.CompDiv(denom).CompMul(oneNegOne).CompMul(projection);
286 
287  mi = minXmaxX_minYmaxY.Get<xiiSwizzle::XZXX>();
288  ma = minXmaxX_minYmaxY.Get<xiiSwizzle::YWYY>();
289  }
290  else
291  {
292  mi = xiiSimdVec4f(-1.0f);
293  ma = xiiSimdVec4f(1.0f);
294  }
295 
296  mi.SetZ(depth - radius);
297  ma.SetZ(depth + radius);
298 
299  return xiiSimdBBox(mi, ma);
300  }

302  template <typename Cluster, typename IntersectionFunc>
303  XII_FORCE_INLINE void FillCluster(const xiiSimdBBox& screenSpaceBounds, xiiUInt32 uiBlockIndex, xiiUInt32 uiMask, Cluster* pClusters, IntersectionFunc func)
304  {
305  xiiSimdVec4f scale = xiiSimdVec4f(0.5f * NUM_CLUSTERS_X, -0.5f * NUM_CLUSTERS_Y, 1.0f, 1.0f);
306  xiiSimdVec4f bias = xiiSimdVec4f(0.5f * NUM_CLUSTERS_X, 0.5f * NUM_CLUSTERS_Y, 0.0f, 0.0f);
307 
308  xiiSimdVec4f mi = xiiSimdVec4f::MulAdd(screenSpaceBounds.m_Min, scale, bias);
309  xiiSimdVec4f ma = xiiSimdVec4f::MulAdd(screenSpaceBounds.m_Max, scale, bias);
310 
311  xiiSimdVec4i minXY_maxXY = xiiSimdVec4i::Truncate(mi.GetCombined<xiiSwizzle::XYXY>(ma));
312 
313  xiiSimdVec4i maxClusterIndex = xiiSimdVec4i(NUM_CLUSTERS_X, NUM_CLUSTERS_Y, NUM_CLUSTERS_X, NUM_CLUSTERS_Y);
314  minXY_maxXY = minXY_maxXY.CompMin(maxClusterIndex - xiiSimdVec4i(1));
315  minXY_maxXY = minXY_maxXY.CompMax(xiiSimdVec4i::ZeroVector());
316 
317  xiiUInt32 xMin = minXY_maxXY.x();
318  xiiUInt32 yMin = minXY_maxXY.w();
319 
320  xiiUInt32 xMax = minXY_maxXY.z();
321  xiiUInt32 yMax = minXY_maxXY.y();
322 
323  xiiUInt32 zMin = GetSliceIndexFromDepth(screenSpaceBounds.m_Min.z());
324  xiiUInt32 zMax = GetSliceIndexFromDepth(screenSpaceBounds.m_Max.z());
325 
326  for (xiiUInt32 z = zMin; z <= zMax; ++z)
327  {
328  for (xiiUInt32 y = yMin; y <= yMax; ++y)
329  {
330  for (xiiUInt32 x = xMin; x <= xMax; ++x)
331  {
332  xiiUInt32 uiClusterIndex = GetClusterIndexFromCoord(x, y, z);
333  if (func(uiClusterIndex))
334  {
335  pClusters[uiClusterIndex].m_BitMask[uiBlockIndex] |= uiMask;
336  }
337  }
338  }
339  }

361  template <typename Cluster>
362  void RasterizeSpotLight(const BoundingCone& spotLightCone, xiiUInt32 uiLightIndex, const xiiSimdMat4f& mViewMatrix, const xiiSimdMat4f& mProjectionMatrix, Cluster* pClusters, xiiSimdBSphere* pClusterBoundingSpheres)
363  {
364  xiiSimdVec4f position = spotLightCone.m_PositionAndRange;
365  xiiSimdFloat range = spotLightCone.m_PositionAndRange.w();
366  xiiSimdVec4f forwardDir = spotLightCone.m_ForwardDir;
367  xiiSimdFloat sinAngle = spotLightCone.m_SinCosAngle.x();
368  xiiSimdFloat cosAngle = spotLightCone.m_SinCosAngle.y();
369 
370  // First calculate a bounding sphere around the cone to get min and max bounds
371  xiiSimdVec4f bSphereCenter;
372  xiiSimdFloat bSphereRadius;
373  if (sinAngle > 0.707107f) // sin(45)
374  {
375  bSphereCenter = position + forwardDir * cosAngle * range;
376  bSphereRadius = sinAngle * range;
377  }
378  else
379  {
380  bSphereRadius = range / (cosAngle + cosAngle);
381  bSphereCenter = position + forwardDir * bSphereRadius;
382  }
383 
384  xiiSimdBSphere spotLightSphere(bSphereCenter, bSphereRadius);
385  xiiSimdBBox screenSpaceBounds = GetScreenSpaceBounds(spotLightSphere, mViewMatrix, mProjectionMatrix);
386 
387  const xiiUInt32 uiBlockIndex = uiLightIndex / 32;
388  const xiiUInt32 uiMask = 1 << (uiLightIndex - uiBlockIndex * 32);
389 
390  FillCluster(screenSpaceBounds, uiBlockIndex, uiMask, pClusters, [&](xiiUInt32 uiClusterIndex) {
391  xiiSimdBSphere clusterSphere = pClusterBoundingSpheres[uiClusterIndex];
392  xiiSimdFloat clusterRadius = clusterSphere.GetRadius();
393 
394  xiiSimdVec4f toConePos = clusterSphere.m_CenterAndRadius - position;
395  xiiSimdFloat projected = forwardDir.Dot<3>(toConePos);
396  xiiSimdFloat distToConeSq = toConePos.Dot<3>(toConePos);
397  xiiSimdFloat distClosestP = cosAngle * (distToConeSq - projected * projected).GetSqrt() - projected * sinAngle;
398 
399  bool angleCull = distClosestP > clusterRadius;
400  bool frontCull = projected > clusterRadius + range;
401  bool backCull = projected < -clusterRadius;
402 
403  return !(angleCull || frontCull || backCull);
404  });
405  }

421  {
422  xiiSimdMat4f decalToWorld = xiiSimdConversion::ToTransform(transform).GetAsMat4();
423  xiiSimdMat4f worldToDecal = decalToWorld.GetInverse();
424 
425  xiiVec3 corners[8];
