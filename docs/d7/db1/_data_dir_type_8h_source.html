190
191 while (uiBytesSkipped < uiBytes)
192 {
193 xiiUInt64 uiBytesToRead = xiiMath::Min<xiiUInt64>(uiBytes - uiBytesSkipped, 1024);
194
195 xiiUInt64 uiBytesRead = Read(uiTempBuffer, uiBytesToRead);

197 uiBytesSkipped += uiBytesRead;

202 }
