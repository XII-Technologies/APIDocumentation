168 virtual xiiUInt64 Skip(xiiUInt64 uiBytes)
169 {
170 xiiUInt8 uiTempBuffer[1024];
171
172 xiiUInt64 uiBytesSkipped = 0;
173
174 while (uiBytesSkipped < uiBytes)
175 {
176 xiiUInt64 uiBytesToRead = xiiMath::Min<xiiUInt64>(uiBytes - uiBytesSkipped, 1024);
177
178 xiiUInt64 uiBytesRead = Read(uiTempBuffer, uiBytesToRead);
179
180 uiBytesSkipped += uiBytesRead;
181
182 // Terminate early if the stream didn't read as many bytes as we requested (EOF for example)
183 if (uiBytesRead < uiBytesToRead)
184 break;
185 }
186
187 return uiBytesSkipped;
188 }
