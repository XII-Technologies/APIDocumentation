129 
130  out_desc.m_uiTypeSize = uiBaseOffset + m_InstanceDataAllocator.GetTotalDataSize();
131  out_desc.m_uiTypeAlignment = xiiMath::Max(out_desc.m_uiTypeAlignment, uiMaxAlignment);
