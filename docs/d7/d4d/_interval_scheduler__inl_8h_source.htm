102 template <typename T>
103 void xiiIntervalScheduler<T>::Update(xiiTime deltaTime, RunWorkCallback runWorkCallback)
104 {
105  if (deltaTime <= xiiTime::Zero())
106  return;
107 
108  m_CurrentTime += deltaTime;
109 
110  if (m_Data.IsEmpty())
111  {
112  m_fNumWorkToSchedule = 0.0;
113  }
114  else
115  {
116  double fNumWork = 0;
117  for (xiiUInt32 i = 0; i < HistogramSize; ++i)
118  {
119  fNumWork += (1.0 / xiiMath::Max(m_HistogramSlotValues[i], deltaTime).GetSeconds()) * m_Histogram[i];
120  }
121  fNumWork *= deltaTime.GetSeconds();
122 
123  if (m_fNumWorkToSchedule == 0.0)
124  {
125  m_fNumWorkToSchedule = fNumWork;
126  }
127  else
128  {
129  // running average of num work per update to prevent huge spikes
130  m_fNumWorkToSchedule = xiiMath::Lerp<double>(m_fNumWorkToSchedule, fNumWork, 0.05);
131  }
132 
133  const float fRemainder = static_cast<float>(xiiMath::Fraction(m_fNumWorkToSchedule));
134  const int pos = static_cast<int>(m_CurrentTime.GetNanoseconds());
135  const xiiUInt32 extra = GetRandomZeroToOne(pos, m_uiSeed) < fRemainder ? 1 : 0;
136  const xiiUInt32 uiScheduleCount = xiiMath::Min(static_cast<xiiUInt32>(m_fNumWorkToSchedule) + extra, m_Data.GetCount());
137 
138  // schedule work
139  {
140  auto it = m_Data.GetIterator();
141  for (xiiUInt32 i = 0; i < uiScheduleCount; ++i, ++it)
142  {
143  auto& data = it.Value();
144  if (data.IsValid())
145  {
146  if (runWorkCallback.IsValid())
147  {
148  runWorkCallback(data.m_Work, m_CurrentTime - data.m_LastScheduledTime);
149  }
150 
151  // add a little bit of random jitter so we don't end up with perfect timings that might collide with other work
152  data.m_DueTime = m_CurrentTime + xiiMath::Max(data.m_Interval, deltaTime) + GetRandomTimeJitter(i, m_uiSeed);
153  data.m_LastScheduledTime = m_CurrentTime;
154  }
155 
156  m_ScheduledWork.PushBack(it);
157  }
158  }
159 
160  // re-sort
161  for (auto& it : m_ScheduledWork)
162  {
163  if (it.Value().IsValid())
164  {
165  // make a copy of data and re-insert at new due time
166  Data data = it.Value();
167  m_WorkIdToData[data.m_Work] = InsertData(data);
168  }
169 
170  m_Data.Remove(it);
171  }
172  m_ScheduledWork.Clear();
173  }
174 }
