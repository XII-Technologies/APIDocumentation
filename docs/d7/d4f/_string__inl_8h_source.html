76template <xiiUInt16 Size>
77void xiiHybridStringBase<Size>::operator=(const char* szString)
78{
79 xiiUInt32 uiElementCount = xiiStringUtils::GetStringElementCount(szString);
80
81 if (szString + uiElementCount < m_Data.GetData() || szString >= m_Data.GetData() + m_Data.GetCount())
82 {
83 // source string is outside our own memory, so no overlapped copy
84 }
85 else
86 {
87 // source string overlaps with our own memory -> we can't increase the size of our memory, as that might invalidate the source data
88 XII_ASSERT_DEBUG(uiElementCount < m_Data.GetCount(), "Invalid copy of overlapping string data.");
89 }
90
91 m_Data.SetCountUninitialized(uiElementCount + 1);
92 xiiStringUtils::Copy(&m_Data[0], uiElementCount + 1, szString);
93}

129template <xiiUInt16 Size>
130xiiStringView xiiHybridStringBase<Size>::GetSubString(xiiUInt32 uiFirstCharacter, xiiUInt32 uiNumCharacters) const
131{
132 const char* szStart = GetData();
133 if (xiiUnicodeUtils::MoveToNextUtf8(szStart, uiFirstCharacter).Failed())
134 return {}; // szStart was moved too far, the result is just an empty string
135
136 const char* szEnd = szStart;
137 xiiUnicodeUtils::MoveToNextUtf8(szEnd, uiNumCharacters).IgnoreResult(); // if it fails, szEnd just points to the end of this string
138
139 return xiiStringView(szStart, szEnd);
140}
