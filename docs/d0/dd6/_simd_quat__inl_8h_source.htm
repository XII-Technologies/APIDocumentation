57 XII_ALWAYS_INLINE xiiSimdMat4f xiiSimdQuat::GetAsMat4() const
58 {
59  const xiiSimdVec4f xyz = m_v;
60  const xiiSimdVec4f x2y2z2 = xyz + xyz;
61  const xiiSimdVec4f xx2yy2zz2 = x2y2z2.CompMul(xyz);
62 
63  // diagonal terms
64  // 1 - (yy2 + zz2)
65  // 1 - (xx2 + zz2)
66  // 1 - (xx2 + yy2)
67  const xiiSimdVec4f yy2_xx2_xx2 = xx2yy2zz2.Get<xiiSwizzle::YXXX>();
68  const xiiSimdVec4f zz2_zz2_yy2 = xx2yy2zz2.Get<xiiSwizzle::ZZYX>();
69  xiiSimdVec4f diagonal = xiiSimdVec4f(1.0f) - (yy2_xx2_xx2 + zz2_zz2_yy2);
70  diagonal.SetW(xiiSimdFloat::Zero());
71 
72  // non diagonal terms
73  // xy2 +- wz2
74  // yz2 +- wx2
75  // xz2 +- wy2
76  const xiiSimdVec4f x_y_x = xyz.Get<xiiSwizzle::XYXX>();
77  const xiiSimdVec4f y2_z2_z2 = x2y2z2.Get<xiiSwizzle::YZZX>();
78  const xiiSimdVec4f base = x_y_x.CompMul(y2_z2_z2);
79 
80  const xiiSimdVec4f z2_x2_y2 = x2y2z2.Get<xiiSwizzle::ZXYX>();
81  const xiiSimdVec4f offset = z2_x2_y2 * m_v.w();
82 
83  const xiiSimdVec4f adds = base + offset;
84  const xiiSimdVec4f subs = base - offset;
85 
86  // final matrix layout
87  // col0 = (diaX, addX, subZ, diaW)
88  const xiiSimdVec4f addX_u_diaX_u = adds.GetCombined<xiiSwizzle::XXXX>(diagonal);
89  const xiiSimdVec4f subZ_u_diaW_u = subs.GetCombined<xiiSwizzle::ZXWX>(diagonal);
90  const xiiSimdVec4f col0 = addX_u_diaX_u.GetCombined<xiiSwizzle::ZXXZ>(subZ_u_diaW_u);
91 
92  // col1 = (subX, diaY, addY, diaW)
93  const xiiSimdVec4f subX_u_diaY_u = subs.GetCombined<xiiSwizzle::XXYX>(diagonal);
94  const xiiSimdVec4f addY_u_diaW_u = adds.GetCombined<xiiSwizzle::YXWX>(diagonal);
95  const xiiSimdVec4f col1 = subX_u_diaY_u.GetCombined<xiiSwizzle::XZXZ>(addY_u_diaW_u);
96 
97  // col2 = (addZ, subY, diaZ, diaW)
98  const xiiSimdVec4f addZ_u_subY_u = adds.GetCombined<xiiSwizzle::ZXYX>(subs);
99  const xiiSimdVec4f col2 = addZ_u_subY_u.GetCombined<xiiSwizzle::XZZW>(diagonal);
100 
101  return xiiSimdMat4f(col0, col1, col2, xiiSimdVec4f(0, 0, 0, 1));
102 }
