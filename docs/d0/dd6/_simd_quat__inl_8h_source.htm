58 XII_ALWAYS_INLINE xiiSimdMat4f xiiSimdQuat::GetAsMat4() const
59 {
60  const xiiSimdVec4f xyz = m_v;
61  const xiiSimdVec4f x2y2z2 = xyz + xyz;
62  const xiiSimdVec4f xx2yy2zz2 = x2y2z2.CompMul(xyz);
63 
64  // diagonal terms
65  // 1 - (yy2 + zz2)
66  // 1 - (xx2 + zz2)
67  // 1 - (xx2 + yy2)
68  const xiiSimdVec4f yy2_xx2_xx2 = xx2yy2zz2.Get<xiiSwizzle::YXXX>();
69  const xiiSimdVec4f zz2_zz2_yy2 = xx2yy2zz2.Get<xiiSwizzle::ZZYX>();
70  xiiSimdVec4f diagonal = xiiSimdVec4f(1.0f) - (yy2_xx2_xx2 + zz2_zz2_yy2);
71  diagonal.SetW(xiiSimdFloat::MakeZero());
72 
73  // non diagonal terms
74  // xy2 +- wz2
75  // yz2 +- wx2
76  // xz2 +- wy2
77  const xiiSimdVec4f x_y_x = xyz.Get<xiiSwizzle::XYXX>();
78  const xiiSimdVec4f y2_z2_z2 = x2y2z2.Get<xiiSwizzle::YZZX>();
79  const xiiSimdVec4f base = x_y_x.CompMul(y2_z2_z2);
80 
81  const xiiSimdVec4f z2_x2_y2 = x2y2z2.Get<xiiSwizzle::ZXYX>();
82  const xiiSimdVec4f offset = z2_x2_y2 * m_v.w();
83 
84  const xiiSimdVec4f adds = base + offset;
85  const xiiSimdVec4f subs = base - offset;
86 
87  // final matrix layout
88  // col0 = (diaX, addX, subZ, diaW)
89  const xiiSimdVec4f addX_u_diaX_u = adds.GetCombined<xiiSwizzle::XXXX>(diagonal);
90  const xiiSimdVec4f subZ_u_diaW_u = subs.GetCombined<xiiSwizzle::ZXWX>(diagonal);
91  const xiiSimdVec4f col0 = addX_u_diaX_u.GetCombined<xiiSwizzle::ZXXZ>(subZ_u_diaW_u);
92 
93  // col1 = (subX, diaY, addY, diaW)
94  const xiiSimdVec4f subX_u_diaY_u = subs.GetCombined<xiiSwizzle::XXYX>(diagonal);
95  const xiiSimdVec4f addY_u_diaW_u = adds.GetCombined<xiiSwizzle::YXWX>(diagonal);
96  const xiiSimdVec4f col1 = subX_u_diaY_u.GetCombined<xiiSwizzle::XZXZ>(addY_u_diaW_u);
97 
98  // col2 = (addZ, subY, diaZ, diaW)
99  const xiiSimdVec4f addZ_u_subY_u = adds.GetCombined<xiiSwizzle::ZXYX>(subs);
100  const xiiSimdVec4f col2 = addZ_u_subY_u.GetCombined<xiiSwizzle::XZZW>(diagonal);
101 
102  return xiiSimdMat4f::MakeFromColumns(col0, col1, col2, xiiSimdVec4f(0, 0, 0, 1));
103 }
