96template <typename KEY, typename VALUE>
97void xiiObjectMetaData<KEY, VALUE>::AttachMetaDataToAbstractGraph(xiiAbstractObjectGraph& inout_graph) const
98{
99 auto& AllNodes = inout_graph.GetAllNodes();
100
101 XII_LOCK(m_pMetaStorage->m_Mutex);
102
103 xiiHashTable<xiiStringView, xiiVariant> DefaultValues;
104
105 // store the default values in an easily accessible hash map, to be able to compare against them
106 {
107 DefaultValues.Reserve(m_DefaultValue.GetDynamicRTTI()->GetProperties().GetCount());
108
109 for (const auto& pProp : m_DefaultValue.GetDynamicRTTI()->GetProperties())
110 {
111 if (pProp->GetCategory() != xiiPropertyCategory::Member)
112 continue;
113
114 DefaultValues[pProp->GetPropertyName()] = xiiReflectionUtils::GetMemberPropertyValue(static_cast<const xiiAbstractMemberProperty*>(pProp), &m_DefaultValue);
115 }
116 }
117
118 // now serialize all properties that differ from the default value
119 {
120 xiiVariant value;
121
122 for (auto it = AllNodes.GetIterator(); it.IsValid(); ++it)
123 {
124 auto* pNode = it.Value();
125 const xiiUuid& guid = pNode->GetGuid();
126
127 const VALUE* pMeta = nullptr;
128 if (!m_pMetaStorage->m_MetaData.TryGetValue(guid, pMeta)) // TryGetValue is not const correct with the second parameter
129 continue; // it is the default object, so all values are default -> skip
130
131 for (const auto& pProp : pMeta->GetDynamicRTTI()->GetProperties())
132 {
133 if (pProp->GetCategory() != xiiPropertyCategory::Member)
134 continue;
135
136 value = xiiReflectionUtils::GetMemberPropertyValue(static_cast<const xiiAbstractMemberProperty*>(pProp), pMeta);
137
138 if (value.IsValid() && DefaultValues[pProp->GetPropertyName()] != value)
139 {
140 pNode->AddProperty(pProp->GetPropertyName(), value);
141 }
142 }
143 }
144 }
145}

148template <typename KEY, typename VALUE>
149void xiiObjectMetaData<KEY, VALUE>::RestoreMetaDataFromAbstractGraph(const xiiAbstractObjectGraph& graph)
150{
151 XII_LOCK(m_pMetaStorage->m_Mutex);
152
153 xiiHybridArray<xiiString, 16> PropertyNames;
154
155 // find all properties (names) that we want to read
156 {
157 for (const auto& pProp : m_DefaultValue.GetDynamicRTTI()->GetProperties())
158 {
159 if (pProp->GetCategory() != xiiPropertyCategory::Member)
160 continue;
161
162 PropertyNames.PushBack(pProp->GetPropertyName());
163 }
164 }
165
166 auto& AllNodes = graph.GetAllNodes();
167
168 for (auto it = AllNodes.GetIterator(); it.IsValid(); ++it)
169 {
170 auto* pNode = it.Value();
171 const xiiUuid& guid = pNode->GetGuid();
172
173 for (const auto& name : PropertyNames)
174 {
175 if (const auto* pProp = pNode->FindProperty(name))
176 {
177 VALUE* pValue = &m_pMetaStorage->m_MetaData[guid];
178
179 xiiReflectionUtils::SetMemberPropertyValue(
180 static_cast<const xiiAbstractMemberProperty*>(pValue->GetDynamicRTTI()->FindPropertyByName(name)), pValue, pProp->m_Value);
181 }
182 }
183 }
