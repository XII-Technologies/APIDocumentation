154

163 template <typename Type>
164 void xiiBoundingBoxSphereTemplate<Type>::Transform(const xiiMat4Template<Type>& mTransform)
165 {
166  m_vCenter = mTransform.TransformPosition(m_vCenter);
167  const xiiVec3Template<Type> Scale = mTransform.GetScalingFactors();
168  m_fSphereRadius *= xiiMath::Max(Scale.x, Scale.y, Scale.z);
169 
170  xiiMat3Template<Type> mAbsRotation = mTransform.GetRotationalPart();
171  for (xiiUInt32 i = 0; i < 9; ++i)
172  {
173  mAbsRotation.m_fElementsCM[i] = xiiMath::Abs(mAbsRotation.m_fElementsCM[i]);

177 }
