145template <typename Type>
146void xiiBoundingBoxSphereTemplate<Type>::ExpandToInclude(const xiiBoundingBoxSphereTemplate& rhs)
147{
148 xiiBoundingBoxTemplate<Type> box;
149 box.m_vMin = m_vCenter - m_vBoxHalfExtends;
150 box.m_vMax = m_vCenter + m_vBoxHalfExtends;
151 box.ExpandToInclude(rhs.GetBox());
152
153 xiiBoundingBoxSphereTemplate<Type> result = xiiBoundingBoxSphereTemplate<Type>::MakeFromBox(box);
154
155 const Type fSphereRadiusA = (m_vCenter - result.m_vCenter).GetLength() + m_fSphereRadius;
156 const Type fSphereRadiusB = (rhs.m_vCenter - result.m_vCenter).GetLength() + rhs.m_fSphereRadius;
157
158 m_vCenter = result.m_vCenter;
159 m_fSphereRadius = xiiMath::Min(result.m_fSphereRadius, xiiMath::Max(fSphereRadiusA, fSphereRadiusB));
160 m_vBoxHalfExtends = result.m_vBoxHalfExtends;
161}

163template <typename Type>
164void xiiBoundingBoxSphereTemplate<Type>::Transform(const xiiMat4Template<Type>& mTransform)
165{
166 m_vCenter = mTransform.TransformPosition(m_vCenter);
167 const xiiVec3Template<Type> Scale = mTransform.GetScalingFactors();
168 m_fSphereRadius *= xiiMath::Max(Scale.x, Scale.y, Scale.z);
169
170 xiiMat3Template<Type> mAbsRotation = mTransform.GetRotationalPart();
171 for (xiiUInt32 i = 0; i < 9; ++i)
172 {
173 mAbsRotation.m_fElementsCM[i] = xiiMath::Abs(mAbsRotation.m_fElementsCM[i]);
174 }
175
176 m_vBoxHalfExtends = mAbsRotation.TransformDirection(m_vBoxHalfExtends).CompMin(xiiVec3(m_fSphereRadius));
177}
