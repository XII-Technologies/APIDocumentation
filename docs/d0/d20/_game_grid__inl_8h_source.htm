100bool xiiGameGrid<CellData>::PickCell(const xiiVec3& vRayStartPos, const xiiVec3& vRayDirNorm, xiiVec2I32* out_pCellCoord, xiiVec3* out_pIntersection) const
101{
102 xiiPlane p = xiiPlane::MakeFromNormalAndPoint(m_mRotateToWorldspace * xiiVec3(0, 0, -1), m_vWorldSpaceOrigin);
103
104 xiiVec3 vPos;
105
106 if (!p.GetRayIntersection(vRayStartPos, vRayDirNorm, nullptr, &vPos))
107 return false;
108
109 if (out_pIntersection)
110 *out_pIntersection = vPos;
111
112 if (out_pCellCoord)
113 *out_pCellCoord = GetCellAtWorldPosition(vPos);
114
115 return true;

128template <class CellData>
129bool xiiGameGrid<CellData>::GetRayIntersection(const xiiVec3& vRayStartWorldSpace, const xiiVec3& vRayDirNormalizedWorldSpace, float fMaxLength, float& out_fIntersection, xiiVec2I32& out_vCellCoord) const
130{
131 const xiiVec3 vRayStart = m_mRotateToGridspace * (vRayStartWorldSpace - m_vWorldSpaceOrigin);
132 const xiiVec3 vRayDir = m_mRotateToGridspace * vRayDirNormalizedWorldSpace;
133
134 xiiVec3 vGridBox(m_uiGridSizeX, m_uiGridSizeY, 1.0f);
135
136 const xiiBoundingBox localBox(xiiVec3(0.0f), m_vLocalSpaceCellSize.CompMul(vGridBox));
137
138 if (localBox.Contains(vRayStart))
139 {
140 // if the ray is already inside the box, we know that a cell is hit
141 out_fIntersection = 0.0f;
142 }
143 else
144 {
145 if (!localBox.GetRayIntersection(vRayStart, vRayDir, &out_fIntersection, nullptr))
146 return false;
147
148 if (out_fIntersection > fMaxLength)
149 return false;
150 }
151
152 const xiiVec3 vEnterPos = vRayStart + vRayDir * out_fIntersection;
153
154 const xiiVec3 vCell = vEnterPos.CompMul(m_vInverseLocalSpaceCellSize);
155
156 // Without the Floor, the border case when the position is outside (-1 / -1) is not immediately detected
157 out_vCellCoord = xiiVec2I32((xiiInt32)xiiMath::Floor(vCell.x), (xiiInt32)xiiMath::Floor(vCell.y));
158 out_vCellCoord.x = xiiMath::Clamp(out_vCellCoord.x, 0, m_uiGridSizeX - 1);
159 out_vCellCoord.y = xiiMath::Clamp(out_vCellCoord.y, 0, m_uiGridSizeY - 1);
160
161 return true;
162}
