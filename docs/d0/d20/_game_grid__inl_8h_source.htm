129 template <class CellData>
130 bool xiiGameGrid<CellData>::GetRayIntersection(const xiiVec3& vRayStartWorldSpace, const xiiVec3& vRayDirNormalizedWorldSpace, float fMaxLength, float& out_fIntersection, xiiVec2I32& out_CellCoord) const
131 {
132  const xiiVec3 vRayStart = m_mRotateToGridspace * (vRayStartWorldSpace - m_vWorldSpaceOrigin);
133  const xiiVec3 vRayDir = m_mRotateToGridspace * vRayDirNormalizedWorldSpace;
134 
135  xiiVec3 vGridBox(m_uiGridSizeX, m_uiGridSizeY, 1.0f);
136 
137  const xiiBoundingBox localBox(xiiVec3(0.0f), m_vLocalSpaceCellSize.CompMul(vGridBox));
138 
139  if (localBox.Contains(vRayStart))
140  {
141  // if the ray is already inside the box, we know that a cell is hit
142  out_fIntersection = 0.0f;
143  }
144  else
145  {
146  if (!localBox.GetRayIntersection(vRayStart, vRayDir, &out_fIntersection, nullptr))
147  return false;
148 
149  if (out_fIntersection > fMaxLength)
150  return false;
151  }
152 
153  const xiiVec3 vEnterPos = vRayStart + vRayDir * out_fIntersection;
154 
155  const xiiVec3 vCell = vEnterPos.CompMul(m_vInverseLocalSpaceCellSize);
156 
157  // Without the Floor, the border case when the position is outside (-1 / -1) is not immediately detected
158  out_CellCoord = xiiVec2I32((xiiInt32)xiiMath::Floor(vCell.x), (xiiInt32)xiiMath::Floor(vCell.y));
159  out_CellCoord.x = xiiMath::Clamp(out_CellCoord.x, 0, m_uiGridSizeX - 1);
160  out_CellCoord.y = xiiMath::Clamp(out_CellCoord.y, 0, m_uiGridSizeY - 1);
161 
162  return true;
163 }
