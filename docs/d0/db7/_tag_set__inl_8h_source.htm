230 template <typename BlockStorageAllocator>
231 void xiiTagSetTemplate<BlockStorageAllocator>::Set(const xiiTag& tag)
232 {
233  XII_ASSERT_DEV(tag.IsValid(), "Only valid tags can be set in a tag set!");
234 
235  if (m_TagBlocks.IsEmpty())
236  {
237  Reallocate(tag.m_uiBlockIndex, tag.m_uiBlockIndex);
238  }
239  else if (IsTagInAllocatedRange(tag) == false)
240  {
241  const xiiUInt32 uiNewBlockStart = xiiMath::Min<xiiUInt32>(tag.m_uiBlockIndex, GetTagBlockStart());
242  const xiiUInt32 uiNewBlockEnd = xiiMath::Max<xiiUInt32>(tag.m_uiBlockIndex, GetTagBlockEnd());
243 
244  Reallocate(uiNewBlockStart, uiNewBlockEnd);
245  }
246 
247  xiiUInt64& tagBlock = m_TagBlocks[tag.m_uiBlockIndex - GetTagBlockStart()];
248 
249  const xiiUInt64 bitMask = XII_BIT(tag.m_uiBitIndex);
250  const bool bBitWasSet = ((tagBlock & bitMask) != 0);
251 
252  tagBlock |= bitMask;
253 
254  if (!bBitWasSet)
255  {
256  IncreaseTagCount();
257  }
258 }

296 template <typename BlockStorageAllocator>
297 bool xiiTagSetTemplate<BlockStorageAllocator>::IsAnySet(const xiiTagSetTemplate& otherSet) const
298 {
299  // If any of the sets is empty nothing can match
300  if (IsEmpty() || otherSet.IsEmpty())
301  return false;
302 
303  // Calculate range to compare
304  const xiiUInt32 uiMaxBlockStart = xiiMath::Max(GetTagBlockStart(), otherSet.GetTagBlockStart());
305  const xiiUInt32 uiMinBlockEnd = xiiMath::Min(GetTagBlockEnd(), otherSet.GetTagBlockEnd());
306 
307  if (uiMaxBlockStart > uiMinBlockEnd)
308  return false;
309 
310  for (xiiUInt32 i = uiMaxBlockStart; i < uiMinBlockEnd; ++i)
311  {
312  const xiiUInt32 uiThisBlockStorageIndex = i - GetTagBlockStart();
313  const xiiUInt32 uiOtherBlockStorageIndex = i - otherSet.GetTagBlockStart();
314 
315  if ((m_TagBlocks[uiThisBlockStorageIndex] & otherSet.m_TagBlocks[uiOtherBlockStorageIndex]) != 0)
316  {
317  return true;
318  }
319  }
320 
321  return false;
322 }

377 template <typename BlockStorageAllocator>
378 void xiiTagSetTemplate<BlockStorageAllocator>::Reallocate(xiiUInt32 uiNewTagBlockStart, xiiUInt32 uiNewMaxBlockIndex)
379 {
380  XII_ASSERT_DEV(uiNewTagBlockStart < xiiSmallInvalidIndex, "Tag block start is too big");
381  const xiiUInt16 uiNewBlockArraySize = static_cast<xiiUInt16>((uiNewMaxBlockIndex - uiNewTagBlockStart) + 1);
382 
383  // Early out for non-filled tag sets
384  if (m_TagBlocks.IsEmpty())
385  {
386  m_TagBlocks.SetCount(uiNewBlockArraySize);
387  SetTagBlockStart(static_cast<xiiUInt16>(uiNewTagBlockStart));
388 
389  return;
390  }
391 
392  XII_ASSERT_DEBUG(uiNewTagBlockStart <= GetTagBlockStart(), "New block start must be smaller or equal to current block start!");
393 
394  xiiSmallArray<xiiUInt64, 32, BlockStorageAllocator> helperArray;
395  helperArray.SetCount(uiNewBlockArraySize);
396 
397  const xiiUInt32 uiOldBlockStartOffset = GetTagBlockStart() - uiNewTagBlockStart;
398 
399  // Copy old data to the new array
400  xiiMemoryUtils::Copy(helperArray.GetData() + uiOldBlockStartOffset, m_TagBlocks.GetData(), m_TagBlocks.GetCount());
401 
402  // Use array ptr copy assignment so it doesn't modify the user data in m_TagBlocks
403  m_TagBlocks = helperArray.GetArrayPtr();
404  SetTagBlockStart(static_cast<xiiUInt16>(uiNewTagBlockStart));
405 }
