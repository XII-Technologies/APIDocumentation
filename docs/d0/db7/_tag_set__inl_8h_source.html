122 // this should be decltype(auto) c = ...; but MSVC 16 is too dumb for that (MSVC 15 works fine)

127 }

242template <typename BlockStorageAllocator>
243void xiiTagSetTemplate<BlockStorageAllocator>::Set(const xiiTag& tag)
244{
245 XII_ASSERT_DEV(tag.IsValid(), "Only valid tags can be set in a tag set!");
246
247 if (m_TagBlocks.IsEmpty())
248 {
249 Reallocate(tag.m_uiBlockIndex, tag.m_uiBlockIndex);
250 }
251 else if (IsTagInAllocatedRange(tag) == false)
252 {
253 const xiiUInt32 uiNewBlockStart = xiiMath::Min<xiiUInt32>(tag.m_uiBlockIndex, GetTagBlockStart());
254 const xiiUInt32 uiNewBlockEnd = xiiMath::Max<xiiUInt32>(tag.m_uiBlockIndex, GetTagBlockEnd());
255
256 Reallocate(uiNewBlockStart, uiNewBlockEnd);
257 }
258
259 xiiUInt64& tagBlock = m_TagBlocks[tag.m_uiBlockIndex - GetTagBlockStart()];
260
261 const xiiUInt64 bitMask = XII_BIT(tag.m_uiBitIndex);
262 const bool bBitWasSet = ((tagBlock & bitMask) != 0);
263
264 tagBlock |= bitMask;
265
266 if (!bBitWasSet)
267 {
268 IncreaseTagCount();
269 }
270}

272template <typename BlockStorageAllocator>
273void xiiTagSetTemplate<BlockStorageAllocator>::Remove(const xiiTag& tag)
274{
275 XII_ASSERT_DEV(tag.IsValid(), "Only valid tags can be cleared from a tag set!");
276
277 if (IsTagInAllocatedRange(tag))
278 {
279 xiiUInt64& tagBlock = m_TagBlocks[tag.m_uiBlockIndex - GetTagBlockStart()];
280
281 const xiiUInt64 bitMask = XII_BIT(tag.m_uiBitIndex);
282 const bool bBitWasSet = ((tagBlock & bitMask) != 0);
283
284 tagBlock &= ~bitMask;
285
286 if (bBitWasSet)
287 {
288 DecreaseTagCount();
289 }
290 }
291}

308template <typename BlockStorageAllocator>
309bool xiiTagSetTemplate<BlockStorageAllocator>::IsAnySet(const xiiTagSetTemplate& otherSet) const
310{
311 // If any of the sets is empty nothing can match
312 if (IsEmpty() || otherSet.IsEmpty())
313 return false;
314
315 // Calculate range to compare
316 const xiiUInt32 uiMaxBlockStart = xiiMath::Max(GetTagBlockStart(), otherSet.GetTagBlockStart());
317 const xiiUInt32 uiMinBlockEnd = xiiMath::Min(GetTagBlockEnd(), otherSet.GetTagBlockEnd());
318
319 if (uiMaxBlockStart > uiMinBlockEnd)
320 return false;
321
322 for (xiiUInt32 i = uiMaxBlockStart; i < uiMinBlockEnd; ++i)
323 {
324 const xiiUInt32 uiThisBlockStorageIndex = i - GetTagBlockStart();
325 const xiiUInt32 uiOtherBlockStorageIndex = i - otherSet.GetTagBlockStart();
326
327 if ((m_TagBlocks[uiThisBlockStorageIndex] & otherSet.m_TagBlocks[uiOtherBlockStorageIndex]) != 0)
328 return true;
329 }
330 return false;
331}

385template <typename BlockStorageAllocator>
386void xiiTagSetTemplate<BlockStorageAllocator>::Reallocate(xiiUInt32 uiNewTagBlockStart, xiiUInt32 uiNewMaxBlockIndex)
387{
388 XII_ASSERT_DEV(uiNewTagBlockStart < xiiSmallInvalidIndex, "Tag block start is too big");
389 const xiiUInt16 uiNewBlockArraySize = static_cast<xiiUInt16>((uiNewMaxBlockIndex - uiNewTagBlockStart) + 1);
390
391 // Early out for non-filled tag sets
392 if (m_TagBlocks.IsEmpty())
393 {
394 m_TagBlocks.SetCount(uiNewBlockArraySize);
395 SetTagBlockStart(static_cast<xiiUInt16>(uiNewTagBlockStart));
396
397 return;
398 }
399
400 XII_ASSERT_DEBUG(uiNewTagBlockStart <= GetTagBlockStart(), "New block start must be smaller or equal to current block start!");
401
402 xiiSmallArray<xiiUInt64, 32, BlockStorageAllocator> helperArray;
403 helperArray.SetCount(uiNewBlockArraySize);
404
405 const xiiUInt32 uiOldBlockStartOffset = GetTagBlockStart() - uiNewTagBlockStart;
406
407 // Copy old data to the new array
408 xiiMemoryUtils::Copy(helperArray.GetData() + uiOldBlockStartOffset, m_TagBlocks.GetData(), m_TagBlocks.GetCount());
409
410 // Use array ptr copy assignment so it doesn't modify the user data in m_TagBlocks
411 m_TagBlocks = helperArray.GetArrayPtr();
412 SetTagBlockStart(static_cast<xiiUInt16>(uiNewTagBlockStart));
413}
