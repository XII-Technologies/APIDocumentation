117template <xiiUInt32 BlockSize>
118void* xiiLargeBlockAllocator<BlockSize>::Allocate(size_t uiAlign)
119{
120 XII_ASSERT_RELEASE(xiiMath::IsPowerOf2((xiiUInt32)uiAlign), "Alignment must be power of two");
121
122 xiiTime fAllocationTime = xiiTime::Now();
123
124 XII_LOCK(m_Mutex);
125
126 void* ptr = nullptr;
127
128 if (!m_FreeBlocks.IsEmpty())
129 {
130 // Re-use a super block
131 xiiUInt32 uiFreeBlockIndex = m_FreeBlocks.PeekBack();
132 m_FreeBlocks.PopBack();
133
134 const xiiUInt32 uiSuperBlockIndex = uiFreeBlockIndex / SuperBlock::NUM_BLOCKS;
135 const xiiUInt32 uiInnerBlockIndex = uiFreeBlockIndex & (SuperBlock::NUM_BLOCKS - 1);
136 SuperBlock& superBlock = m_SuperBlocks[uiSuperBlockIndex];
137 ++superBlock.m_uiUsedBlocks;
138
139 ptr = xiiMemoryUtils::AddByteOffset(superBlock.m_pBasePtr, uiInnerBlockIndex * BlockSize);
140 }
141 else
142 {
143 // Allocate a new super block
144 void* pMemory = xiiPageAllocator::AllocatePage(SuperBlock::SIZE_IN_BYTES);
145 XII_CHECK_ALIGNMENT(pMemory, uiAlign);
146
147 SuperBlock superBlock;
148 superBlock.m_pBasePtr = pMemory;
149 superBlock.m_uiUsedBlocks = 1;
150
151 m_SuperBlocks.PushBack(superBlock);
152
153 const xiiUInt32 uiBlockBaseIndex = (m_SuperBlocks.GetCount() - 1) * SuperBlock::NUM_BLOCKS;
154 for (xiiUInt32 i = SuperBlock::NUM_BLOCKS - 1; i > 0; --i)
155 {
156 m_FreeBlocks.PushBack(uiBlockBaseIndex + i);
157 }
158
159 ptr = pMemory;
160 }
161
162 if (m_TrackingMode >= xiiAllocatorTrackingMode::AllocationStats)
163 {
164 xiiMemoryTracker::AddAllocation(m_Id, m_TrackingMode, ptr, BlockSize, uiAlign, xiiTime::Now() - fAllocationTime);
165 }
166
167 return ptr;
168}

170template <xiiUInt32 BlockSize>
171void xiiLargeBlockAllocator<BlockSize>::Deallocate(void* ptr)
172{
173 XII_LOCK(m_Mutex);
174
175 if (m_TrackingMode >= xiiAllocatorTrackingMode::AllocationStats)
176 {
177 xiiMemoryTracker::RemoveAllocation(m_Id, ptr);
178 }
179
180 // find super block
181 bool bFound = false;
182 xiiUInt32 uiSuperBlockIndex = m_SuperBlocks.GetCount();
183 std::ptrdiff_t diff = 0;
184
185 for (; uiSuperBlockIndex-- > 0;)
186 {
187 diff = (char*)ptr - (char*)m_SuperBlocks[uiSuperBlockIndex].m_pBasePtr;
188 if (diff >= 0 && diff < SuperBlock::SIZE_IN_BYTES)
189 {
190 bFound = true;
191 break;
192 }
193 }
194
195 XII_IGNORE_UNUSED(bFound);
196 XII_ASSERT_DEV(bFound, "'{0}' was not allocated with this allocator", xiiArgP(ptr));
197
198 SuperBlock& superBlock = m_SuperBlocks[uiSuperBlockIndex];
199 --superBlock.m_uiUsedBlocks;
200
201 if (superBlock.m_uiUsedBlocks == 0 && m_FreeBlocks.GetCount() > SuperBlock::NUM_BLOCKS * 4)
202 {
203 // give memory back
204 xiiPageAllocator::DeallocatePage(superBlock.m_pBasePtr);
205
206 m_SuperBlocks.RemoveAtAndSwap(uiSuperBlockIndex);
207 const xiiUInt32 uiLastSuperBlockIndex = m_SuperBlocks.GetCount();
208
209 // patch free list
210 for (xiiUInt32 i = 0; i < m_FreeBlocks.GetCount(); ++i)
211 {
212 const xiiUInt32 uiIndex = m_FreeBlocks[i];
213 const xiiUInt32 uiSBIndex = uiIndex / SuperBlock::NUM_BLOCKS;
214
215 if (uiSBIndex == uiSuperBlockIndex)
216 {
217 // points to the block we just removed
218 m_FreeBlocks.RemoveAtAndSwap(i);
219 --i;
220 }
221 else if (uiSBIndex == uiLastSuperBlockIndex)
222 {
223 // points to the block we just swapped
224 m_FreeBlocks[i] = uiSuperBlockIndex * SuperBlock::NUM_BLOCKS + (uiIndex & (SuperBlock::NUM_BLOCKS - 1));
225 }
226 }
227 }
228 else
229 {
230 // add block to free list
231 const xiiUInt32 uiInnerBlockIndex = (xiiUInt32)(diff / BlockSize);
232 m_FreeBlocks.PushBack(uiSuperBlockIndex * SuperBlock::NUM_BLOCKS + uiInnerBlockIndex);
233 }
