41{
42 const xiiUInt64 uiCount = array.GetCount();

67template <typename KeyType, typename ValueType, typename Comparer>
68xiiResult xiiDeduplicationWriteContext::WriteMap(xiiStreamWriter& ref_stream, const xiiMapBase<KeyType, ValueType, Comparer>& map, WriteMapMode mode)
69{
70 const xiiUInt64 uiWriteSize = map.GetCount();
71 XII_SUCCEED_OR_RETURN(ref_stream.WriteQWordValue(&uiWriteSize));
72
73 if (mode == WriteMapMode::DedupKey)
74 {
75 for (auto It = map.GetIterator(); It.IsValid(); ++It)
76 {
77 XII_SUCCEED_OR_RETURN(WriteObject(ref_stream, It.Key()));
78 XII_SUCCEED_OR_RETURN(xiiStreamWriterUtil::Serialize<ValueType>(ref_stream, It.Value()));
79 }
80 }
81 else if (mode == WriteMapMode::DedupValue)
82 {
83 for (auto It = map.GetIterator(); It.IsValid(); ++It)
84 {
85 XII_SUCCEED_OR_RETURN(xiiStreamWriterUtil::Serialize<KeyType>(ref_stream, It.Key()));
86 XII_SUCCEED_OR_RETURN(WriteObject(ref_stream, It.Value()));
87 }
88 }
89 else
90 {
91 for (auto It = map.GetIterator(); It.IsValid(); ++It)
92 {
93 XII_SUCCEED_OR_RETURN(WriteObject(ref_stream, It.Key()));
94 XII_SUCCEED_OR_RETURN(WriteObject(ref_stream, It.Value()));
95 }
96 }
97
98 return XII_SUCCESS;
99}

101template <typename T>
102xiiResult xiiDeduplicationWriteContext::WriteObjectInternal(xiiStreamWriter& stream, const T* pObject)
103{
104 xiiUInt32 uiIndex = xiiInvalidIndex;
105
106 if (pObject)
107 {
108 bool bIsRealObject = !m_Objects.TryGetValue(pObject, uiIndex);
109 stream << bIsRealObject;
110
111 if (bIsRealObject)
112 {
113 uiIndex = m_Objects.GetCount();
114 m_Objects.Insert(pObject, uiIndex);
115
116 return xiiStreamWriterUtil::Serialize<T>(stream, *pObject);
117 }
118 else
119 {
120 stream << uiIndex;
121 }
122 }
123 else
124 {
125 stream << false;
126 stream << xiiInvalidIndex;
127 }
128
129 return XII_SUCCESS;
130}
