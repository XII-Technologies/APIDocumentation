473 template <typename Type>
474 const xiiVec3Template<Type> xiiVec3Template<Type>::GetRefractedVector(const xiiVec3Template<Type>& vNormal, Type fRefIndex1, Type fRefIndex2) const
475 {
476  XII_ASSERT_DEBUG(vNormal.IsNormalized(), "vNormal must be normalized.");
477 
478  const Type n = fRefIndex1 / fRefIndex2;
479  const Type cosI = this->Dot(vNormal);
480  const Type sinT2 = n * n * (1.0f - (cosI * cosI));
481 
482  // invalid refraction
483  if (sinT2 > 1.0f)
484  return (*this);
485 
486  return ((n * (*this)) - (n + xiiMath::Sqrt(1.0f - sinT2)) * vNormal);
