138template <typename IdType, typename ValueType>
139void xiiIdTableBase<IdType, ValueType>::Reserve(IndexType capacity)
140{
141 if (m_Capacity >= capacity + CAPACITY_ALIGNMENT)
142 return;
143
144 const xiiUInt64 uiCurCap64 = static_cast<xiiUInt64>(this->m_Capacity);
145 xiiUInt64 uiNewCapacity64 = uiCurCap64 + (uiCurCap64 / 2);
146
147 uiNewCapacity64 = xiiMath::Max<xiiUInt64>(uiNewCapacity64, capacity + CAPACITY_ALIGNMENT);
148
149 // the maximum value must leave room for the capacity alignment computation below (without overflowing the 32 bit range)
150 uiNewCapacity64 = xiiMath::Min<xiiUInt64>(uiNewCapacity64, 0xFFFFFFFFllu - (CAPACITY_ALIGNMENT - 1));
151
152 uiNewCapacity64 = (uiNewCapacity64 + (CAPACITY_ALIGNMENT - 1)) & ~(CAPACITY_ALIGNMENT - 1);
153
154 SetCapacity(static_cast<IndexType>(uiNewCapacity64 & 0xFFFFFFFF));
155}

226template <typename IdType, typename ValueType>
227bool xiiIdTableBase<IdType, ValueType>::Remove(const IdType id, ValueType* out_pOldValue /*= nullptr*/)
228{
229 if (m_Capacity <= id.m_InstanceIndex)
230 return false;
231
232 const IndexType uiIndex = id.m_InstanceIndex;
233 Entry& entry = m_pEntries[uiIndex];
234 if (!entry.id.IsIndexAndGenerationEqual(id))
235 return false;
236
237 if (out_pOldValue != nullptr)
238 *out_pOldValue = std::move(m_pEntries[uiIndex].value);
239
240 xiiMemoryUtils::Destruct(&entry.value, 1);
241
242 entry.id.m_InstanceIndex = m_pEntries[m_FreelistEnqueue].id.m_InstanceIndex;
243 ++entry.id.m_Generation;
244
245 // at wrap around, prevent generation from becoming 0, to ensure that a zero initialized array could ever contain a valid ID
246 if (entry.id.m_Generation == 0)
247 entry.id.m_Generation = 1;
248
249 m_pEntries[m_FreelistEnqueue].id.m_InstanceIndex = static_cast<decltype(entry.id.m_InstanceIndex)>(uiIndex);
250 m_FreelistEnqueue = uiIndex;
251
252 --m_Count;
253 return true;
254}

343template <typename IdType, typename ValueType>
344bool xiiIdTableBase<IdType, ValueType>::IsFreelistValid() const
345{
346 if (m_pEntries == nullptr)
347 return true;
348
349 IndexType uiIndex = m_FreelistDequeue;
350 const Entry* pEntry = m_pEntries + uiIndex;
351
352 while (pEntry->id.m_InstanceIndex < m_Capacity)
353 {
354 uiIndex = pEntry->id.m_InstanceIndex;
355 pEntry = m_pEntries + uiIndex;
356 }
357
358 return uiIndex == m_FreelistEnqueue;
359}

364void xiiIdTableBase<IdType, ValueType>::SetCapacity(IndexType uiCapacity)
365{
366 Entry* pNewEntries = XII_NEW_RAW_BUFFER(m_pAllocator, Entry, (size_t)uiCapacity);
367
368 for (IndexType i = 0; i < m_Capacity; ++i)

379 m_pEntries = pNewEntries;
