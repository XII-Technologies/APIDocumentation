274 template <typename T>
275 XII_FORCE_INLINE bool xiiVariant::operator==(const T& other) const
276 {
277  if (IsFloatingPoint())
278  {
279  if constexpr (TypeDeduction<T>::value > Type::Invalid && TypeDeduction<T>::value <= Type::Double)
280  {
281  return ConvertNumber<double>() == static_cast<double>(other);
282  }
283 
284  return false;
285  }
286  else if (IsNumber())
287  {
288  if constexpr (TypeDeduction<T>::value > Type::Invalid && TypeDeduction<T>::value <= Type::Double)
289  {
290  return ConvertNumber<xiiInt64>() == static_cast<xiiInt64>(other);
291  }
292 
293  return false;
294  }
295 
296  if constexpr (std::is_same_v<T, xiiHashedString>)
297  {
298  if (m_uiType == Type::TempHashedString)
299  {
300  return Cast<xiiTempHashedString>() == other;
301  }
302  }
303  else if constexpr (std::is_same_v<T, xiiTempHashedString>)
304  {
305  if (m_uiType == Type::HashedString)
306  {
307  return Cast<xiiHashedString>() == other;
308  }
309  }
310 
311  using StorageType = typename TypeDeduction<T>::StorageType;
312  XII_ASSERT_DEV(IsA<StorageType>(), "Stored type '{0}' does not match comparison type '{1}'", m_uiType, TypeDeduction<T>::value);
313  return Cast<StorageType>() == other;
314 }

353 template <typename T, typename std::enable_if_t<xiiVariantTypeDeduction<T>::classification == xiiVariantClass::PointerCast, xiiInt32>>
354 XII_ALWAYS_INLINE bool xiiVariant::IsA() const
355 {
356  if (m_uiType == TypeDeduction<T>::value)
357  {
358  const xiiTypedPointer& ptr = *reinterpret_cast<const xiiTypedPointer*>(&m_Data);
359  // Always allow cast to void*.
360  if constexpr (std::is_same<T, void*>::value || std::is_same<T, const void*>::value)
361  {
362  return true;
363  }
364  else if (ptr.m_pType)
365  {
366  using NonPointerT = typename xiiTypeTraits<T>::NonConstReferencePointerType;
367  const xiiRTTI* pType = xiiGetStaticRTTI<NonPointerT>();
368  return IsDerivedFrom(ptr.m_pType, pType);
369  }
370  else if (!ptr.m_pObject)
371  {
372  // nullptr can be converted to anything
373  return true;
374  }
375  }
376  return false;
377 }

514 XII_FORCE_INLINE void xiiVariant::InitTypedObject(const T& value, xiiTraitInt<0>)
