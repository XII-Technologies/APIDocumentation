86  // Index based remapping ignores address identity and solely uses the object's parent index to define
87  // its guid. Set it to true if the native changes are complete clear and replace operations and
88  // not incremental changes to the existing data.
89  void ApplyNativePropertyChangesToObjectManager(bool bForceIndexBasedRemapping = false)
90  {
91  XII_PROFILE_SCOPE("ApplyNativePropertyChangesToObjectManager");
92  // Create object manager graph
93  xiiAbstractObjectGraph origGraph;
94  xiiAbstractObjectNode* pOrigRootNode = nullptr;
95  {
96  xiiDocumentObjectConverterWriter writer(&origGraph, this->GetObjectManager());
97  pOrigRootNode = writer.AddObjectToGraph(GetPropertyObject());
98  }
99 
100  // Create native object graph
101  xiiAbstractObjectGraph graph;
102  xiiAbstractObjectNode* pRootNode = nullptr;
103  {
104  // The xiiApplyNativePropertyChangesContext takes care of generating guids for native pointers that match those
105  // of the object manager.
106  xiiApplyNativePropertyChangesContext nativeChangesContext(m_Context, origGraph);
107  xiiRttiConverterWriter rttiConverter(&graph, &nativeChangesContext, true, true);
108  nativeChangesContext.RegisterObject(pOrigRootNode->GetGuid(), xiiGetStaticRTTI<PropertyType>(), GetProperties());
109  pRootNode = rttiConverter.AddObjectToGraph(GetProperties(), "Object");
110  }
111 
112  // Remapping is no longer necessary as xiiApplyNativePropertyChangesContext takes care of mapping to the original nodes.
113  // However, if the native changes are done like clear+rebuild everything, then no original object will be found and
114  // every pointer will be deleted and re-created. Forcing the remapping (which works entirely via index and ignores
115  // pointer addresses) will yield better results (e.g. no changes on two back-to -back transform calls).
116  if (bForceIndexBasedRemapping)
117  {
118  // Remap native guids so they match the object manager (stuff like embedded classes will not have a guid on the native side).
119  graph.ReMapNodeGuidsToMatchGraph(pRootNode, origGraph, pOrigRootNode);
120  }
121 
122  xiiDeque<xiiAbstractGraphDiffOperation> diffResult;
123  graph.CreateDiffWithBaseGraph(origGraph, diffResult);
124 
125  // if index-based remapping is used, we MUST send a change event of some kind
126  // since the underlying data structures (memory locations) might have been changed,
127  // even if there is no actual change to the content
128  // the command history will detect that there was no change and actually send a "TransactionCanceled" event, but that is enough for other code to react to
129  if (!diffResult.IsEmpty() || bForceIndexBasedRemapping)
130  {
131  // As we messed up the native side the object mirror is no longer synced and needs to be destroyed.
132  m_ObjectMirror.Clear();
133 
134  // Apply diff while object mirror is down.
135  this->GetObjectAccessor()->StartTransaction("Apply Native Property Changes to Object");
136 
137  xiiDocumentObjectConverterReader::ApplyDiffToObject(this->GetObjectAccessor(), GetPropertyObject(), diffResult);
138 
139  // Re-apply document
140  m_ObjectMirror.SendDocument();
141 
142  this->GetObjectAccessor()->FinishTransaction();
143  }
144  }
