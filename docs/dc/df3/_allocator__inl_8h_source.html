61template <typename A, xiiAllocatorTrackingMode TrackingMode>
62void* xiiInternal::xiiAllocatorImpl<A, TrackingMode>::Allocate(size_t uiSize, size_t uiAlign, xiiMemoryUtils::DestructorFunction destructorFunc)
63{
64 // zero size allocations always return nullptr without tracking (since deallocate nullptr is ignored)
65 if (uiSize == 0)
66 return nullptr;
67
68 XII_ASSERT_DEBUG(xiiMath::IsPowerOf2((xiiUInt32)uiAlign), "Alignment must be power of two");
69
70 xiiTime fAllocationTime = xiiTime::Now();
71
72 void* ptr = m_allocator.Allocate(uiSize, uiAlign);
73 XII_ASSERT_DEV(ptr != nullptr, "Could not allocate {0} bytes. Out of memory?", uiSize);
74
75 if constexpr (TrackingMode >= xiiAllocatorTrackingMode::AllocationStats)
76 {
77 xiiMemoryTracker::AddAllocation(this->m_Id, TrackingMode, ptr, uiSize, uiAlign, xiiTime::Now() - fAllocationTime);
78 }
79
80 return ptr;
81}
