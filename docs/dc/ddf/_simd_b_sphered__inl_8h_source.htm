40  XII_ASSERT_DEBUG(uiNumPoints > 0, "The array must contain at least one point.");
41 
42  const xiiSimdVec4d* pCur = pPoints;
43 
44  xiiSimdVec4d vCenter = xiiSimdVec4d::ZeroVector();
45  for (xiiUInt32 i = 0; i < uiNumPoints; ++i)
