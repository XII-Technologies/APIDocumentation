36const xiiTimestamp xiiDateTime::GetTimestamp() const
37{
38 tm timeinfo = {0};
39
40 timeinfo.tm_sec = m_uiSecond; /* seconds after the minute - [0,59] */
41 timeinfo.tm_min = m_uiMinute; /* minutes after the hour - [0,59] */
42 timeinfo.tm_hour = m_uiHour; /* hours since midnight - [0,23] */
43 timeinfo.tm_mday = m_uiDay; /* day of the month - [1,31] */
44 timeinfo.tm_mon = m_uiMonth - 1; /* months since January - [0,11] */
45 timeinfo.tm_year = m_iYear - 1900; /* years since 1900 */
46 timeinfo.tm_isdst = 0; /* daylight savings time flag */
47 timeinfo.tm_zone = "UTC";
48
49 time64_t iTimeStamp = timegm64(&timeinfo);
50 // If it can't round trip it is assumed to be invalid.
51 tm timeinfoRoundtrip = {0};
52 if (gmtime64_r(&iTimeStamp, &timeinfoRoundtrip) == nullptr)
53 return xiiTimestamp::MakeInvalid();
54
55 // mktime may have 'patched' our time to be valid, we don't want that to count as a valid date.
56 if (timeinfoRoundtrip != timeinfo)
57 return xiiTimestamp::MakeInvalid();
58
59 iTimeStamp += timeinfo.tm_gmtoff;
60 // Subtract one hour if daylight saving time was activated by mktime.
61 if (timeinfo.tm_isdst == 1)
62 iTimeStamp -= 3600;
63 return xiiTimestamp::MakeFromInt(iTimeStamp, xiiSIUnitOfTime::Second);
64}
