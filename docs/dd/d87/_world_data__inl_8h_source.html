199 XII_FORCE_INLINE bool WorldData::RegisteredUpdateFunction::operator<(const RegisteredUpdateFunction& other) const
200 {
201 // higher priority comes first
202 if (m_fPriority != other.m_fPriority)
203 return m_fPriority > other.m_fPriority;
204
205 // sort by function name to ensure determinism
206 xiiInt32 iNameComp = xiiStringUtils::Compare(m_sFunctionName, other.m_sFunctionName);
207 XII_ASSERT_DEV(iNameComp != 0, "An update function with the same name and same priority is already registered. This breaks determinism.");
208 return iNameComp < 0;
209 }

237 XII_FORCE_INLINE void WorldData::WriteMarker::Lock()
238 {
239 // already locked by this thread?
240 if (m_Data.m_WriteThreadID != xiiThreadUtils::GetCurrentThreadID())
241 {
242 XII_ASSERT_DEV(m_Data.m_iReadCounter == 0, "World '{0}' cannot be marked for writing because it is already marked for reading.", m_Data.m_sName);
243 XII_ASSERT_DEV(m_Data.m_WriteThreadID == (xiiThreadID)0,
244 "World '{0}' cannot be marked for writing because it is already marked for writing by another thread.", m_Data.m_sName);
245
246 m_Data.m_WriteThreadID = xiiThreadUtils::GetCurrentThreadID();
247 m_Data.m_iReadCounter.Increment(); // allow reading as well
248 }
249
250 m_Data.m_iWriteCounter++;
251 }
