36template <typename Class, typename Type, typename Container>
37class xiiAccessorSetProperty : public xiiTypedSetProperty<Type>
38{
39public:
40 using ContainerType = typename xiiTypeTraits<Container>::NonConstReferenceType;
41 using RealType = typename xiiTypeTraits<Type>::NonConstReferenceType;
42
43 using InsertFunc = void (Class::*)(Type value);
44 using RemoveFunc = void (Class::*)(Type value);
45 using GetValuesFunc = Container (Class::*)() const;
46
47 xiiAccessorSetProperty(xiiStringView sPropertyName, GetValuesFunc getValues, InsertFunc insert, RemoveFunc remove) :
48 xiiTypedSetProperty<Type>(sPropertyName)
49 {
50 XII_ASSERT_DEBUG(getValues != nullptr, "The get values function of an set property cannot be nullptr.");
51
52 m_GetValues = getValues;
53 m_Insert = insert;
54 m_Remove = remove;
55
56 if (m_Insert == nullptr || m_Remove == nullptr)
57 xiiAbstractSetProperty::m_Flags.Add(xiiPropertyFlags::ReadOnly);
58 }
59
60
61 virtual bool IsEmpty(const void* pInstance) const override { return (static_cast<const Class*>(pInstance)->*m_GetValues)().IsEmpty(); }
62
63 virtual void Clear(void* pInstance) const override
64 {
65 XII_ASSERT_DEBUG(m_Insert != nullptr && m_Remove != nullptr, "The property '{0}' has no remove and insert function, thus it is read-only",
66 xiiAbstractProperty::GetPropertyName());
67
68 // We must not cache the container c here as the Remove can make it invalid
69 // e.g. xiiArrayPtr by value.
70 while (!IsEmpty(pInstance))
71 {
72 // This should be decltype(auto) c = ...; but MSVC 16 does not deduce that (MSVC 15 works fine)
73 decltype((static_cast<const Class*>(pInstance)->*m_GetValues)()) c = (static_cast<const Class*>(pInstance)->*m_GetValues)();
74 auto it = cbegin(c);
75 RealType value = *it;
76 Remove(pInstance, &value);
77 }
78 }
79
80 virtual void Insert(void* pInstance, const void* pObject) const override
81 {
82 XII_ASSERT_DEBUG(m_Insert != nullptr, "The property '{0}' has no insert function, thus it is read-only.", xiiAbstractProperty::GetPropertyName());
83
84 (static_cast<Class*>(pInstance)->*m_Insert)(*static_cast<const RealType*>(pObject));
85 }
86
87 virtual void Remove(void* pInstance, const void* pObject) const override
88 {
89 XII_ASSERT_DEBUG(m_Remove != nullptr, "The property '{0}' has no setter function, thus it is read-only.", xiiAbstractProperty::GetPropertyName());
90
91 (static_cast<Class*>(pInstance)->*m_Remove)(*static_cast<const RealType*>(pObject));
92 }
93
94 virtual bool Contains(const void* pInstance, const void* pObject) const override
95 {
96 for (const auto& value : (static_cast<const Class*>(pInstance)->*m_GetValues)())
97 {
98 if (value == *static_cast<const RealType*>(pObject))
99 return true;
100 }
101 return false;
102 }
103
104 virtual void GetValues(const void* pInstance, xiiDynamicArray<xiiVariant>& out_keys) const override
105 {
106 out_keys.Clear();
107 for (const auto& value : (static_cast<const Class*>(pInstance)->*m_GetValues)())
108 {
109 out_keys.PushBack(xiiVariant(value));
110 }
111 }
112
113private:
114 GetValuesFunc m_GetValues;
115 InsertFunc m_Insert;
116 RemoveFunc m_Remove;
117};
