67template <typename EventData, typename MutexType, xiiEventType EventType>
68void xiiEventBase<EventData, MutexType, EventType>::RemoveEventHandler(const Handler& handler) const
69{
70 XII_ASSERT_DEV(handler.IsComparable(), "Lambdas that capture data cannot be removed via function pointer. Use a xiiEventSubscriptionID instead.");
71
72 XII_LOCK(m_Mutex);
73
74 if constexpr (EventType == xiiEventType::Default)
75 {
76 if constexpr (std::is_same_v<MutexType, xiiNoMutex>)
77 {
78 XII_ASSERT_DEV(m_uiRecursionDepth == 0, "Can't add or remove event handlers while broadcasting (without a mutex). Either enable the use of a mutex on this event, or switch to xiiCopyOnBroadcastEvent if this should be allowed. Since this event does not have a mutex, this error can also happen due to multi-threaded access.");
79 }
80 }
81
82 for (xiiUInt32 idx = 0; idx < m_EventHandlers.GetCount(); ++idx)
83 {
84 if (m_EventHandlers[idx].m_Handler.IsEqualIfComparable(handler))
85 {
86 if constexpr (EventType == xiiEventType::Default)
87 {
88 // if this event does not copy the handlers, and we are currently broadcasting
89 // we can't shrink the size of the array, however, we can replace elements (the check above says that we have a mutex, so this is fine)
90
91 if (m_uiRecursionDepth > 0)
92 {
93 // we just write an invalid handler here, and let the broadcast function clean it up for us
94 m_EventHandlers[idx].m_Handler = {};
95 m_EventHandlers[idx].m_SubscriptionID = {};
96 return;
97 }
98 }
99
100 // if we are not broadcasting, or the broadcast uses a copy anyway, we can just modify the handler array directly
101 m_EventHandlers.RemoveAtAndCopy(idx);
102 return;
103 }
104 }
105
106 XII_ASSERT_DEV(false, "xiiEvent::RemoveEventHandler: Handler has not been registered or already been unregistered.");
107}

188template <typename EventData, typename MutexType, xiiEventType EventType>
189void xiiEventBase<EventData, MutexType, EventType>::Broadcast(EventData eventData, xiiUInt8 uiMaxRecursionDepth)
190{
191 if constexpr (EventType == xiiEventType::Default)
192 {
193 XII_LOCK(m_Mutex);
194
195 XII_ASSERT_DEV(m_uiRecursionDepth <= uiMaxRecursionDepth, "The event has been triggered recursively or from several threads simultaneously.");
196
197 if (m_uiRecursionDepth > uiMaxRecursionDepth)
198 return;
199
200#if XII_ENABLED(XII_COMPILE_FOR_DEVELOPMENT)
201 XII_ASSERT_ALWAYS(m_pSelf == this, "The xiiEvent was relocated in memory. This is not allowed, as it breaks the Unsubscribers.");
202#endif
203
204 m_uiRecursionDepth++;
205
206 // RAII to ensure correctness in case exceptions are used
207 auto scopeExit = xiiMakeScopeExit([&]() {
208 m_uiRecursionDepth--;
209 });
210
211 // don't execute handlers that are added while we are broadcasting
212 xiiUInt32 uiMaxHandlers = m_EventHandlers.GetCount();
213
214 for (xiiUInt32 ui = 0; ui < uiMaxHandlers;)
215 {
216 if (m_EventHandlers[ui].m_Handler.IsValid())
217 {
218 m_EventHandlers[ui].m_Handler(eventData);
219 ++ui;
220 }
221 else
222 {
223 m_EventHandlers.RemoveAtAndCopy(ui);
224 --uiMaxHandlers;
225 }
226 }
227 }
228 else
229 {
230 xiiHybridArray<HandlerData, 16> eventHandlers;
231 {
232 XII_LOCK(m_Mutex);
233
234 if constexpr (RecursionDepthSupported)
235 {
236 XII_ASSERT_DEV(m_uiRecursionDepth <= uiMaxRecursionDepth, "The event has been triggered recursively or from several threads simultaneously.");
237
238 if (m_uiRecursionDepth > uiMaxRecursionDepth)
239 return;
240
241#if XII_ENABLED(XII_COMPILE_FOR_DEVELOPMENT)
242 XII_ASSERT_ALWAYS(m_pSelf == this, "The xiiEvent was relocated in memory. This is not allowed, as it breaks the Unsubscribers.");
243#endif
244
245 m_uiRecursionDepth++;
246 }
247 else
248 {
249 XII_ASSERT_DEV(uiMaxRecursionDepth == 255, "uiMaxRecursionDepth is not supported if xiiEventType::CopyOnBroadcast is used and the event needs to be threadsafe.");
250 }
251
252#if XII_ENABLED(XII_COMPILE_FOR_DEVELOPMENT)
253 XII_ASSERT_ALWAYS(m_pSelf == this, "The xiiEvent was relocated in memory. This is not allowed, as it breaks the Unsubscribers.");
254#endif
255
256 eventHandlers = m_EventHandlers;
257 }
258
259 // RAII to ensure correctness in case exceptions are used
260 auto scopeExit = xiiMakeScopeExit([&]() {
261 // Bug in MSVC 2017. Can't use if constexpr.
262#if XII_ENABLED(XII_COMPILER_MSVC) && _MSC_VER < 1920
263 if (RecursionDepthSupported)
264 {
265 m_uiRecursionDepth--;
266 }
267#else
268 if constexpr (RecursionDepthSupported)
269 {
270 m_uiRecursionDepth--;
271 }
272#endif
273 });
274
275 const xiiUInt32 uiHandlerCount = eventHandlers.GetCount();
276 for (xiiUInt32 ui = 0; ui < uiHandlerCount; ++ui)
277 {
278 eventHandlers[ui].m_Handler(eventData);
279 }
280 }
281}
