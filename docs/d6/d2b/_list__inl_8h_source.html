54{
55 ListElement* pNode;
56
57 if (m_pFreeElementStack == nullptr)
58 {
59 m_Elements.PushBack();
60 pNode = &m_Elements.PeekBack();
61 }

64 pNode = m_pFreeElementStack;

72template <typename T>
73void xiiListBase<T>::ReleaseNode(ListElement* pNode)
74{
75 xiiMemoryUtils::Destruct<ListElement>(pNode, 1);
76
77 if (pNode == &m_Elements.PeekBack())
78 {
79 m_Elements.PopBack();
80 }
81 else if (pNode == &m_Elements.PeekFront())
82 {
83 m_Elements.PopFront();
84 }
85 else

87 pNode->m_pNext = m_pFreeElementStack;
88 m_pFreeElementStack = pNode;
