323 template <typename Type>
324 Type xiiBoundingBoxTemplate<Type>::GetDistanceSquaredTo(const xiiBoundingBoxTemplate<Type>& rhs) const
325 {
326  // This will return zero for overlapping boxes
327 
328  XII_NAN_ASSERT(this);
329  XII_NAN_ASSERT(&rhs);
330 
331  Type fDistSQR = 0.0f;
332 
333  {
334  if (rhs.m_vMin.x > m_vMax.x)
335  {
336  fDistSQR += xiiMath::Square(rhs.m_vMin.x - m_vMax.x);
337  }
338  else if (rhs.m_vMax.x < m_vMin.x)
339  {
340  fDistSQR += xiiMath::Square(m_vMin.x - rhs.m_vMax.x);
341  }
342  }
343 
344  {
345  if (rhs.m_vMin.y > m_vMax.y)
346  {
347  fDistSQR += xiiMath::Square(rhs.m_vMin.y - m_vMax.y);
348  }
349  else if (rhs.m_vMax.y < m_vMin.y)
350  {
351  fDistSQR += xiiMath::Square(m_vMin.y - rhs.m_vMax.y);
352  }
353  }
354 
355  {
356  if (rhs.m_vMin.z > m_vMax.z)
357  {
358  fDistSQR += xiiMath::Square(rhs.m_vMin.z - m_vMax.z);
359  }
360  else if (rhs.m_vMax.z < m_vMin.z)
361  {
362  fDistSQR += xiiMath::Square(m_vMin.z - rhs.m_vMax.z);
363  }
364  }
365 
366  return fDistSQR;
367 }

375 template <typename Type>
376 bool xiiBoundingBoxTemplate<Type>::GetRayIntersection(const xiiVec3Template<Type>& vStartPos, const xiiVec3Template<Type>& vRayDir, Type* out_pIntersectionDistance, xiiVec3Template<Type>* out_pIntersection) const
377 {
378  // This code was taken from: http://people.csail.mit.edu/amy/papers/box-jgt.pdf
379  // "An Efficient and Robust Ray-Box Intersection Algorithm"
380  // Contrary to previous implementation, this one actually works with ray/box configurations
381  // that produce division by zero and multiplication with infinity (which can produce NaNs).
382 
383  XII_ASSERT_DEBUG(xiiMath::SupportsInfinity<Type>(), "This type does not support infinite values, which is required for this algorithm.");
384  XII_ASSERT_DEBUG(vStartPos.IsValid(), "Ray start position must be valid.");
385  XII_ASSERT_DEBUG(vRayDir.IsValid(), "Ray direction must be valid.");
386 
387  XII_NAN_ASSERT(this);
388 
389  float tMin, tMax;
390 
391  // Compare along X and Z axis, find intersection point
392  {
393  float tMinY, tMaxY;
394 
395  const float fDivX = 1.0f / vRayDir.x;
396  const float fDivY = 1.0f / vRayDir.y;
397 
398  if (vRayDir.x >= 0.0f)
399  {
400  tMin = (m_vMin.x - vStartPos.x) * fDivX;
401  tMax = (m_vMax.x - vStartPos.x) * fDivX;
402  }
403  else
404  {
405  tMin = (m_vMax.x - vStartPos.x) * fDivX;
406  tMax = (m_vMin.x - vStartPos.x) * fDivX;
407  }
408 
409  if (vRayDir.y >= 0.0f)
410  {
411  tMinY = (m_vMin.y - vStartPos.y) * fDivY;
412  tMaxY = (m_vMax.y - vStartPos.y) * fDivY;
413  }
414  else
415  {
416  tMinY = (m_vMax.y - vStartPos.y) * fDivY;
417  tMaxY = (m_vMin.y - vStartPos.y) * fDivY;
418  }
419 
420  if (tMin > tMaxY || tMinY > tMax)
421  return false;
422 
423  if (tMinY > tMin)
424  tMin = tMinY;
425  if (tMaxY < tMax)
426  tMax = tMaxY;
427  }
428 
429  // Compare along Z axis and previous result, find intersection point
430  {
431  float tMinZ, tMaxZ;
432 
433  const float fDivZ = 1.0f / vRayDir.z;
434 
435  if (vRayDir.z >= 0.0f)
436  {
437  tMinZ = (m_vMin.z - vStartPos.z) * fDivZ;
438  tMaxZ = (m_vMax.z - vStartPos.z) * fDivZ;
439  }
440  else
441  {
442  tMinZ = (m_vMax.z - vStartPos.z) * fDivZ;
443  tMaxZ = (m_vMin.z - vStartPos.z) * fDivZ;
444  }
445 
446  if (tMin > tMaxZ || tMinZ > tMax)
447  return false;
448 
449  if (tMinZ > tMin)
450  tMin = tMinZ;
451  if (tMaxZ < tMax)
452  tMax = tMaxZ;
453  }
454 
455  // rays that start inside the box are considered as not hitting the box
456  if (tMax <= 0.0f)
457  return false;
458 
459  if (out_pIntersectionDistance)
460  *out_pIntersectionDistance = tMin;
461 
462  if (out_pIntersection)
463  *out_pIntersection = vStartPos + tMin * vRayDir;
464 
465  return true;
466 }
