321 {
322 const Type divider = edge2 - edge1;
323
324 if (divider == (Type)0)
325 {

451 template <typename T, typename T2>
452 T EvaluateBezierCurve(T2 t, const T& startPoint, const T& controlPoint1, const T& controlPoint2, const T& endPoint)
453 {
454 const T2 mt = 1 - t;
455
456 const T2 f1 = mt * mt * mt;
457 const T2 f2 = 3 * mt * mt * t;
458 const T2 f3 = 3 * mt * t * t;
459 const T2 f4 = t * t * t;
460
461 return f1 * startPoint + f2 * controlPoint1 + f3 * controlPoint2 + f4 * endPoint;
462 }
