99inline void xiiSimdBBoxSphered::ExpandToInclude(const xiiSimdBBoxSphered& rhs)
100{
101 xiiSimdBBoxd box = GetBox();
102 box.ExpandToInclude(rhs.GetBox());
103
104 xiiSimdVec4d center = box.GetCenter();
105 xiiSimdVec4d boxHalfExtents = center - box.m_Min;
106 xiiSimdDouble tmpRadius = boxHalfExtents.GetLength<3>();
107
108 const xiiSimdDouble fSphereRadiusA = (m_CenterAndRadius - center).GetLength<3>() + m_CenterAndRadius.w();
109 const xiiSimdDouble fSphereRadiusB = (rhs.m_CenterAndRadius - center).GetLength<3>() + rhs.m_CenterAndRadius.w();
110
111 m_CenterAndRadius = center;
112 m_CenterAndRadius.SetW(tmpRadius.Min(fSphereRadiusA.Max(fSphereRadiusB)));
113 m_BoxHalfExtents = boxHalfExtents;
114}

121XII_ALWAYS_INLINE void xiiSimdBBoxSphered::Transform(const xiiSimdMat4d& mMat)
122{
123 xiiSimdDouble radius = m_CenterAndRadius.w();
124 m_CenterAndRadius = mMat.TransformPosition(m_CenterAndRadius);
125
126 xiiSimdDouble maxRadius = mMat.m_col0.Dot<3>(mMat.m_col0);
127 maxRadius = maxRadius.Max(mMat.m_col1.Dot<3>(mMat.m_col1));
128 maxRadius = maxRadius.Max(mMat.m_col2.Dot<3>(mMat.m_col2));
129 radius *= maxRadius.GetSqrt();
130
131 m_CenterAndRadius.SetW(radius);
132
133 xiiSimdVec4d newHalfExtents = mMat.m_col0.Abs() * m_BoxHalfExtents.x();
134 newHalfExtents += mMat.m_col1.Abs() * m_BoxHalfExtents.y();
135 newHalfExtents += mMat.m_col2.Abs() * m_BoxHalfExtents.z();
136
137 m_BoxHalfExtents = newHalfExtents.CompMin(xiiSimdVec4d(radius));
138}
