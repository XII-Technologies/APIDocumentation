201 template <typename K, typename H>
202 void xiiHashSetBase<K, H>::Reserve(xiiUInt32 uiCapacity)
203 {
204  const xiiUInt64 uiCap64 = static_cast<xiiUInt64>(uiCapacity);
205  xiiUInt64 uiNewCapacity64 = uiCap64 + (uiCap64 * 2 / 3); // ensure a maximum load of 60%
206 
207  uiNewCapacity64 = xiiMath::Min<xiiUInt64>(uiNewCapacity64, 0x80000000llu); // the largest power-of-two in 32 bit
208 
209  xiiUInt32 uiNewCapacity32 = static_cast<xiiUInt32>(uiNewCapacity64 & 0xFFFFFFFF);
210  XII_ASSERT_DEBUG(uiCapacity <= uiNewCapacity32, "xiiHashSet/Map do not support more than 2 billion entries.");
211 
212  if (m_uiCapacity >= uiNewCapacity32)
213  return;
214 
215  uiNewCapacity32 = xiiMath::Max<xiiUInt32>(xiiMath::PowerOfTwo_Ceil(uiNewCapacity32), CAPACITY_ALIGNMENT);
216  SetCapacity(uiNewCapacity32);
217 }

264 template <typename K, typename H>
265 template <typename CompatibleKeyType>
266 bool xiiHashSetBase<K, H>::Insert(CompatibleKeyType&& key)
267 {
268  Reserve(m_uiCount + 1);
269 
270  xiiUInt32 uiIndex = H::Hash(key) & (m_uiCapacity - 1);
271  xiiUInt32 uiDeletedIndex = xiiInvalidIndex;
272 
273  xiiUInt32 uiCounter = 0;
274  while (!IsFreeEntry(uiIndex) && uiCounter < m_uiCapacity)
275  {
276  if (IsDeletedEntry(uiIndex))
277  {
278  if (uiDeletedIndex == xiiInvalidIndex)
279  uiDeletedIndex = uiIndex;
280  }
281  else if (H::Equal(m_pEntries[uiIndex], key))
282  {
283  return true;
284  }
285  ++uiIndex;
286  if (uiIndex == m_uiCapacity)
287  uiIndex = 0;
288 
289  ++uiCounter;
290  }
291 
292  // new entry
293  uiIndex = uiDeletedIndex != xiiInvalidIndex ? uiDeletedIndex : uiIndex;
294 
295  // Constructions might either be a move or a copy.
296  xiiMemoryUtils::CopyOrMoveConstruct(&m_pEntries[uiIndex], std::forward<CompatibleKeyType>(key));
297 
298  MarkEntryAsValid(uiIndex);
299  ++m_uiCount;
300 
301  return false;
302 }

329 template <typename K, typename H>
330 void xiiHashSetBase<K, H>::RemoveInternal(xiiUInt32 uiIndex)
331 {
332  xiiMemoryUtils::Destruct(&m_pEntries[uiIndex], 1);
333 
334  xiiUInt32 uiNextIndex = uiIndex + 1;
335  if (uiNextIndex == m_uiCapacity)
336  uiNextIndex = 0;
337 
338  // if the next entry is free we are at the end of a chain and
339  // can immediately mark this entry as free as well
340  if (IsFreeEntry(uiNextIndex))
341  {
342  MarkEntryAsFree(uiIndex);
343 
344  // run backwards and free all deleted entries in this chain
345  xiiUInt32 uiPrevIndex = (uiIndex != 0) ? uiIndex : m_uiCapacity;
346  --uiPrevIndex;
347 
348  while (IsDeletedEntry(uiPrevIndex))
349  {
350  MarkEntryAsFree(uiPrevIndex);
351 
352  if (uiPrevIndex == 0)
353  uiPrevIndex = m_uiCapacity;
354  --uiPrevIndex;
355  }
356  }
357  else
358  {
359  MarkEntryAsDeleted(uiIndex);
360  }
361 
362  --m_uiCount;
363 }

482 inline xiiUInt32 xiiHashSetBase<K, H>::FindEntry(xiiUInt32 uiHash, const CompatibleKeyType& key) const
483 {
484  if (m_uiCapacity > 0)
485  {
486  xiiUInt32 uiIndex = uiHash & (m_uiCapacity - 1);
487  xiiUInt32 uiCounter = 0;
488  while (!IsFreeEntry(uiIndex) && uiCounter < m_uiCapacity)
489  {
