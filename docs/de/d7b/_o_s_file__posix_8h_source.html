40xiiResult xiiOSFile::InternalOpen(xiiStringView sFile, xiiFileOpenMode::Enum OpenMode, xiiFileShareMode::Enum FileShareMode)
41{
42 xiiStringBuilder sFileCopy = sFile;
43 const char* szFile = sFileCopy;
44
45 xiiInt32 fd = -1;
46 switch (OpenMode)
47 {
48 // O_CLOEXEC = don't forward to child processes
49 case xiiFileOpenMode::Read:
50 fd = open(szFile, O_RDONLY | O_CLOEXEC);
51 break;
52 case xiiFileOpenMode::Write:
53 case xiiFileOpenMode::Append:
54 fd = open(szFile, O_CREAT | O_WRONLY | O_CLOEXEC, 0644);
55 break;
56 default:
57 break;
58 }
59
60 if (FileShareMode == xiiFileShareMode::Default)
61 {
62 if (OpenMode == xiiFileOpenMode::Read)
63 {
64 FileShareMode = xiiFileShareMode::SharedReads;
65 }
66 else
67 {
68 FileShareMode = xiiFileShareMode::Exclusive;
69 }
70 }
71
72 if (fd == -1)
73 {
74 return XII_FAILURE;
75 }
76
77 const xiiInt32 iSharedMode = (FileShareMode == xiiFileShareMode::Exclusive) ? LOCK_EX : LOCK_SH;
78 const xiiTime sleepTime = xiiTime::MakeFromMilliseconds(20);
79 xiiInt32 iRetries = m_bRetryOnSharingViolation ? 20 : 1;
80
81 while (flock(fd, iSharedMode | LOCK_NB /* do not block */) != 0)
82 {
83 xiiInt32 errorCode = errno;
84 iRetries--;
85 if (iRetries == 0 || errorCode != EWOULDBLOCK)
86 {
87 // error, could not get a lock
88 xiiLog::Error("Failed to get a {} lock for file {}, error {}", (FileShareMode == xiiFileShareMode::Exclusive) ? "Exculsive" : "Shared", szFile, errno);
89 close(fd);
90 return XII_FAILURE;
91 }
92 xiiThreadUtils::Sleep(sleepTime);
93 }
94
95 switch (OpenMode)
96 {
97 case xiiFileOpenMode::Read:
98 m_FileData.m_pFileHandle = fdopen(fd, "rb");
99 break;
100 case xiiFileOpenMode::Write:
101 if (ftruncate(fd, 0) < 0)
102 {
103 close(fd);
104 return XII_FAILURE;
105 }
106 m_FileData.m_pFileHandle = fdopen(fd, "wb");
107 break;
108 case xiiFileOpenMode::Append:
109 m_FileData.m_pFileHandle = fdopen(fd, "ab");
110
111 // in append mode we need to set the file pointer to the end explicitly, otherwise GetFilePosition might return 0 the first time
112 if (m_FileData.m_pFileHandle != nullptr)
113 InternalSetFilePosition(0, xiiFileSeekMode::FromEnd);
114
115 break;
116 default:
117 break;
118 }
119
120 if (m_FileData.m_pFileHandle == nullptr)
121 {
122 close(fd);
123 }
124
125 if (m_FileData.m_pFileHandle == nullptr)
126 {
127 return XII_FAILURE;
128 }
129
130 // lock will be released automatically when the file is closed
131 return XII_SUCCESS;
132}

139xiiResult xiiOSFile::InternalWrite(const void* pBuffer, xiiUInt64 uiBytes)
140{
141 const xiiUInt32 uiBatchBytes = 1024 * 1024 * 1024; // 1 GB
142
143 // first write out all the data in 1GB batches
144 while (uiBytes > uiBatchBytes)
145 {
146 if (fwrite(pBuffer, 1, uiBatchBytes, m_FileData.m_pFileHandle) != uiBatchBytes)
147 {
148 xiiLog::Error("fwrite 1GB failed for '{}'", m_sFileName);
149 return XII_FAILURE;
150 }
151
152 uiBytes -= uiBatchBytes;
153 pBuffer = xiiMemoryUtils::AddByteOffset(pBuffer, uiBatchBytes);
154 }
155
156 if (uiBytes > 0)
157 {
158 const xiiUInt32 uiBytes32 = static_cast<xiiUInt32>(uiBytes);
159
160 if (fwrite(pBuffer, 1, uiBytes32, m_FileData.m_pFileHandle) != uiBytes)
161 {
162 xiiLog::Error("fwrite failed for '{}'", m_sFileName);
163 return XII_FAILURE;
164 }
165 }
166
167 return XII_SUCCESS;
168}

170xiiUInt64 xiiOSFile::InternalRead(void* pBuffer, xiiUInt64 uiBytes)
171{
172 xiiUInt64 uiBytesRead = 0;
173
174 const xiiUInt32 uiBatchBytes = 1024 * 1024 * 1024; // 1 GB
175
176 // first write out all the data in 1GB batches
177 while (uiBytes > uiBatchBytes)
178 {
179 const xiiUInt64 uiReadThisTime = fread(pBuffer, 1, uiBatchBytes, m_FileData.m_pFileHandle);
180 uiBytesRead += uiReadThisTime;
181
182 if (uiReadThisTime != uiBatchBytes)
183 return uiBytesRead;
184
185 uiBytes -= uiBatchBytes;
186 pBuffer = xiiMemoryUtils::AddByteOffset(pBuffer, uiBatchBytes);
187 }
188
189 if (uiBytes > 0)
190 {
191 const xiiUInt32 uiBytes32 = static_cast<xiiUInt32>(uiBytes);
192
193 uiBytesRead += fread(pBuffer, 1, uiBytes32, m_FileData.m_pFileHandle);
194 }
195
196 return uiBytesRead;
197}

284xiiResult xiiOSFile::InternalCreateDirectory(xiiStringView sDirectory)
285{
286 // handle drive letters as always successful
287 if (xiiStringUtils::GetCharacterCount(sDirectory.GetStartPointer(), sDirectory.GetEndPointer()) <= 1) // '/'
288 return XII_SUCCESS;
289
290#if XII_ENABLED(XII_PLATFORM_WINDOWS)
291 xiiInt32 iRes = _mkdir(xiiString(sDirectory));
292#else
293 xiiInt32 iRes = mkdir(xiiString(sDirectory), 0777);
294#endif
295
296 if (iRes == 0 || (iRes == -1 && errno == EEXIST))
297 return XII_SUCCESS;
298
299 // If we were not allowed to access the folder but it alreay exists, we treat the operation as successful.
300 // Note that this is espcially relevant for calls to xiiOSFile::CreateDirectoryStructure where we may call mkdir on top level directories that are
301 // not accessible.
302 if (errno == EACCES && InternalExistsDirectory(sDirectory))
303 return XII_SUCCESS;
304
305 return XII_FAILURE;
306}

478 xiiResult UpdateCurrentFile(xiiFileStats& curFile, const xiiStringBuilder& curPath, DIR* hSearch, const xiiString& wildcardSearch)
479 {
480 struct dirent* hCurrentFile = readdir(hSearch);
481 if (hCurrentFile == nullptr)
482 return XII_FAILURE;
483
484 if (!wildcardSearch.IsEmpty())
485 {
486 while (fnmatch(wildcardSearch.GetData(), hCurrentFile->d_name, FNM_NOESCAPE) != 0)
487 {
488 hCurrentFile = readdir(hSearch);
489 if (hCurrentFile == nullptr)
490 return XII_FAILURE;
491 }
492 }
493
494 xiiStringBuilder absFileName = curPath;
495 absFileName.AppendPath(hCurrentFile->d_name);
496
497 struct stat fileStat = {};
498 stat(absFileName.GetData(), &fileStat);
499
500 curFile.m_uiFileSize = fileStat.st_size;
501 curFile.m_bIsDirectory = hCurrentFile->d_type == DT_DIR;
502 curFile.m_sParentPath = curPath;
503 curFile.m_sName = hCurrentFile->d_name;
504 curFile.m_LastModificationTime = xiiTimestamp::MakeFromInt(fileStat.st_mtime, xiiSIUnitOfTime::Second);
505
506 return XII_SUCCESS;
507 }

510void xiiFileSystemIterator::StartSearch(xiiStringView sSearchTerm, xiiBitflags<xiiFileSystemIteratorFlags> flags /*= xiiFileSystemIteratorFlags::All*/)
511{
512 XII_ASSERT_DEV(m_Data.m_Handles.IsEmpty(), "Cannot start another search.");
513
514 m_sSearchTerm = sSearchTerm;
515
516 xiiStringBuilder sSearch = sSearchTerm;
517 sSearch.MakeCleanPath();
518
519 // same as just passing in the folder path, so remove this
520 if (sSearch.EndsWith("/*"))
521 sSearch.Shrink(0, 2);
522
523 // Remove a trailing slash if any
524 sSearch.Trim(nullptr, "/");
525
526 // Since the use of wildcard-ed file names will disable recursion, we ensure both are not used simultaneously.
527 const bool bHasWildcard = sSearch.FindLastSubString("*") || sSearch.FindLastSubString("?");
528 if (flags.IsSet(xiiFileSystemIteratorFlags::Recursive) == true && bHasWildcard == true)
529 {
530 XII_ASSERT_DEV(false, "Recursive file iteration does not support wildcards. Either don't use recursion, or filter the filenames manually.");
531 return;
532 }
533
534 if (bHasWildcard)
535 {
536 m_Data.m_wildcardSearch = sSearch.GetFileNameAndExtension();
537 m_sCurPath = sSearch.GetFileDirectory();
538 }
539 else
540 {
541 m_Data.m_wildcardSearch.Clear();
542 m_sCurPath = sSearch;
543 }
544
545 XII_ASSERT_DEV(m_sCurPath.IsAbsolutePath(), "The path '{0}' is not absolute.", m_sCurPath);
546
547 m_Flags = flags;
548
549 DIR* hSearch = opendir(m_sCurPath.GetData());
550
551 if (hSearch == nullptr)
552 return;
553
554 if (UpdateCurrentFile(m_CurFile, m_sCurPath, hSearch, m_Data.m_wildcardSearch).Failed())
555 {
556 return;
557 }
558
559 m_Data.m_Handles.PushBack(hSearch);
560
561 if ((m_CurFile.m_sName == "..") || (m_CurFile.m_sName == "."))
562 {
563 Next(); // will search for the next file or folder that is not ".." or "." ; might return false though
564 return;
565 }
566
567 if (m_CurFile.m_bIsDirectory)
568 {
569 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFolders))
570 {
571 Next();
572 return;
573 }
574 }
575 else
576 {
577 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFiles))
578 {
579 Next();
580 return;
581 }
582 }
583}

585xiiInt32 xiiFileSystemIterator::InternalNext()
586{
587 constexpr xiiInt32 CallInternalNext = 2;
588
589 if (m_Data.m_Handles.IsEmpty())
590 return XII_FAILURE;
591
592 if (m_Flags.IsSet(xiiFileSystemIteratorFlags::Recursive) && m_CurFile.m_bIsDirectory && (m_CurFile.m_sName != "..") && (m_CurFile.m_sName != "."))
593 {
594 m_sCurPath.AppendPath(m_CurFile.m_sName.GetData());
595
596 DIR* hSearch = opendir(m_sCurPath.GetData());
597
598 if (hSearch != nullptr && UpdateCurrentFile(m_CurFile, m_sCurPath, hSearch, m_Data.m_wildcardSearch).Succeeded())
599 {
600 m_Data.m_Handles.PushBack(hSearch);
601
602 if ((m_CurFile.m_sName == "..") || (m_CurFile.m_sName == "."))
603 return CallInternalNext; // will search for the next file or folder that is not ".." or "." ; might return false though
604
605 if (m_CurFile.m_bIsDirectory)
606 {
607 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFolders))
608 return CallInternalNext;
609 }
610 else
611 {
612 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFiles))
613 return CallInternalNext;
614 }
615
616 return XII_SUCCESS;
617 }
618
619 // if the recursion did not work, just iterate in this folder further
620 }
621
622 if (UpdateCurrentFile(m_CurFile, m_sCurPath, (DIR*)m_Data.m_Handles.PeekBack(), m_Data.m_wildcardSearch).Failed())
623 {
624 // nothing found in this directory anymore
625 closedir((DIR*)m_Data.m_Handles.PeekBack());
626 m_Data.m_Handles.PopBack();
627
628 if (m_Data.m_Handles.IsEmpty())
629 return XII_FAILURE;
630
631 m_sCurPath.PathParentDirectory();
632 if (m_sCurPath.GetElementCount() > 1 && m_sCurPath.EndsWith("/"))
633 {
634 m_sCurPath.Shrink(0, 1);
635 }
636
637 return CallInternalNext;
638 }
639
640 if ((m_CurFile.m_sName == "..") || (m_CurFile.m_sName == "."))
641 return CallInternalNext;
642
643 if (m_CurFile.m_bIsDirectory)
644 {
645 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFolders))
646 return CallInternalNext;
647 }
648 else
649 {
650 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFiles))
651 return CallInternalNext;
652 }
653
654 return XII_SUCCESS;
655}
