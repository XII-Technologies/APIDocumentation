57template <class Container>
58bool xiiBitfield<Container>::IsAnyBitSet(xiiUInt32 uiFirstBit /*= 0*/, xiiUInt32 uiNumBits /*= 0xFFFFFFFF*/) const
59{
60 if (m_uiCount == 0 || uiNumBits == 0)
61 return false;
62
63 XII_ASSERT_DEBUG(uiFirstBit < m_uiCount, "Cannot access bit {0}, the bitfield only has {1} bits.", uiFirstBit, m_uiCount);
64
65 const xiiUInt32 uiLastBit = xiiMath::Min<xiiUInt32>(uiFirstBit + uiNumBits, m_uiCount - 1);
66
67 const xiiUInt32 uiFirstInt = GetBitInt(uiFirstBit);
68 const xiiUInt32 uiLastInt = GetBitInt(uiLastBit);
69
70 // all within the same int
71 if (uiFirstInt == uiLastInt)
72 {
73 for (xiiUInt32 i = uiFirstBit; i <= uiLastBit; ++i)
74 {
75 if (IsBitSet(i))
76 return true;
77 }
78 }
79 else
80 {
81 const xiiUInt32 uiNextIntBit = (uiFirstInt + 1) * 32;
82 const xiiUInt32 uiPrevIntBit = uiLastInt * 32;
83
84 // check the bits in the first int individually
85 for (xiiUInt32 i = uiFirstBit; i < uiNextIntBit; ++i)
86 {
87 if (IsBitSet(i))
88 return true;
89 }
90
91 // check the bits in the ints in between with one operation
92 for (xiiUInt32 i = uiFirstInt + 1; i < uiLastInt; ++i)
93 {
94 if ((m_Container[i] & 0xFFFFFFFF) != 0)
95 return true;
96 }
97
98 // check the bits in the last int individually
99 for (xiiUInt32 i = uiPrevIntBit; i <= uiLastBit; ++i)
100 {
101 if (IsBitSet(i))
102 return true;
103 }
104 }
105
106 return false;
107}

376template <class Container>
377void xiiBitfield<Container>::ConstIterator::FindNextChunk(xiiUInt32 uiStartChunk)
378{
379 if (uiStartChunk < m_pBitfield->m_Container.GetCount())
380 {
381 const xiiUInt32 uiLastChunk = m_pBitfield->m_Container.GetCount() - 1;
382 for (xiiUInt32 i = uiStartChunk; i < uiLastChunk; ++i)
383 {
384 if (m_pBitfield->m_Container[i] != 0)
385 {
386 m_uiChunk = i;
387 m_Iterator = sub_iterator(m_pBitfield->m_Container[i]);
388 return;
389 }
390 }
391
392 const xiiUInt32 uiMask = 0xFFFFFFFF >> (32 - (m_pBitfield->m_uiCount - (uiLastChunk << 5)));
393 if ((m_pBitfield->m_Container[uiLastChunk] & uiMask) != 0)
394 {
395 m_uiChunk = uiLastChunk;
396 m_Iterator = sub_iterator(m_pBitfield->m_Container[uiLastChunk] & uiMask);
397 return;
398 }
399 }
400
401 // End iterator.
402 m_pBitfield = nullptr;
403 m_uiChunk = 0;
404 m_Iterator = sub_iterator();
405}
