287 {

297 virtual xiiInternal::NewInstance<void> CloneInternal(const void* pObject, xiiAllocatorBase* pAllocator) override // [tested]
298 {
299 if (pAllocator == nullptr)
300 {
301 pAllocator = AllocatorWrapper::GetAllocator();
302 }
303
304 return CloneImpl(pObject, pAllocator, xiiTraitInt<std::is_copy_constructible<CLASS>::value>());
305 }
306
308 virtual void Deallocate(void* pObject, xiiAllocatorBase* pAllocator) override // [tested]
309 {
310 if (pAllocator == nullptr)
311 {
312 pAllocator = AllocatorWrapper::GetAllocator();
313 }
314
315 CLASS* pPointer = static_cast<CLASS*>(pObject);
316 XII_DELETE(pAllocator, pPointer);
317 }

319private:

323 return xiiInternal::NewInstance<void>(nullptr, pAllocator);
