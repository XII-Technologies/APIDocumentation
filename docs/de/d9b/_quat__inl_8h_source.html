271template <typename Type>
272xiiQuatTemplate<Type> xiiQuatTemplate<Type>::MakeFromMat3(const xiiMat3Template<Type>& m)
273{
274 XII_NAN_ASSERT(&m);
275
276 const Type trace = m.Element(0, 0) + m.Element(1, 1) + m.Element(2, 2);
277 const Type half = (Type)0.5;
278
279 Type val[4];
280
281 if (trace > (Type)0)
282 {
283 Type s = xiiMath::Sqrt(trace + (Type)1);
284 Type t = half / s;
285
286 val[0] = (m.Element(1, 2) - m.Element(2, 1)) * t;
287 val[1] = (m.Element(2, 0) - m.Element(0, 2)) * t;
288 val[2] = (m.Element(0, 1) - m.Element(1, 0)) * t;
289
290 val[3] = half * s;
291 }
292 else
293 {
294 const xiiInt32 next[] = {1, 2, 0};
295 xiiInt32 i = 0;
296
297 if (m.Element(1, 1) > m.Element(0, 0))
298 i = 1;
299
300 if (m.Element(2, 2) > m.Element(i, i))
301 i = 2;
302
303 xiiInt32 j = next[i];
304 xiiInt32 k = next[j];
305
306 Type s = xiiMath::Sqrt(m.Element(i, i) - (m.Element(j, j) + m.Element(k, k)) + (Type)1);
307 Type t = half / s;
308
309 val[i] = half * s;
310 val[3] = (m.Element(j, k) - m.Element(k, j)) * t;
311 val[j] = (m.Element(i, j) + m.Element(j, i)) * t;
312 val[k] = (m.Element(i, k) + m.Element(k, i)) * t;
313 }
314
315 xiiQuatTemplate<Type> q;
316 q.x = val[0];
317 q.y = val[1];
318 q.z = val[2];
319 q.w = val[3];
320 return q;
321}

359template <typename Type>
360xiiQuatTemplate<Type> xiiQuatTemplate<Type>::MakeShortestRotation(const xiiVec3Template<Type>& vDirFrom, const xiiVec3Template<Type>& vDirTo)
361{
362 const xiiVec3Template<Type> v0 = vDirFrom.GetNormalized();
363 const xiiVec3Template<Type> v1 = vDirTo.GetNormalized();
364
365 const Type fDot = v0.Dot(v1);
366
367 // if both vectors are identical -> no rotation needed
368 if (xiiMath::IsEqual(fDot, (Type)1, (Type)0.0000001))
369 {
370 return MakeIdentity();
371 }
372 else if (xiiMath::IsEqual(fDot, (Type)-1, (Type)0.0000001)) // if both vectors are opposing
373 {
374 // find an axis, that is not identical and not opposing, xiiVec3Template::Cross-product to find perpendicular vector, rotate around that
375 if (xiiMath::Abs(v0.Dot(xiiVec3Template<Type>(1, 0, 0))) < (Type)0.8)
376 return MakeFromAxisAndAngle(v0.CrossRH(xiiVec3Template<Type>(1, 0, 0)).GetNormalized(), xiiAngleTemplate<Type>::MakeFromRadian(xiiMath::Pi<Type>()));
377 else
378 return MakeFromAxisAndAngle(v0.CrossRH(xiiVec3Template<Type>(0, 1, 0)).GetNormalized(), xiiAngleTemplate<Type>::MakeFromRadian(xiiMath::Pi<Type>()));
379 }
380
381 const xiiVec3Template<Type> c = v0.CrossRH(v1);
382 const Type d = v0.Dot(v1);
383 const Type s = xiiMath::Sqrt(((Type)1 + d) * (Type)2);
384
385 XII_ASSERT_DEBUG(c.IsValid(), "SetShortestRotation failed.");
386
387 const Type fOneDivS = 1.0f / s;
388
389 xiiQuatTemplate<Type> q;
390 q.x = c.x * fOneDivS;
391 q.y = c.y * fOneDivS;
392 q.z = c.z * fOneDivS;
393 q.w = s / (Type)2;
394 q.Normalize();
395
396 return q;
397}

399template <typename Type>
400xiiQuatTemplate<Type> xiiQuatTemplate<Type>::MakeSlerp(const xiiQuatTemplate<Type>& qFrom, const xiiQuatTemplate<Type>& qTo, Type t)
401{
402 XII_ASSERT_DEBUG((t >= (Type)0) && (t <= (Type)1), "Invalid lerp factor.");
403
404 const Type one = 1;
405 const Type qdelta = (Type)1 - (Type)0.001;
406
407 const Type fDot = (qFrom.x * qTo.x + qFrom.y * qTo.y + qFrom.z * qTo.z + qFrom.w * qTo.w);
408
409 Type cosTheta = fDot;
410
411 bool bFlipSign = false;
412 if (cosTheta < (Type)0)
413 {
414 bFlipSign = true;
415 cosTheta = -cosTheta;
416 }
417
418 Type t0, t1;
419
420 if (cosTheta < qdelta)
421 {
422 xiiAngleTemplate<Type> theta = xiiMath::ACos((Type)cosTheta);
423
424 // use sqrtInv(1+c^2) instead of 1.0/sin(theta)
425 const Type iSinTheta = (Type)1 / xiiMath::Sqrt(one - (cosTheta * cosTheta));
426 const xiiAngleTemplate<Type> tTheta = static_cast<Type>(t) * theta;
427
428 Type s0 = xiiMath::Sin(theta - tTheta);
429 Type s1 = xiiMath::Sin(tTheta);
430
431 t0 = s0 * iSinTheta;
432 t1 = s1 * iSinTheta;
433 }
434 else
435 {
436 // If q0 is nearly the same as q1 we just linearly interpolate
437 t0 = one - t;
438 t1 = t;
439 }
440
441 if (bFlipSign)
442 t1 = -t1;
443
444 xiiQuatTemplate<Type> q;
445
446 q.x = t0 * qFrom.x;
447 q.y = t0 * qFrom.y;
448 q.z = t0 * qFrom.z;
449 q.w = t0 * qFrom.w;
450
451 q.x += t1 * qTo.x;
452 q.y += t1 * qTo.y;
453 q.z += t1 * qTo.z;
454 q.w += t1 * qTo.w;
455
456 q.Normalize();
457
458 return q;
459}

467template <typename Type>
468void xiiQuatTemplate<Type>::GetAsEulerAngles(xiiAngleTemplate<Type>& out_x, xiiAngleTemplate<Type>& out_y, xiiAngleTemplate<Type>& out_z) const
469{
470 // Taken from https://en.wikipedia.org/wiki/Conversion_between_quaternions_and_Euler_angles
471 // and http://www.euclideanspace.com/maths/geometry/rotations/conversions/quaternionToEuler/
472 // adapted to our convention (yaw->pitch->roll, ZYX order or 3-2-1 order)
473
474 auto& yaw = out_z;
475 auto& pitch = out_y;
476 auto& roll = out_x;
477
478 const double fSingularityTest = w * y - z * x;
479 const double fSingularityThreshold = 0.4999995;
480
481 if (fSingularityTest > fSingularityThreshold) // singularity at north pole
482 {
483 yaw = -2.0f * xiiMath::ATan2(x, w);
484 pitch = xiiAngleTemplate<Type>::MakeFromDegree(90.0f);
485 roll = xiiAngleTemplate<Type>::MakeFromDegree(0.0f);
486 }
487 else if (fSingularityTest < -fSingularityThreshold) // singularity at south pole
488 {
489 yaw = 2.0f * xiiMath::ATan2(x, w);
490 pitch = xiiAngleTemplate<Type>::MakeFromDegree(-90.0f);
491 roll = xiiAngleTemplate<Type>::MakeFromDegree(0.0f);
492 }
493 else
494 {
495 // yaw (z-axis rotation)
496 const double siny = 2.0 * (w * z + x * y);
497 const double cosy = 1.0 - 2.0 * (y * y + z * z);
498 yaw = xiiMath::ATan2((Type)siny, (Type)cosy);
499
500 // pitch (y-axis rotation)
501 pitch = xiiMath::ASin(2.0f * (Type)fSingularityTest);
502
503 // roll (x-axis rotation)
504 const double sinr = 2.0 * (w * x + y * z);
505 const double cosr = 1.0 - 2.0 * (x * x + y * y);
506 roll = xiiMath::ATan2((Type)sinr, (Type)cosr);
507 }
508}

516 const auto& pitch = y;
517 const auto& roll = x;

519 const double sy = xiiMath::Sin(yaw * (Type)0.5);

528 q.y = (Type)(cy * sp * cr + sy * cp * sr);
529 q.z = (Type)(sy * cp * cr - cy * sp * sr);
530 return q;
