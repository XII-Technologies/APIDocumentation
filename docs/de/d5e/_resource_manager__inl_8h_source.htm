55xiiTypedResourceHandle<ResourceType> xiiResourceManager::CreateResource(xiiStringView sResourceID, DescriptorType&& descriptor, xiiStringView sResourceDescription)
56{
57 static_assert(std::is_rvalue_reference<DescriptorType&&>::value, "Please std::move the descriptor into this function");
58
59 XII_LOG_BLOCK("xiiResourceManager::CreateResource", sResourceID);
60
61 XII_LOCK(s_ResourceMutex);
62
63 xiiTypedResourceHandle<ResourceType> hResource(GetResource<ResourceType>(sResourceID, false));
64
65 ResourceType* pResource = BeginAcquireResource(hResource, xiiResourceAcquireMode::PointerOnly);
66 pResource->SetResourceDescription(sResourceDescription);
67 pResource->m_Flags.Add(xiiResourceFlags::IsCreatedResource);
68
69 XII_ASSERT_DEV(pResource->GetLoadingState() == xiiResourceState::Unloaded, "CreateResource was called on a resource that is already created");
70
71 // If this does not compile, you either passed in the wrong descriptor type for the given resource type
72 // or you forgot to std::move the descriptor when calling CreateResource
73 {
74 auto localDescriptor = std::move(descriptor);
75 xiiResourceLoadDesc ld = pResource->CreateResource(std::move(localDescriptor));
76 pResource->VerifyAfterCreateResource(ld);
77 }
78
79 XII_ASSERT_DEV(pResource->GetLoadingState() != xiiResourceState::Unloaded, "CreateResource did not set the loading state properly.");
80
81 EndAcquireResource(pResource);
82
83 return hResource;
84}

114template <typename ResourceType>
115ResourceType* xiiResourceManager::BeginAcquireResource(const xiiTypedResourceHandle<ResourceType>& hResource, xiiResourceAcquireMode mode, const xiiTypedResourceHandle<ResourceType>& hFallbackResource, xiiResourceAcquireResult* out_pAcquireResult /*= nullptr*/)
116{
117 XII_ASSERT_DEV(hResource.IsValid(), "Cannot acquire a resource through an invalid handle!");
118
119#if XII_ENABLED(XII_COMPILE_FOR_DEVELOPMENT)
120 const xiiResource* pCurrentlyUpdatingContent = xiiResource::GetCurrentlyUpdatingContent();
121 if (pCurrentlyUpdatingContent != nullptr)
122 {
123 XII_LOCK(s_ResourceMutex);
124 XII_ASSERT_DEV(mode == xiiResourceAcquireMode::PointerOnly || IsResourceTypeAcquireDuringUpdateContentAllowed(pCurrentlyUpdatingContent->GetDynamicRTTI(), xiiGetStaticRTTI<ResourceType>()),
125 "Trying to acquire a resource of type '{0}' during '{1}::UpdateContent()'. This has to be enabled by calling "
126 "xiiResourceManager::AllowResourceTypeAcquireDuringUpdateContent<{1}, {0}>(); at engine startup, for example in "
127 "xiiGameApplication::Init_SetupDefaultResources().",
128 xiiGetStaticRTTI<ResourceType>()->GetTypeName(), pCurrentlyUpdatingContent->GetDynamicRTTI()->GetTypeName());
129 }
130#endif
131
132 ResourceType* pResource = (ResourceType*)hResource.m_hTypeless.m_pResource;
133
134 // XII_ASSERT_DEV(pResource->m_iLockCount < 20, "You probably forgot somewhere to call 'EndAcquireResource' in sync with 'BeginAcquireResource'.");
135 XII_ASSERT_DEBUG(pResource->GetDynamicRTTI()->template IsDerivedFrom<ResourceType>(),
136 "The requested resource does not have the same type ('{0}') as the resource handle ('{1}').", pResource->GetDynamicRTTI()->GetTypeName(),
137 xiiGetStaticRTTI<ResourceType>()->GetTypeName());
138
139 if (mode == xiiResourceAcquireMode::AllowLoadingFallback && GetForceNoFallbackAcquisition() > 0)
140 {
141 mode = xiiResourceAcquireMode::BlockTillLoaded;
142 }
143
144 if (mode == xiiResourceAcquireMode::PointerOnly)
145 {
146 if (out_pAcquireResult)
147 *out_pAcquireResult = xiiResourceAcquireResult::Final;
148
149 // pResource->m_iLockCount.Increment();
150 return pResource;
151 }
152
153 // only set the last accessed time stamp, if it is actually needed, pointer-only access might not mean that the resource is used
154 // productively
155 pResource->m_LastAcquire = GetLastFrameUpdate();
156
157 if (pResource->GetLoadingState() != xiiResourceState::LoadedResourceMissing)
158 {
159 if (pResource->GetLoadingState() != xiiResourceState::Loaded)
160 {
161 // if BlockTillLoaded is specified, it will prepended to the preload array, thus will be loaded immediately
162 InternalPreloadResource(pResource, mode >= xiiResourceAcquireMode::BlockTillLoaded);
163
164 if (mode == xiiResourceAcquireMode::AllowLoadingFallback &&
165 (pResource->m_hLoadingFallback.IsValid() || hFallbackResource.IsValid() || GetResourceTypeLoadingFallback<ResourceType>().IsValid()))
166 {
167 // return the fallback resource for now, if there is one
168 if (out_pAcquireResult)
169 *out_pAcquireResult = xiiResourceAcquireResult::LoadingFallback;
170
171 // Fallback order is as follows:
172 // 1) Prefer any resource specific fallback resource
173 // 2) If not available, use the fallback that is given to BeginAcquireResource, as that is at least specific to the situation
174 // 3) If nothing else is available, take the fallback for the whole resource type
175
176 if (pResource->m_hLoadingFallback.IsValid())
177 return (ResourceType*)BeginAcquireResource(pResource->m_hLoadingFallback, xiiResourceAcquireMode::BlockTillLoaded);
178 else if (hFallbackResource.IsValid())
179 return (ResourceType*)BeginAcquireResource(hFallbackResource, xiiResourceAcquireMode::BlockTillLoaded);
180 else
181 return (ResourceType*)BeginAcquireResource(GetResourceTypeLoadingFallback<ResourceType>(), xiiResourceAcquireMode::BlockTillLoaded);
182 }
183
184 EnsureResourceLoadingState(pResource, xiiResourceState::Loaded);
185 }
186 else
187 {
188 // as long as there are more quality levels available, schedule the resource for more loading
189 // accessing IsQueuedForLoading without a lock here is save because InternalPreloadResource() will lock and early out if necessary
190 // and accidentally skipping InternalPreloadResource() is no problem
191 if (IsQueuedForLoading(pResource) == false && pResource->GetNumQualityLevelsLoadable() > 0)
192 InternalPreloadResource(pResource, false);
193 }
194 }
195
196 if (pResource->GetLoadingState() == xiiResourceState::LoadedResourceMissing)
197 {
198 // When you get a crash with a stack overflow in this code path, then the resource to be used as the
199 // 'missing resource' replacement might be missing itself.
200
201 if (xiiResourceManager::GetResourceTypeMissingFallback<ResourceType>().IsValid())
202 {
203 if (out_pAcquireResult)
204 *out_pAcquireResult = xiiResourceAcquireResult::MissingFallback;
205
206 return (ResourceType*)BeginAcquireResource(
207 xiiResourceManager::GetResourceTypeMissingFallback<ResourceType>(), xiiResourceAcquireMode::BlockTillLoaded);
208 }
209
210 if (mode != xiiResourceAcquireMode::AllowLoadingFallback_NeverFail && mode != xiiResourceAcquireMode::BlockTillLoaded_NeverFail)
211 {
212 XII_REPORT_FAILURE("The resource '{0}' of type '{1}' is missing and no fallback is available", pResource->GetResourceID(),
213 xiiGetStaticRTTI<ResourceType>()->GetTypeName());
214 }
215
216 if (out_pAcquireResult)
217 *out_pAcquireResult = xiiResourceAcquireResult::None;
218
219 return nullptr;
220 }
221
222 if (out_pAcquireResult)
223 *out_pAcquireResult = xiiResourceAcquireResult::Final;
224
225 // pResource->m_iLockCount.Increment();
226 return pResource;
227}

242template <typename ResourceType>
243xiiLockedObject<xiiMutex, xiiDynamicArray<xiiResource*>> xiiResourceManager::GetAllResourcesOfType()
244{
245 const xiiRTTI* pBaseType = xiiGetStaticRTTI<ResourceType>();
246
247 auto& container = GetLoadedResourceOfTypeTempContainer();
248
249 // We use a static container here to ensure its life-time is extended beyond
250 // calls to this function as the locked object does not own the passed-in object
251 // and thus does not extend the data life-time. It is safe to do this, as the
252 // locked object holding the container ensures the container will not be
253 // accessed concurrently.
254 xiiLockedObject<xiiMutex, xiiDynamicArray<xiiResource*>> loadedResourcesLock(s_ResourceMutex, &container);
255
256 container.Clear();
257
258 for (auto itType = GetLoadedResources().GetIterator(); itType.IsValid(); itType.Next())
259 {
260 const xiiRTTI* pDerivedType = itType.Key();
261
262 if (pDerivedType->IsDerivedFrom(pBaseType))
263 {
264 const LoadedResources& lr = GetLoadedResources()[pDerivedType];
265
266 container.Reserve(container.GetCount() + lr.m_Resources.GetCount());
267
268 for (auto itResource : lr.m_Resources)
269 {
270 container.PushBack(itResource.Value());
271 }
272 }
273 }
274
275 return loadedResourcesLock;
276}
