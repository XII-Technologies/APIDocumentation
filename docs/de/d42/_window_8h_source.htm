65  bool operator==(xiiWindowHandle& rhs)
66  {
67  if (type != rhs.type)
68  return false;
69 
70  if (type == Type::SDL)
71  {
72  return sdlWindow == rhs.sdlWindow;
73  }
74  else
75  {
76  // We don't compare the connection because we only want to know if we reference the same window.
77  return xcbWindow.m_hWindow == rhs.xcbWindow.m_hWindow;
78  }
79  }
