55  {
56  Invalid = 0,
57  SDL = 1, // Used by the Runtime

59  };

63  {

66  };
67 
68  bool operator==(xiiWindowHandle& rhs)
69  {
70  if (type != rhs.type)
71  return false;
72 
73  if (type == Type::SDL)
74  {
75  return sdlWindow == rhs.sdlWindow;
76  }
77  else
78  {
79  // We don't compare the connection because we only want to know if we reference the same window.
80  return xcbWindow.m_hWindow == rhs.xcbWindow.m_hWindow;
81  }
82  }
