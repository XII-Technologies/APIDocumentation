94inline void xiiStringBuilder::Append(xiiUInt32 uiChar)
95{
96 char szChar[6] = {0, 0, 0, 0, 0, 0};
97 char* pChar = &szChar[0];
98
99 xiiUnicodeUtils::EncodeUtf32ToUtf8(uiChar, pChar);
100 xiiUInt32 uiCharLen = (xiiUInt32)(pChar - szChar);
101 xiiUInt32 uiOldCount = m_Data.GetCount();
102 m_Data.SetCountUninitialized(uiOldCount + uiCharLen);
103 uiOldCount--;
104 for (xiiUInt32 i = 0; i < uiCharLen; i++)
105 {
106 m_Data[uiOldCount + i] = szChar[i];
107 }
108 m_Data[uiOldCount + uiCharLen] = '\0';
109}

178inline void xiiStringBuilder::ChangeCharacter(iterator& ref_it, xiiUInt32 uiCharacter)
179{
180 XII_ASSERT_DEV(ref_it.IsValid(), "The given character iterator does not point to a valid character.");
181 XII_ASSERT_DEV(ref_it.GetData() >= GetData() && ref_it.GetData() < GetData() + GetElementCount(),
182 "The given character iterator does not point into this string. It was either created from another string, or this string "
183 "has been reallocated in the mean time.");
184
185 // this is only an optimization for pure ASCII strings
186 // without it, the code below would still work
187 if (xiiUnicodeUtils::IsASCII(*ref_it) && xiiUnicodeUtils::IsASCII(uiCharacter))
188 {
189 char* pPos = const_cast<char*>(ref_it.GetData()); // yes, I know...
190 *pPos = uiCharacter & 0xFF;
191 return;
192 }
193
194 ChangeCharacterNonASCII(ref_it, uiCharacter);
195}

212template <typename Container>
213bool xiiUnicodeUtils::RepairNonUtf8Text(const char* pStartData, const char* pEndData, Container& out_result)
214{
215 if (xiiUnicodeUtils::IsValidUtf8(pStartData, pEndData))
216 {
217 out_result = xiiStringView(pStartData, pEndData);
218 return false;
219 }
220
221 out_result.Clear();
222
223 xiiHybridArray<char, 1024> fixedText;
224 xiiUnicodeUtils::UtfInserter<char, decltype(fixedText)> inserter(&fixedText);
225
226 while (pStartData < pEndData)
227 {
228 const xiiUInt32 uiChar = xiiUnicodeUtils::DecodeUtf8ToUtf32(pStartData);
229 xiiUnicodeUtils::EncodeUtf32ToUtf8(uiChar, inserter);
230 }
231
232 XII_ASSERT_DEV(xiiUnicodeUtils::IsValidUtf8(fixedText.GetData(), fixedText.GetData() + fixedText.GetCount()), "Repaired text is still not a valid Utf8 string.");
233
234 out_result = xiiStringView(fixedText.GetData(), fixedText.GetCount());
235 return true;
236}
