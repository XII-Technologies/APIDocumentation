26 xiiResult xiiOSFile::InternalOpen(xiiStringView sFile, xiiFileOpenMode::Enum OpenMode, xiiFileShareMode::Enum FileShareMode)
27 {
28  const xiiTime sleepTime = xiiTime::Milliseconds(20);
29  xiiInt32 iRetries = 20;
30 
31  if (FileShareMode == xiiFileShareMode::Default)
32  {
33  // when 'default' share mode is requested, use 'share reads' when opening a file for reading
34  // and use 'exclusive' when opening a file for writing
35 
36  if (OpenMode == xiiFileOpenMode::Read)
37  {
38  FileShareMode = xiiFileShareMode::SharedReads;
39  }
40  else
41  {
42  FileShareMode = xiiFileShareMode::Exclusive;
43  }
44  }
45 
46  DWORD dwSharedMode = 0; // exclusive access
47  if (FileShareMode == xiiFileShareMode::SharedReads)
48  {
49  dwSharedMode = FILE_SHARE_READ;
50  }
51 
52  while (iRetries > 0)
53  {
54  SetLastError(ERROR_SUCCESS);
55  DWORD error = 0;
56 
57  switch (OpenMode)
58  {
59  case xiiFileOpenMode::Read:
60  m_FileData.m_pFileHandle = CreateFileW(xiiDosDevicePath(sFile), GENERIC_READ, dwSharedMode, nullptr, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, nullptr);
61  break;
62 
63  case xiiFileOpenMode::Write:
64  m_FileData.m_pFileHandle = CreateFileW(xiiDosDevicePath(sFile), GENERIC_WRITE, dwSharedMode, nullptr, CREATE_ALWAYS, FILE_ATTRIBUTE_NORMAL, nullptr);
65  break;
66 
67  case xiiFileOpenMode::Append:
68  m_FileData.m_pFileHandle = CreateFileW(xiiDosDevicePath(sFile), FILE_APPEND_DATA, dwSharedMode, nullptr, OPEN_ALWAYS, FILE_ATTRIBUTE_NORMAL, nullptr);
69 
70  // in append mode we need to set the file pointer to the end explicitly, otherwise GetFilePosition might return 0 the first time
71  if ((m_FileData.m_pFileHandle != nullptr) && (m_FileData.m_pFileHandle != INVALID_HANDLE_VALUE))
72  InternalSetFilePosition(0, xiiFileSeekMode::FromEnd);
73 
74  break;
75 
76  XII_DEFAULT_CASE_NOT_IMPLEMENTED
77  }
78 
79  const xiiResult res = ((m_FileData.m_pFileHandle != nullptr) && (m_FileData.m_pFileHandle != INVALID_HANDLE_VALUE)) ? XII_SUCCESS : XII_FAILURE;
80 
81  if (res.Failed())
82  {
83  if (xiiOSFile::ExistsDirectory(sFile))
84  {
85  // trying to 'open' a directory fails with little useful error codes such as 'access denied'
86  return XII_FAILURE;
87  }
88 
89  error = GetLastError();
90 
91  // file does not exist
92  if (error == ERROR_FILE_NOT_FOUND || error == ERROR_PATH_NOT_FOUND)
93  return res;
94  // badly formed path, happens when two absolute paths are concatenated
95  if (error == ERROR_INVALID_NAME)
96  return res;
97 
98  if (error == ERROR_SHARING_VIOLATION
99  // these two situations happen when the xiiInspector is connected
100  // for some reason, the networking blocks file reading (when run on the same machine)
101  // retrying fixes the problem, but can introduce very long stalls
102  || error == WSAEWOULDBLOCK || error == ERROR_SUCCESS)
103  {
104  if (m_bRetryOnSharingViolation)
105  {
106  --iRetries;
107  xiiThreadUtils::Sleep(sleepTime);
108  continue; // try again
109  }
110  else
111  {
112  return res;
113  }
114  }
115 
116  // anything else, print an error (for now)
117  xiiLog::Error("CreateFile failed with error {0}", xiiArgErrorCode(error));
118  }
119 
120  return res;
121  }
122 
123  return XII_FAILURE;
124 }

132 xiiResult xiiOSFile::InternalWrite(const void* pBuffer, xiiUInt64 uiBytes)
133 {
134  const xiiUInt32 uiBatchBytes = 1024 * 1024 * 1024; // 1 GB
135 
136  // first write out all the data in 1GB batches
137  while (uiBytes > uiBatchBytes)
138  {
139  DWORD uiBytesWritten = 0;
140  if ((!WriteFile(m_FileData.m_pFileHandle, pBuffer, uiBatchBytes, &uiBytesWritten, nullptr)) || (uiBytesWritten != uiBatchBytes))
141  return XII_FAILURE;
142 
143  uiBytes -= uiBatchBytes;
144  pBuffer = xiiMemoryUtils::AddByteOffset(pBuffer, uiBatchBytes);
145  }
146 
147  if (uiBytes > 0)
148  {
149  const xiiUInt32 uiBytes32 = static_cast<xiiUInt32>(uiBytes);
150 
151  DWORD uiBytesWritten = 0;
152  if ((!WriteFile(m_FileData.m_pFileHandle, pBuffer, uiBytes32, &uiBytesWritten, nullptr)) || (uiBytesWritten != uiBytes32))
153  return XII_FAILURE;
154  }
155 
156  return XII_SUCCESS;
157 }

159 xiiUInt64 xiiOSFile::InternalRead(void* pBuffer, xiiUInt64 uiBytes)
160 {
161  xiiUInt64 uiBytesRead = 0;
162 
163  const xiiUInt32 uiBatchBytes = 1024 * 1024 * 1024; // 1 GB
164 
165  // first write out all the data in 1GB batches
166  while (uiBytes > uiBatchBytes)
167  {
168  DWORD uiBytesReadThisTime = 0;
169  if (!ReadFile(m_FileData.m_pFileHandle, pBuffer, uiBatchBytes, &uiBytesReadThisTime, nullptr))
170  return uiBytesRead + uiBytesReadThisTime;
171 
172  uiBytesRead += uiBytesReadThisTime;
173 
174  if (uiBytesReadThisTime != uiBatchBytes)
175  return uiBytesRead;
176 
177  uiBytes -= uiBatchBytes;
178  pBuffer = xiiMemoryUtils::AddByteOffset(pBuffer, uiBatchBytes);
179  }
180 
181  if (uiBytes > 0)
182  {
183  const xiiUInt32 uiBytes32 = static_cast<xiiUInt32>(uiBytes);
184 
185  DWORD uiBytesReadThisTime = 0;
186  if (!ReadFile(m_FileData.m_pFileHandle, pBuffer, uiBytes32, &uiBytesReadThisTime, nullptr))
187  return uiBytesRead + uiBytesReadThisTime;
188 
189  uiBytesRead += uiBytesReadThisTime;
190  }
191 
192  return uiBytesRead;
193 }

347 void xiiFileSystemIterator::StartSearch(xiiStringView sSearchStart, xiiBitflags<xiiFileSystemIteratorFlags> flags /*= xiiFileSystemIteratorFlags::All*/)
348 {
349  XII_ASSERT_DEV(m_Data.m_Handles.IsEmpty(), "Cannot start another search.");
350 
351  m_sSearchTerm = sSearchStart;
352 
353  xiiStringBuilder sSearch = sSearchStart;
354  sSearch.MakeCleanPath();
355 
356  // same as just passing in the folder path, so remove this
357  if (sSearch.EndsWith("/*"))
358  sSearch.Shrink(0, 2);
359 
360  // The Windows documentation disallows trailing (back)slashes.
361  sSearch.Trim(nullptr, "/");
362 
363  // Since the use of wildcard-ed file names will disable recursion, we ensure both are not used simultaneously.
364  const bool bHasWildcard = sSearch.FindLastSubString("*") || sSearch.FindLastSubString("?");
365  XII_ASSERT_DEV(flags.IsSet(xiiFileSystemIteratorFlags::Recursive) == false || bHasWildcard == false, "Recursive file iteration does not support wildcards. Either don't use recursion, or filter the filenames manually.");
366 
367  m_sCurPath = sSearch.GetFileDirectory();
368 
369  XII_ASSERT_DEV(sSearch.IsAbsolutePath(), "The path '{0}' is not absolute.", m_sCurPath);
370 
371  m_Flags = flags;
372 
373  WIN32_FIND_DATAW data;
374  HANDLE hSearch = FindFirstFileW(xiiDosDevicePath(sSearch), &data);
375 
376  if ((hSearch == nullptr) || (hSearch == INVALID_HANDLE_VALUE))
377  return;
378 
379  m_CurFile.m_uiFileSize = HighLowToUInt64(data.nFileSizeHigh, data.nFileSizeLow);
380  m_CurFile.m_bIsDirectory = (data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
381  m_CurFile.m_sParentPath = m_sCurPath;
382  m_CurFile.m_sName = data.cFileName;
383  m_CurFile.m_LastModificationTime.SetInt64(FileTimeToEpoch(data.ftLastWriteTime), xiiSIUnitOfTime::Microsecond);
384 
385  m_Data.m_Handles.PushBack(hSearch);
386 
387  if (xiiOSFile::ExistsDirectory(sSearch))
388  {
389  // when calling FindFirstFileW with a path to a folder (e.g. "C:/test") it will report "test" as the very first item
390  // which is typically NOT what one wants, instead you want items INSIDE that folder to be reported
391  // this is especially annoying when 'Recursion' is disabled, as "C:/test" would result in "C:/test" being reported
392  // but no items inside it
393  // therefore, when the start search points to a directory, we enable recursion for one call to 'Next', thus enter
394  // the directory, and then switch it back again; all following calls to 'Next' will then iterate through the sub directory
395 
396  const bool bRecursive = m_Flags.IsSet(xiiFileSystemIteratorFlags::Recursive);
397  m_Flags.Add(xiiFileSystemIteratorFlags::Recursive);
398 
399  Next();
400 
401  m_Flags.AddOrRemove(xiiFileSystemIteratorFlags::Recursive, bRecursive);
402  return;
403  }
404 
405  if ((m_CurFile.m_sName == "..") || (m_CurFile.m_sName == "."))
406  {
407  Next(); // will search for the next file or folder that is not ".." or "." ; might return false though
408  return;
409  }
410 
411  if (m_CurFile.m_bIsDirectory)
412  {
413  if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFolders))
414  {
415  Next();
416  return;
417  }
418  }
419  else
420  {
421  if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFiles))
422  {
423  Next();
424  return;
425  }
426  }
427 }

429 xiiInt32 xiiFileSystemIterator::InternalNext()
430 {
431  constexpr xiiInt32 ReturnFailure = 0;
432  constexpr xiiInt32 ReturnSuccess = 1;
433  constexpr xiiInt32 ReturnCallInternalNext = 2;
434 
435  if (m_Data.m_Handles.IsEmpty())
436  return ReturnFailure;
437 
438  if (m_Flags.IsSet(xiiFileSystemIteratorFlags::Recursive) && m_CurFile.m_bIsDirectory && (m_CurFile.m_sName != "..") && (m_CurFile.m_sName != "."))
439  {
440  m_sCurPath.AppendPath(m_CurFile.m_sName);
441 
442  xiiStringBuilder sNewSearch = m_sCurPath;
443  sNewSearch.AppendPath("*");
444 
445  WIN32_FIND_DATAW data;
446  HANDLE hSearch = FindFirstFileW(xiiDosDevicePath(sNewSearch), &data);
447 
448  if ((hSearch != nullptr) && (hSearch != INVALID_HANDLE_VALUE))
449  {
450  m_CurFile.m_uiFileSize = HighLowToUInt64(data.nFileSizeHigh, data.nFileSizeLow);
451  m_CurFile.m_bIsDirectory = (data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
452  m_CurFile.m_sParentPath = m_sCurPath;
453  m_CurFile.m_sName = data.cFileName;
454  m_CurFile.m_LastModificationTime.SetInt64(FileTimeToEpoch(data.ftLastWriteTime), xiiSIUnitOfTime::Microsecond);
455 
456  m_Data.m_Handles.PushBack(hSearch);
457 
458  if ((m_CurFile.m_sName == "..") || (m_CurFile.m_sName == "."))
459  return ReturnCallInternalNext; // will search for the next file or folder that is not ".." or "." ; might return false though
460 
461  if (m_CurFile.m_bIsDirectory)
462  {
463  if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFolders))
464  return ReturnCallInternalNext;
465  }
466  else
467  {
468  if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFiles))
469  return ReturnCallInternalNext;
470  }
471 
472  return ReturnSuccess;
473  }
474 
475  // if the recursion did not work, just iterate in this folder further
476  }
477 
478  WIN32_FIND_DATAW data;
479  if (!FindNextFileW(m_Data.m_Handles.PeekBack(), &data))
480  {
481  // nothing found in this directory anymore
482  FindClose(m_Data.m_Handles.PeekBack());
483  m_Data.m_Handles.PopBack();
484 
485  if (m_Data.m_Handles.IsEmpty())
486  return ReturnFailure;
487 
488  m_sCurPath.PathParentDirectory();
489  if (m_sCurPath.EndsWith("/"))
490  {
491  m_sCurPath.Shrink(0, 1); // Remove trailing /
492  }
493 
494  return ReturnCallInternalNext;
495  }
496 
497  m_CurFile.m_uiFileSize = HighLowToUInt64(data.nFileSizeHigh, data.nFileSizeLow);
498  m_CurFile.m_bIsDirectory = (data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
499  m_CurFile.m_sParentPath = m_sCurPath;
500  m_CurFile.m_sName = data.cFileName;
501  m_CurFile.m_LastModificationTime.SetInt64(FileTimeToEpoch(data.ftLastWriteTime), xiiSIUnitOfTime::Microsecond);
502 
503  if ((m_CurFile.m_sName == "..") || (m_CurFile.m_sName == "."))
504  return ReturnCallInternalNext;
505 
506  if (m_CurFile.m_bIsDirectory)
507  {
508  if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFolders))
509  return ReturnCallInternalNext;
510  }
511  else
512  {
513  if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFiles))
514  return ReturnCallInternalNext;
515  }
516 
517  return ReturnSuccess;
518 }
