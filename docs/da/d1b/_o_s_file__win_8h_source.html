26xiiResult xiiOSFile::InternalOpen(xiiStringView sFile, xiiFileOpenMode::Enum OpenMode, xiiFileShareMode::Enum FileShareMode)
27{
28 const xiiTime sleepTime = xiiTime::MakeFromMilliseconds(20);
29 xiiInt32 iRetries = 20;
30
31 if (FileShareMode == xiiFileShareMode::Default)
32 {
33 // when 'default' share mode is requested, use 'share reads' when opening a file for reading
34 // and use 'exclusive' when opening a file for writing
35
36 if (OpenMode == xiiFileOpenMode::Read)
37 {
38 FileShareMode = xiiFileShareMode::SharedReads;
39 }
40 else
41 {
42 FileShareMode = xiiFileShareMode::Exclusive;
43 }
44 }
45
46 DWORD dwSharedMode = 0; // exclusive access
47 if (FileShareMode == xiiFileShareMode::SharedReads)
48 {
49 dwSharedMode = FILE_SHARE_READ;
50 }
51
52 while (iRetries > 0)
53 {
54 SetLastError(ERROR_SUCCESS);
55 DWORD error = 0;
56
57 switch (OpenMode)
58 {
59 case xiiFileOpenMode::Read:
60 m_FileData.m_pFileHandle = CreateFileW(xiiDosDevicePath(sFile), GENERIC_READ, dwSharedMode, nullptr, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, nullptr);
61 break;
62
63 case xiiFileOpenMode::Write:
64 m_FileData.m_pFileHandle = CreateFileW(xiiDosDevicePath(sFile), GENERIC_WRITE, dwSharedMode, nullptr, CREATE_ALWAYS, FILE_ATTRIBUTE_NORMAL, nullptr);
65 break;
66
67 case xiiFileOpenMode::Append:
68 m_FileData.m_pFileHandle = CreateFileW(xiiDosDevicePath(sFile), FILE_APPEND_DATA, dwSharedMode, nullptr, OPEN_ALWAYS, FILE_ATTRIBUTE_NORMAL, nullptr);
69
70 // in append mode we need to set the file pointer to the end explicitly, otherwise GetFilePosition might return 0 the first time
71 if ((m_FileData.m_pFileHandle != nullptr) && (m_FileData.m_pFileHandle != INVALID_HANDLE_VALUE))
72 InternalSetFilePosition(0, xiiFileSeekMode::FromEnd);
73
74 break;
75
76 XII_DEFAULT_CASE_NOT_IMPLEMENTED
77 }
78
79 const xiiResult res = ((m_FileData.m_pFileHandle != nullptr) && (m_FileData.m_pFileHandle != INVALID_HANDLE_VALUE)) ? XII_SUCCESS : XII_FAILURE;
80
81 if (res.Failed())
82 {
83 if (xiiOSFile::ExistsDirectory(sFile))
84 {
85 // trying to 'open' a directory fails with little useful error codes such as 'access denied'
86 return XII_FAILURE;
87 }
88
89 error = GetLastError();
90
91 // file does not exist
92 if (error == ERROR_FILE_NOT_FOUND || error == ERROR_PATH_NOT_FOUND)
93 return res;
94 // badly formed path, happens when two absolute paths are concatenated
95 if (error == ERROR_INVALID_NAME)
96 return res;
97
98 if (error == ERROR_SHARING_VIOLATION
99 // these two situations happen when the xiiInspector is connected
100 // for some reason, the networking blocks file reading (when run on the same machine)
101 // retrying fixes the problem, but can introduce very long stalls
102 || error == WSAEWOULDBLOCK || error == ERROR_SUCCESS)
103 {
104 if (m_bRetryOnSharingViolation)
105 {
106 --iRetries;
107 xiiThreadUtils::Sleep(sleepTime);
108 continue; // try again
109 }
110 else
111 {
112 return res;
113 }
114 }
115
116 // anything else, print an error (for now)
117 xiiLog::Error("CreateFile failed with error {0}", xiiArgErrorCode(error));
118 }
119
120 return res;
121 }
122
123 return XII_FAILURE;
124}

132xiiResult xiiOSFile::InternalWrite(const void* pBuffer, xiiUInt64 uiBytes)
133{
134 const xiiUInt32 uiBatchBytes = 1024 * 1024 * 1024; // 1 GB
135
136 // first write out all the data in 1GB batches
137 while (uiBytes > uiBatchBytes)
138 {
139 DWORD uiBytesWritten = 0;
140 if ((!WriteFile(m_FileData.m_pFileHandle, pBuffer, uiBatchBytes, &uiBytesWritten, nullptr)) || (uiBytesWritten != uiBatchBytes))
141 return XII_FAILURE;
142
143 uiBytes -= uiBatchBytes;
144 pBuffer = xiiMemoryUtils::AddByteOffset(pBuffer, uiBatchBytes);
145 }
146
147 if (uiBytes > 0)
148 {
149 const xiiUInt32 uiBytes32 = static_cast<xiiUInt32>(uiBytes);
150
151 DWORD uiBytesWritten = 0;
152 if ((!WriteFile(m_FileData.m_pFileHandle, pBuffer, uiBytes32, &uiBytesWritten, nullptr)) || (uiBytesWritten != uiBytes32))
153 return XII_FAILURE;
154 }
155
156 return XII_SUCCESS;
157}

159xiiUInt64 xiiOSFile::InternalRead(void* pBuffer, xiiUInt64 uiBytes)
160{
161 xiiUInt64 uiBytesRead = 0;
162
163 const xiiUInt32 uiBatchBytes = 1024 * 1024 * 1024; // 1 GB
164
165 // first write out all the data in 1GB batches
166 while (uiBytes > uiBatchBytes)
167 {
168 DWORD uiBytesReadThisTime = 0;
169 if (!ReadFile(m_FileData.m_pFileHandle, pBuffer, uiBatchBytes, &uiBytesReadThisTime, nullptr))
170 return uiBytesRead + uiBytesReadThisTime;
171
172 uiBytesRead += uiBytesReadThisTime;
173
174 if (uiBytesReadThisTime != uiBatchBytes)
175 return uiBytesRead;
176
177 uiBytes -= uiBatchBytes;
178 pBuffer = xiiMemoryUtils::AddByteOffset(pBuffer, uiBatchBytes);
179 }
180
181 if (uiBytes > 0)
182 {
183 const xiiUInt32 uiBytes32 = static_cast<xiiUInt32>(uiBytes);
184
185 DWORD uiBytesReadThisTime = 0;
186 if (!ReadFile(m_FileData.m_pFileHandle, pBuffer, uiBytes32, &uiBytesReadThisTime, nullptr))
187 return uiBytesRead + uiBytesReadThisTime;
188
189 uiBytesRead += uiBytesReadThisTime;
190 }
191
192 return uiBytesRead;
193}

294xiiResult xiiOSFile::InternalGetFileStats(xiiStringView sFileOrFolder, xiiFileStats& out_Stats)
295{
296 xiiStringBuilder s = sFileOrFolder;
297
298 // FindFirstFile does not like paths that end with a separator, so remove them all
299 s.Trim(nullptr, "/\\");
300
301 // handle the case that this query is done on the 'device part' of a path
302 if (s.GetCharacterCount() <= 2) // 'C:', 'D:', 'E' etc.
303 {
304 s.ToUpper();
305
306 out_Stats.m_uiFileSize = 0;
307 out_Stats.m_bIsDirectory = true;
308 out_Stats.m_sParentPath.Clear();
309 out_Stats.m_sName = s;
310 out_Stats.m_LastModificationTime = xiiTimestamp::MakeInvalid();
311 return XII_SUCCESS;
312 }
313
314 WIN32_FIND_DATAW data;
315 HANDLE hSearch = FindFirstFileW(xiiDosDevicePath(s), &data);
316
317 if ((hSearch == nullptr) || (hSearch == INVALID_HANDLE_VALUE))
318 return XII_FAILURE;
319
320 out_Stats.m_uiFileSize = HighLowToUInt64(data.nFileSizeHigh, data.nFileSizeLow);
321 out_Stats.m_bIsDirectory = (data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
322 out_Stats.m_sParentPath = sFileOrFolder;
323 out_Stats.m_sParentPath.PathParentDirectory();
324 out_Stats.m_sName = data.cFileName;
325 out_Stats.m_LastModificationTime = xiiTimestamp::MakeFromInt(FileTimeToEpoch(data.ftLastWriteTime), xiiSIUnitOfTime::Microsecond);
326
327 FindClose(hSearch);
328 return XII_SUCCESS;
329}

349void xiiFileSystemIterator::StartSearch(xiiStringView sSearchStart, xiiBitflags<xiiFileSystemIteratorFlags> flags /*= xiiFileSystemIteratorFlags::All*/)
350{
351 XII_ASSERT_DEV(m_Data.m_Handles.IsEmpty(), "Cannot start another search.");
352
353 m_sSearchTerm = sSearchStart;
354
355 xiiStringBuilder sSearch = sSearchStart;
356 sSearch.MakeCleanPath();
357
358 // same as just passing in the folder path, so remove this
359 if (sSearch.EndsWith("/*"))
360 sSearch.Shrink(0, 2);
361
362 // The Windows documentation disallows trailing (back)slashes.
363 sSearch.Trim(nullptr, "/");
364
365 // Since the use of wildcard-ed file names will disable recursion, we ensure both are not used simultaneously.
366 const bool bHasWildcard = sSearch.FindLastSubString("*") || sSearch.FindLastSubString("?");
367 XII_ASSERT_DEV(flags.IsSet(xiiFileSystemIteratorFlags::Recursive) == false || bHasWildcard == false, "Recursive file iteration does not support wildcards. Either don't use recursion, or filter the filenames manually.");
368
369 m_sCurPath = sSearch.GetFileDirectory();
370
371 XII_ASSERT_DEV(sSearch.IsAbsolutePath(), "The path '{0}' is not absolute.", m_sCurPath);
372
373 m_Flags = flags;
374
375 WIN32_FIND_DATAW data;
376 HANDLE hSearch = FindFirstFileW(xiiDosDevicePath(sSearch), &data);
377
378 if ((hSearch == nullptr) || (hSearch == INVALID_HANDLE_VALUE))
379 return;
380
381 m_CurFile.m_uiFileSize = HighLowToUInt64(data.nFileSizeHigh, data.nFileSizeLow);
382 m_CurFile.m_bIsDirectory = (data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
383 m_CurFile.m_sParentPath = m_sCurPath;
384 m_CurFile.m_sName = data.cFileName;
385 m_CurFile.m_LastModificationTime = xiiTimestamp::MakeFromInt(FileTimeToEpoch(data.ftLastWriteTime), xiiSIUnitOfTime::Microsecond);
386
387 m_Data.m_Handles.PushBack(hSearch);
388
389 if (xiiOSFile::ExistsDirectory(sSearch))
390 {
391 // when calling FindFirstFileW with a path to a folder (e.g. "C:/test") it will report "test" as the very first item
392 // which is typically NOT what one wants, instead you want items INSIDE that folder to be reported
393 // this is especially annoying when 'Recursion' is disabled, as "C:/test" would result in "C:/test" being reported
394 // but no items inside it
395 // therefore, when the start search points to a directory, we enable recursion for one call to 'Next', thus enter
396 // the directory, and then switch it back again; all following calls to 'Next' will then iterate through the sub directory
397
398 const bool bRecursive = m_Flags.IsSet(xiiFileSystemIteratorFlags::Recursive);
399 m_Flags.Add(xiiFileSystemIteratorFlags::Recursive);
400
401 Next();
402
403 m_Flags.AddOrRemove(xiiFileSystemIteratorFlags::Recursive, bRecursive);
404 return;
405 }
406
407 if ((m_CurFile.m_sName == "..") || (m_CurFile.m_sName == "."))
408 {
409 Next(); // will search for the next file or folder that is not ".." or "." ; might return false though
410 return;
411 }
412
413 if (m_CurFile.m_bIsDirectory)
414 {
415 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFolders))
416 {
417 Next();
418 return;
419 }
420 }
421 else
422 {
423 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFiles))
424 {
425 Next();
426 return;
427 }
428 }
429}

431xiiInt32 xiiFileSystemIterator::InternalNext()
432{
433 constexpr xiiInt32 ReturnFailure = 0;
434 constexpr xiiInt32 ReturnSuccess = 1;
435 constexpr xiiInt32 ReturnCallInternalNext = 2;
436
437 if (m_Data.m_Handles.IsEmpty())
438 return ReturnFailure;
439
440 if (m_Flags.IsSet(xiiFileSystemIteratorFlags::Recursive) && m_CurFile.m_bIsDirectory && (m_CurFile.m_sName != "..") && (m_CurFile.m_sName != "."))
441 {
442 m_sCurPath.AppendPath(m_CurFile.m_sName);
443
444 xiiStringBuilder sNewSearch = m_sCurPath;
445 sNewSearch.AppendPath("*");
446
447 WIN32_FIND_DATAW data;
448 HANDLE hSearch = FindFirstFileW(xiiDosDevicePath(sNewSearch), &data);
449
450 if ((hSearch != nullptr) && (hSearch != INVALID_HANDLE_VALUE))
451 {
452 m_CurFile.m_uiFileSize = HighLowToUInt64(data.nFileSizeHigh, data.nFileSizeLow);
453 m_CurFile.m_bIsDirectory = (data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
454 m_CurFile.m_sParentPath = m_sCurPath;
455 m_CurFile.m_sName = data.cFileName;
456 m_CurFile.m_LastModificationTime = xiiTimestamp::MakeFromInt(FileTimeToEpoch(data.ftLastWriteTime), xiiSIUnitOfTime::Microsecond);
457
458 m_Data.m_Handles.PushBack(hSearch);
459
460 if ((m_CurFile.m_sName == "..") || (m_CurFile.m_sName == "."))
461 return ReturnCallInternalNext; // will search for the next file or folder that is not ".." or "." ; might return false though
462
463 if (m_CurFile.m_bIsDirectory)
464 {
465 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFolders))
466 return ReturnCallInternalNext;
467 }
468 else
469 {
470 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFiles))
471 return ReturnCallInternalNext;
472 }
473
474 return ReturnSuccess;
475 }
476
477 // if the recursion did not work, just iterate in this folder further
478 }
479
480 WIN32_FIND_DATAW data;
481 if (!FindNextFileW(m_Data.m_Handles.PeekBack(), &data))
482 {
483 // nothing found in this directory anymore
484 FindClose(m_Data.m_Handles.PeekBack());
485 m_Data.m_Handles.PopBack();
486
487 if (m_Data.m_Handles.IsEmpty())
488 return ReturnFailure;
489
490 m_sCurPath.PathParentDirectory();
491 if (m_sCurPath.EndsWith("/"))
492 {
493 m_sCurPath.Shrink(0, 1); // Remove trailing /
494 }
495
496 return ReturnCallInternalNext;
497 }
498
499 m_CurFile.m_uiFileSize = HighLowToUInt64(data.nFileSizeHigh, data.nFileSizeLow);
500 m_CurFile.m_bIsDirectory = (data.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0;
501 m_CurFile.m_sParentPath = m_sCurPath;
502 m_CurFile.m_sName = data.cFileName;
503 m_CurFile.m_LastModificationTime = xiiTimestamp::MakeFromInt(FileTimeToEpoch(data.ftLastWriteTime), xiiSIUnitOfTime::Microsecond);
504
505 if ((m_CurFile.m_sName == "..") || (m_CurFile.m_sName == "."))
506 return ReturnCallInternalNext;
507
508 if (m_CurFile.m_bIsDirectory)
509 {
510 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFolders))
511 return ReturnCallInternalNext;
512 }
513 else
514 {
515 if (!m_Flags.IsSet(xiiFileSystemIteratorFlags::ReportFiles))
516 return ReturnCallInternalNext;
517 }
518
519 return ReturnSuccess;
520}
