34template <typename Type>
35void xiiBoundingSphereTemplate<Type>::ExpandToInclude(const xiiBoundingBoxTemplate<Type>& rhs)
36{
37 // compute the min and max extends of the AABB relative to the sphere (sphere center is the new origin)
38 const xiiVec3 vDiffMax = rhs.m_vMax - m_vCenter;
39 const xiiVec3 vDiffMin = rhs.m_vMin - m_vCenter;
40
41 // compute the absolute distance to each AABB extremum, per axis
42 const xiiVec3 vDiffMaxAbs(xiiMath::Abs(vDiffMax.x), xiiMath::Abs(vDiffMax.y), xiiMath::Abs(vDiffMax.z));
43 const xiiVec3 vDiffMinAbs(xiiMath::Abs(vDiffMin.x), xiiMath::Abs(vDiffMin.y), xiiMath::Abs(vDiffMin.z));
44
45 // take the maximum distance for each axis, to compute the point that is the farthest away from the sphere
46 const xiiVec3 vMostDistantPoint = vDiffMinAbs.CompMax(vDiffMaxAbs);
47
48 const Type fDistSQR = vMostDistantPoint.GetLengthSquared();
49
50 if (xiiMath::Square(m_fRadius) < fDistSQR)
51 m_fRadius = xiiMath::Sqrt(fDistSQR);
52}

62template <typename Type>
63bool xiiBoundingSphereTemplate<Type>::Contains(const xiiBoundingBoxTemplate<Type>& rhs) const
64{
65 // compute the min and max extends of the AABB relative to the sphere (sphere center is the new origin)
66 const xiiVec3 vDiffMax = rhs.m_vMax - m_vCenter;
67 const xiiVec3 vDiffMin = rhs.m_vMin - m_vCenter;
68
69 // compute the absolute distance to each AABB extremum, per axis
70 const xiiVec3 vDiffMaxAbs(xiiMath::Abs(vDiffMax.x), xiiMath::Abs(vDiffMax.y), xiiMath::Abs(vDiffMax.z));
71 const xiiVec3 vDiffMinAbs(xiiMath::Abs(vDiffMin.x), xiiMath::Abs(vDiffMin.y), xiiMath::Abs(vDiffMin.z));
72
73 // take the maximum distance for each axis, to compute the point that is the farthest away from the sphere
74 const xiiVec3 vMostDistantPoint = vDiffMinAbs.CompMax(vDiffMaxAbs);
75
76 // if the squared length of that point is still smaller than the sphere radius, it is inside the sphere
77 // and thus the whole AABB is inside the sphere
78 return vMostDistantPoint.GetLengthSquared() <= m_fRadius * m_fRadius;
79}

186template <typename Type>
187xiiMat3Template<Type> xiiMat3Template<Type>::MakeAxisRotation(const xiiVec3Template<Type>& vAxis, xiiAngleTemplate<Type> angle)
188{
189 XII_ASSERT_DEBUG(vAxis.IsNormalized(0.1f), "vAxis must be normalized.");
190
191 const Type cos = xiiMath::Cos(angle);
192 const Type sin = xiiMath::Sin(angle);
193 const Type oneminuscos = (Type)1 - cos;
194
195 const Type xy = vAxis.x * vAxis.y;
196 const Type xz = vAxis.x * vAxis.z;
197 const Type yz = vAxis.y * vAxis.z;
198
199 const Type xsin = vAxis.x * sin;
200 const Type ysin = vAxis.y * sin;
201 const Type zsin = vAxis.z * sin;
202
203 const Type onecos_xy = oneminuscos * xy;
204 const Type onecos_xz = oneminuscos * xz;
205 const Type onecos_yz = oneminuscos * yz;
206
207 xiiMat3Template<Type> res;
208
209 // Column 1
210 res.Element(0, 0) = cos + (oneminuscos * (vAxis.x * vAxis.x));
211 res.Element(0, 1) = onecos_xy + zsin;
212 res.Element(0, 2) = onecos_xz - ysin;
213
214 // Column 2 )
215 res.Element(1, 0) = onecos_xy - zsin;
216 res.Element(1, 1) = cos + (oneminuscos * (vAxis.y * vAxis.y));
217 res.Element(1, 2) = onecos_yz + xsin;
218
219 // Column 3 )
220 res.Element(2, 0) = onecos_xz + ysin;
221 res.Element(2, 1) = onecos_yz - xsin;
222 res.Element(2, 2) = cos + (oneminuscos * (vAxis.z * vAxis.z));
223
224 return res;
225}

227template <typename Type>
228xiiResult xiiMat3Template<Type>::Invert(Type fEpsilon)
229{
230 const Type fDet = Element(0, 0) * (Element(2, 2) * Element(1, 1) - Element(1, 2) * Element(2, 1)) -
231 Element(0, 1) * (Element(2, 2) * Element(1, 0) - Element(1, 2) * Element(2, 0)) +
232 Element(0, 2) * (Element(2, 1) * Element(1, 0) - Element(1, 1) * Element(2, 0));
233
234 if (xiiMath::IsZero(fDet, fEpsilon))
235 return XII_FAILURE;
236
237 const Type fOneDivDet = (Type)1 / fDet;
238
239 xiiMat3Template<Type> Inverse;
240
241 Inverse.Element(0, 0) = (Element(2, 2) * Element(1, 1) - Element(1, 2) * Element(2, 1));
242 Inverse.Element(0, 1) = -(Element(2, 2) * Element(0, 1) - Element(0, 2) * Element(2, 1));
243 Inverse.Element(0, 2) = (Element(1, 2) * Element(0, 1) - Element(0, 2) * Element(1, 1));
244
245 Inverse.Element(1, 0) = -(Element(2, 2) * Element(1, 0) - Element(1, 2) * Element(2, 0));
246 Inverse.Element(1, 1) = (Element(2, 2) * Element(0, 0) - Element(0, 2) * Element(2, 0));
247 Inverse.Element(1, 2) = -(Element(1, 2) * Element(0, 0) - Element(0, 2) * Element(1, 0));
248
249 Inverse.Element(2, 0) = (Element(2, 1) * Element(1, 0) - Element(1, 1) * Element(2, 0));
250 Inverse.Element(2, 1) = -(Element(2, 1) * Element(0, 0) - Element(0, 1) * Element(2, 0));
251 Inverse.Element(2, 2) = (Element(1, 1) * Element(0, 0) - Element(0, 1) * Element(1, 0));
252
253 *this = Inverse * fOneDivDet;
254 return XII_SUCCESS;
255}

307template <typename Type>
308xiiResult xiiMat4Template<Type>::Invert(Type fEpsilon)
309{
310 xiiMat4Template<Type> Inverse;
311
312 const Type fDet = GetDeterminantOf4x4Matrix(*this);
313
314 if (xiiMath::IsZero(fDet, fEpsilon))
315 return XII_FAILURE;
316
317 Type fOneDivDet = xiiMath::Invert(fDet);
318
319 for (xiiInt32 i = 0; i < 4; ++i)
320 {
321
322 Inverse.Element(i, 0) = GetDeterminantOf3x3SubMatrix(*this, i, 0) * fOneDivDet;
323 fOneDivDet = -fOneDivDet;
324 Inverse.Element(i, 1) = GetDeterminantOf3x3SubMatrix(*this, i, 1) * fOneDivDet;
325 fOneDivDet = -fOneDivDet;
326 Inverse.Element(i, 2) = GetDeterminantOf3x3SubMatrix(*this, i, 2) * fOneDivDet;
327 fOneDivDet = -fOneDivDet;
328 Inverse.Element(i, 3) = GetDeterminantOf3x3SubMatrix(*this, i, 3) * fOneDivDet;
329 }
330
331 *this = Inverse;
332 return XII_SUCCESS;
333}
