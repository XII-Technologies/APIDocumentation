348template <typename ArrayType, typename ValueType>
349xiiResult xiiStreamReader::ReadArray(xiiArrayBase<ValueType, ArrayType>& ref_array)
350{
351 xiiUInt64 uiCount = 0;
352 XII_SUCCEED_OR_RETURN(ReadQWordValue(&uiCount));
353
354 if (uiCount < xiiMath::MaxValue<xiiUInt32>())
355 {
356 ref_array.Clear();
357
358 if (uiCount > 0)
359 {
360 static_cast<ArrayType&>(ref_array).SetCount(static_cast<xiiUInt32>(uiCount));
361
362 XII_SUCCEED_OR_RETURN(xiiStreamReaderUtil::DeserializeArray<ValueType>(*this, ref_array.GetData(), uiCount));
363 }
364
365 return XII_SUCCESS;
366 }
367 else
368 {
369 // Containers currently use 32 bit for counts internally. Value from file is too large.
370 return XII_FAILURE;
371 }
372}
