16 // Helper function to detect a 64-bit Windows
17 bool Is64BitWindows()
18 {
19 #if defined(_WIN64)
20  return true; // 64-bit programs run only on Win64 (although if we get to Win128 this will be wrong probably)
21 #elif defined(_WIN32)
22  // 32-bit programs run on both 32-bit and 64-bit Windows
23  // Note that we used IsWow64Process before which is not available on UWP.
24  SYSTEM_INFO info;
25  GetNativeSystemInfo(&info);
26  // According to documentation: "The processor architecture of the installed operating system."
27  return info.wProcessorArchitecture == PROCESSOR_ARCHITECTURE_AMD64;
28 #else
29  return false; // Win64 does not support Win16
30 #endif
31 }

124 float xiiSystemInformation::GetCPUUtilization() const
125 {
126 #if XII_ENABLED(XII_PLATFORM_WINDOWS_DESKTOP)
127 
128  LARGE_INTEGER kernel, user, idle;
129  GetSystemTimes((FILETIME*)&idle, (FILETIME*)&kernel, (FILETIME*)&user);
130 
131  static thread_local uint64_t lastKernel = 0u, lastIdle = 0u, lastUser = 0u;
132 
133  auto kernelTime = kernel.QuadPart - lastKernel;
134  auto idleTime = idle.QuadPart - lastIdle;
135  auto userTime = user.QuadPart - lastUser;
136 
137  lastKernel = kernel.QuadPart;
138  lastUser = user.QuadPart;
139  lastIdle = idle.QuadPart;
140 
141  auto util = static_cast<float>(kernelTime + userTime - idleTime) / (kernelTime + userTime);
142 
143  return xiiMath::Clamp(util, 0.f, 1.f) * 100.f;
144 
145 #else
146  XII_ASSERT_NOT_IMPLEMENTED;
147  return 0.0f;
