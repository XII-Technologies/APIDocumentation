62 xiiResult xiiDirectoryWatcher::OpenDirectory(xiiStringView sAbsolutePath, xiiBitflags<Watch> whatToWatch)
63 {
64  XII_ASSERT_DEV(m_sDirectoryPath.IsEmpty(), "Directory already open, call CloseDirectory first!");
65  xiiStringBuilder sPath(sAbsolutePath);
66  sPath.MakeCleanPath();
67  sPath.Trim("/");
68 
69  m_pImpl->m_whatToWatch = whatToWatch;
70  m_pImpl->m_filter = FILE_NOTIFY_CHANGE_FILE_NAME;
71  if (whatToWatch.IsSet(Watch::Writes) || whatToWatch.AreAllSet(Watch::Deletes | Watch::Subdirectories))
72  {
73  m_pImpl->m_filter |= FILE_NOTIFY_CHANGE_LAST_WRITE;
74  m_pImpl->m_mirror = XII_DEFAULT_NEW(xiiFileSystemMirrorType);
75  m_pImpl->m_mirror->AddDirectory(sPath).AssertSuccess();
76  }
77 
78  if (whatToWatch.IsAnySet(Watch::Deletes | Watch::Creates | Watch::Renames))
79  {
80  m_pImpl->m_filter |= FILE_NOTIFY_CHANGE_DIR_NAME;
81  }
82 
83  m_pImpl->m_directoryHandle = CreateFileW(xiiDosDevicePath(sPath), FILE_LIST_DIRECTORY, FILE_SHARE_READ | FILE_SHARE_WRITE | FILE_SHARE_DELETE,
84  nullptr, OPEN_EXISTING, FILE_FLAG_BACKUP_SEMANTICS | FILE_FLAG_OVERLAPPED, nullptr);
85  if (m_pImpl->m_directoryHandle == INVALID_HANDLE_VALUE)
86  {
87  return XII_FAILURE;
88  }
89 
90  m_pImpl->m_overlappedEvent = CreateEvent(nullptr, TRUE, FALSE, nullptr);
91  if (m_pImpl->m_overlappedEvent == INVALID_HANDLE_VALUE)
92  {
93  return XII_FAILURE;
94  }
95 
96  m_pImpl->DoRead();
97  m_sDirectoryPath = sPath;
98 
99  return XII_SUCCESS;
100 }

129 void xiiDirectoryWatcher::EnumerateChanges(EnumerateChangesFunction func, xiiTime waitUpTo)
130 {
131  XII_ASSERT_DEV(!m_sDirectoryPath.IsEmpty(), "No directory opened!");
132  while (WaitForSingleObject(m_pImpl->m_overlappedEvent, static_cast<DWORD>(waitUpTo.GetMilliseconds())) == WAIT_OBJECT_0)
133  {
134  waitUpTo = xiiTime::Zero(); // only wait on the first call to GetQueuedCompletionStatus
135 
136  DWORD numberOfBytes = 0;
137  GetOverlappedResult(m_pImpl->m_directoryHandle, &m_pImpl->m_overlapped, &numberOfBytes, FALSE);
138 
139  // Copy the buffer
140  xiiHybridArray<xiiUInt8, 4096> buffer;
141  buffer.SetCountUninitialized(numberOfBytes);
142  buffer.GetArrayPtr().CopyFrom(m_pImpl->m_buffer.GetArrayPtr().GetSubArray(0, numberOfBytes));
143 
144  // Reissue the read request
145  m_pImpl->DoRead();
146 
147  if (numberOfBytes == 0)
148  {
149  return;
150  }
151 
152  const xiiBitflags<xiiDirectoryWatcher::Watch> whatToWatch = m_pImpl->m_whatToWatch;
153 
154  xiiFileSystemMirrorType* mirror = m_pImpl->m_mirror.Borrow();
155 
156  MoveEvent lastMoveFrom;
157 
158  // Progress the messages
159  auto info = (const FILE_NOTIFY_EXTENDED_INFORMATION*)buffer.GetData();
160  while (true)
161  {
162  auto directory = xiiArrayPtr<const WCHAR>(info->FileName, info->FileNameLength / sizeof(WCHAR));
163  xiiInt32 bytesNeeded = WideCharToMultiByte(CP_UTF8, 0, directory.GetPtr(), directory.GetCount(), nullptr, 0, nullptr, nullptr);
164  if (bytesNeeded > 0)
165  {
166  xiiHybridArray<char, 1024> dir;
167  dir.SetCountUninitialized(bytesNeeded);
168  WideCharToMultiByte(CP_UTF8, 0, directory.GetPtr(), directory.GetCount(), dir.GetData(), dir.GetCount(), nullptr, nullptr);
169  xiiDirectoryWatcherAction action = xiiDirectoryWatcherAction::None;
170  bool fireEvent = false;
171 
172  xiiStringBuilder eventFilePath = m_sDirectoryPath;
173  eventFilePath.AppendPath(xiiStringView(dir.GetData(), dir.GetCount()));
174  eventFilePath.MakeCleanPath();
175 
176  if ((info->FileAttributes & FILE_ATTRIBUTE_DIRECTORY) == 0)
177  {
178  switch (info->Action)
179  {
180  case FILE_ACTION_ADDED:
181  DEBUG_LOG("FILE_ACTION_ADDED {} ({})", eventFilePath, info->LastModificationTime.QuadPart);
182  action = xiiDirectoryWatcherAction::Added;
183  fireEvent = whatToWatch.IsSet(xiiDirectoryWatcher::Watch::Creates);
184  if (mirror)
185  {
186  bool fileAlreadyExists = false;
187  mirror->AddFile(eventFilePath.GetData(), true, &fileAlreadyExists, nullptr).AssertSuccess();
188  if (fileAlreadyExists)
189  {
190  fireEvent = false;
191  }
192  }
193  break;
194  case FILE_ACTION_REMOVED:
195  DEBUG_LOG("FILE_ACTION_REMOVED {} ({})", eventFilePath, info->LastModificationTime.QuadPart);
196  action = xiiDirectoryWatcherAction::Removed;
197  fireEvent = whatToWatch.IsSet(xiiDirectoryWatcher::Watch::Deletes);
198  if (mirror)
199  {
200  mirror->RemoveFile(eventFilePath.GetData()).AssertSuccess();
201  }
202  break;
203  case FILE_ACTION_MODIFIED:
204  {
205  DEBUG_LOG("FILE_ACTION_MODIFIED {} ({})", eventFilePath, info->LastModificationTime.QuadPart);
206  action = xiiDirectoryWatcherAction::Modified;
207  fireEvent = whatToWatch.IsAnySet(xiiDirectoryWatcher::Watch::Writes);
208  bool fileAreadyKnown = false;
209  bool addPending = false;
210  if (mirror)
211  {
212  mirror->AddFile(eventFilePath.GetData(), false, &fileAreadyKnown, &addPending).AssertSuccess();
213  }
214  if (fileAreadyKnown && addPending)
215  {
216  fireEvent = false;
217  }
218  }
219  break;
220  case FILE_ACTION_RENAMED_OLD_NAME:
221  DEBUG_LOG("FILE_ACTION_RENAMED_OLD_NAME {} ({})", eventFilePath, info->LastModificationTime.QuadPart);
222  XII_ASSERT_DEV(lastMoveFrom.IsEmpty(), "there should be no pending move from");
223  action = xiiDirectoryWatcherAction::RenamedOldName;
224  fireEvent = whatToWatch.IsAnySet(xiiDirectoryWatcher::Watch::Renames);
225  XII_ASSERT_DEV(lastMoveFrom.IsEmpty(), "there should be no pending last move from");
226  lastMoveFrom = {eventFilePath, false};
227  break;
228  case FILE_ACTION_RENAMED_NEW_NAME:
229  DEBUG_LOG("FILE_ACTION_RENAMED_NEW_NAME {} ({})", eventFilePath, info->LastModificationTime.QuadPart);
230  action = xiiDirectoryWatcherAction::RenamedNewName;
231  fireEvent = whatToWatch.IsAnySet(xiiDirectoryWatcher::Watch::Renames);
232  XII_ASSERT_DEV(!lastMoveFrom.IsEmpty() && !lastMoveFrom.isDirectory, "last move from doesn't match");
233  if (mirror)
234  {
235  mirror->RemoveFile(lastMoveFrom.path).AssertSuccess();
236  mirror->AddFile(eventFilePath, false, nullptr, nullptr).AssertSuccess();
237  }
238  lastMoveFrom.Clear();
239  break;
240  }
241 
242  if (fireEvent)
243  {
244  func(eventFilePath, action, xiiDirectoryWatcherType::File);
245  }
246  }
247  else
248  {
249  switch (info->Action)
250  {
251  case FILE_ACTION_ADDED:
252  {
253  DEBUG_LOG("DIR_ACTION_ADDED {}", eventFilePath);
254  bool directoryAlreadyKnown = false;
255  if (mirror)
256  {
257  mirror->AddDirectory(eventFilePath, &directoryAlreadyKnown).AssertSuccess();
258  }
259 
260  if (whatToWatch.IsSet(Watch::Creates) && !directoryAlreadyKnown)
261  {
262  func(eventFilePath, xiiDirectoryWatcherAction::Added, xiiDirectoryWatcherType::Directory);
263  }
264 
265  // Whenever we add a directory we might be "to late" to see changes inside it.
266  // So iterate the file system and make sure we track all files / subdirectories
267  xiiFileSystemIterator subdirIt;
268 
269  subdirIt.StartSearch(eventFilePath.GetData(),
270  whatToWatch.IsSet(xiiDirectoryWatcher::Watch::Subdirectories) ? xiiFileSystemIteratorFlags::ReportFilesAndFoldersRecursive : xiiFileSystemIteratorFlags::ReportFiles);
271 
272  xiiStringBuilder tmpPath2;
273  for (; subdirIt.IsValid(); subdirIt.Next())
274  {
275  const xiiFileStats& stats = subdirIt.GetStats();
276  stats.GetFullPath(tmpPath2);
277  if (stats.m_bIsDirectory)
278  {
279  directoryAlreadyKnown = false;
280  if (mirror)
281  {
282  mirror->AddDirectory(tmpPath2, &directoryAlreadyKnown).AssertSuccess();
283  }
284  if (whatToWatch.IsSet(xiiDirectoryWatcher::Watch::Creates) && !directoryAlreadyKnown)
285  {
286  func(tmpPath2, xiiDirectoryWatcherAction::Added, xiiDirectoryWatcherType::Directory);
287  }
288  }
289  else
290  {
291  bool fileExistsAlready = false;
292  if (mirror)
293  {
294  mirror->AddFile(tmpPath2, false, &fileExistsAlready, nullptr).AssertSuccess();
295  }
296  if (whatToWatch.IsSet(xiiDirectoryWatcher::Watch::Creates) && !fileExistsAlready)
297  {
298  func(tmpPath2, xiiDirectoryWatcherAction::Added, xiiDirectoryWatcherType::File);
299  }
300  }
301  }
302  }
303  break;
304  case FILE_ACTION_REMOVED:
305  DEBUG_LOG("DIR_ACTION_REMOVED {}", eventFilePath);
306  if (whatToWatch.IsSet(Watch::Deletes))
307  {
308  if (mirror && whatToWatch.IsSet(Watch::Subdirectories))
309  {
310  mirror->Enumerate(eventFilePath, [&](const xiiStringBuilder& sPath, xiiFileSystemMirrorType::Type type) {
311  func(sPath, xiiDirectoryWatcherAction::Removed, (type == xiiFileSystemMirrorType::Type::File) ? xiiDirectoryWatcherType::File : xiiDirectoryWatcherType::Directory);
312  })
313  .AssertSuccess();
314  }
315  func(eventFilePath, xiiDirectoryWatcherAction::Removed, xiiDirectoryWatcherType::Directory);
316  }
317  if (mirror)
318  {
319  mirror->RemoveDirectory(eventFilePath).AssertSuccess();
320  }
321  break;
322  case FILE_ACTION_RENAMED_OLD_NAME:
323  DEBUG_LOG("DIR_ACTION_OLD_NAME {}", eventFilePath);
324  XII_ASSERT_DEV(lastMoveFrom.IsEmpty(), "there should be no pending move from");
325  lastMoveFrom = {eventFilePath, true};
326  break;
327  case FILE_ACTION_RENAMED_NEW_NAME:
328  DEBUG_LOG("DIR_ACTION_NEW_NAME {}", eventFilePath);
329  XII_ASSERT_DEV(!lastMoveFrom.IsEmpty(), "rename old name and rename new name should always appear in pairs");
330  if (mirror)
331  {
332  mirror->MoveDirectory(lastMoveFrom.path, eventFilePath).AssertSuccess();
333  }
334  if (whatToWatch.IsSet(Watch::Renames))
335  {
336  func(lastMoveFrom.path, xiiDirectoryWatcherAction::RenamedOldName, xiiDirectoryWatcherType::Directory);
337  func(eventFilePath, xiiDirectoryWatcherAction::RenamedNewName, xiiDirectoryWatcherType::Directory);
338  }
339  lastMoveFrom.Clear();
340  break;
341  default:
342  break;
343  }
344  }
345  }
346  if (info->NextEntryOffset == 0)
347  break;
348  else
349  info = (const FILE_NOTIFY_EXTENDED_INFORMATION*)(((xiiUInt8*)info) + info->NextEntryOffset);
350  }
351  }
352 }

355 void xiiDirectoryWatcher::EnumerateChanges(xiiArrayPtr<xiiDirectoryWatcher*> watchers, EnumerateChangesFunction func, xiiTime waitUpTo)
356 {
357  xiiHybridArray<HANDLE, 16> events;
358  events.SetCount(watchers.GetCount());
359 
360  for (xiiUInt32 i = 0; i < watchers.GetCount(); ++i)
361  {
362  events[i] = watchers[i]->m_pImpl->m_overlappedEvent;
363  }
364 
365  // Wait for any of the watchers to have some data ready
366  if (WaitForMultipleObjects(events.GetCount(), events.GetData(), FALSE, static_cast<DWORD>(waitUpTo.GetMilliseconds())) == WAIT_TIMEOUT)
367  {
368  return;
369  }
370 
371  // Iterate all of them to make sure we report all changes up to this point.
372  for (xiiDirectoryWatcher* watcher : watchers)
373  {
374  watcher->EnumerateChanges(func);
375  }
