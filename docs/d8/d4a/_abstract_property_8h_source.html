92 template <class Type>
93 static xiiBitflags<xiiPropertyFlags> GetParameterFlags()
94 {
95 using CleanType = typename xiiTypeTraits<Type>::NonConstReferencePointerType;
96
97 xiiBitflags<xiiPropertyFlags> flags;
98 constexpr xiiVariantType::Enum type = static_cast<xiiVariantType::Enum>(xiiVariantTypeDeduction<CleanType>::value);
99 if constexpr (std::is_same<CleanType, xiiVariant>::value || std::is_same<Type, const char*>::value || // We treat const char* as a basic type and not a pointer.
100 (type >= xiiVariantType::FirstStandardType && type <= xiiVariantType::LastStandardType))
101 flags.Add(xiiPropertyFlags::StandardType);
102 else if constexpr (xiiIsEnum<CleanType>::value)
103 flags.Add(xiiPropertyFlags::IsEnum);
104 else if constexpr (xiiIsBitflags<CleanType>::value)
105 flags.Add(xiiPropertyFlags::Bitflags);
106 else
107 flags.Add(xiiPropertyFlags::Class);
108
109 if constexpr (std::is_const<typename xiiTypeTraits<Type>::NonReferencePointerType>::value)
110 flags.Add(xiiPropertyFlags::Const);
111
112 if constexpr (std::is_pointer<Type>::value && !std::is_same<Type, const char*>::value)
113 flags.Add(xiiPropertyFlags::Pointer);
114
115 if constexpr (std::is_reference<Type>::value)
116 flags.Add(xiiPropertyFlags::Reference);
117
118 return flags;
119 }
120};
