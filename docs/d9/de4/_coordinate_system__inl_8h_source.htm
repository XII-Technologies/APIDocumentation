20  mSourceFromId.SetColumn(0, source.m_vRightDir);

24  Type fTargetScale = target.m_vForwardDir.GetLengthSquared();
25  XII_ASSERT_DEV(xiiMath::IsEqual(fTargetScale, target.m_vRightDir.GetLengthSquared(), xiiMath::DefaultEpsilon<Type>()),
26  "Only uniformly scaled coordinate systems are supported");
27  XII_ASSERT_DEV(xiiMath::IsEqual(fTargetScale, target.m_vUpDir.GetLengthSquared(), xiiMath::DefaultEpsilon<Type>()),
28  "Only uniformly scaled coordinate systems are supported");
29 
30  xiiMat3Template<Type> mTargetFromId;
31  mTargetFromId.SetColumn(0, target.m_vRightDir);
32  mTargetFromId.SetColumn(1, target.m_vUpDir);
33  mTargetFromId.SetColumn(2, target.m_vForwardDir);
34 
35  m_mSourceToTarget = mTargetFromId * mSourceFromId.GetInverse();
36  m_mSourceToTarget.SetColumn(0, m_mSourceToTarget.GetColumn(0).GetNormalized());
37  m_mSourceToTarget.SetColumn(1, m_mSourceToTarget.GetColumn(1).GetNormalized());
38  m_mSourceToTarget.SetColumn(2, m_mSourceToTarget.GetColumn(2).GetNormalized());
39 
40  m_fWindingSwap = m_mSourceToTarget.GetDeterminant() < 0 ? static_cast<Type>(-1) : static_cast<Type>(1);
41  m_fSourceToTargetScale = static_cast<Type>(1) / xiiMath::Sqrt(fSourceScale) * xiiMath::Sqrt(fTargetScale);
42  m_mTargetToSource = m_mSourceToTarget.GetInverse();

44 }
