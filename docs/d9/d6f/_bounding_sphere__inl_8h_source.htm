44template <typename Type>
45XII_FORCE_INLINE xiiBoundingSphereTemplate<Type> xiiBoundingSphereTemplate<Type>::MakeFromPoints(const xiiVec3Template<Type>* pPoints, xiiUInt32 uiNumPoints, xiiUInt32 uiStride /*= sizeof(xiiVec3Template<Type>)*/)
46{
47 XII_ASSERT_DEBUG(pPoints != nullptr, "The array must not be empty.");
48 XII_ASSERT_DEBUG(uiStride >= sizeof(xiiVec3Template<Type>), "The data must not overlap.");
49 XII_ASSERT_DEBUG(uiNumPoints > 0, "The array must contain at least one point.");
50
51 const xiiVec3Template<Type>* pCur = &pPoints[0];
52
53 xiiVec3Template<Type> vCenter(0.0f);
54
55 for (xiiUInt32 i = 0; i < uiNumPoints; ++i)
56 {
57 vCenter += *pCur;
58 pCur = xiiMemoryUtils::AddByteOffset(pCur, uiStride);
59 }
60
61 vCenter /= (Type)uiNumPoints;
62
63 Type fMaxDistSQR = 0.0f;
64
65 pCur = &pPoints[0];
66 for (xiiUInt32 i = 0; i < uiNumPoints; ++i)
67 {
68 const Type fDistSQR = (*pCur - vCenter).GetLengthSquared();
69 fMaxDistSQR = xiiMath::Max(fMaxDistSQR, fDistSQR);
70
71 pCur = xiiMemoryUtils::AddByteOffset(pCur, uiStride);
72 }
73
74 xiiBoundingSphereTemplate<Type> res;
75 res.m_vCenter = vCenter;
76 res.m_fRadius = xiiMath::Sqrt(fMaxDistSQR);
77
78 XII_ASSERT_DEBUG(res.IsValid(), "The point cloud contained corrupted data.");
79
80 return res;
81}

168template <typename Type>
169void xiiBoundingSphereTemplate<Type>::ScaleFromOrigin(const xiiVec3Template<Type>& vScale)
170{
171 XII_ASSERT_DEBUG(vScale.x >= 0.0f, "Cannot invert the sphere.");
172 XII_ASSERT_DEBUG(vScale.y >= 0.0f, "Cannot invert the sphere.");
173 XII_ASSERT_DEBUG(vScale.z >= 0.0f, "Cannot invert the sphere.");
174
175 m_vCenter = m_vCenter.CompMul(vScale);
176
177 // scale the radius by the maximum scaling factor (the sphere cannot become an ellipsoid,
178 // so to be a 'bounding' sphere, it should be as large as possible
179 m_fRadius *= xiiMath::Max(vScale.x, vScale.y, vScale.z);
180}

230template <typename Type>
231const xiiVec3Template<Type> xiiBoundingSphereTemplate<Type>::GetClampedPoint(const xiiVec3Template<Type>& vPoint)
232{
233 const xiiVec3Template<Type> vDir = vPoint - m_vCenter;
234 const Type fDistSQR = vDir.GetLengthSquared();
235
236 // return the point, if it is already inside the sphere
237 if (fDistSQR <= xiiMath::Square(m_fRadius))
238 return vPoint;
239
240 // otherwise return a point on the surface of the sphere
241
242 const Type fLength = xiiMath::Sqrt(fDistSQR);
243
244 return m_vCenter + m_fRadius * (vDir / fLength);
245}

336template <typename Type>
337bool xiiBoundingSphereTemplate<Type>::GetRayIntersection(const xiiVec3Template<Type>& vRayStartPos, const xiiVec3Template<Type>& vRayDirNormalized, Type* out_pIntersectionDistance /* = nullptr */, xiiVec3Template<Type>* out_pIntersection /* = nullptr */) const
338{
339 XII_ASSERT_DEBUG(vRayDirNormalized.IsNormalized(), "The ray direction must be normalized.");
340
341 // Ugly Code taken from 'Real Time Rendering First Edition' Page 299
342
343 const Type fRadiusSQR = xiiMath::Square(m_fRadius);
344 const xiiVec3Template<Type> vRelPos = m_vCenter - vRayStartPos;
345
346 const Type d = vRelPos.Dot(vRayDirNormalized);
347 const Type fRelPosLenSQR = vRelPos.GetLengthSquared();
348
349 if (d < 0.0f && fRelPosLenSQR > fRadiusSQR)
350 return false;
351
352 const Type m2 = fRelPosLenSQR - xiiMath::Square(d);
353
354 if (m2 > fRadiusSQR)
355 return false;
356
357 const Type q = xiiMath::Sqrt(fRadiusSQR - m2);
358
359 Type fIntersectionTime;
360
361 if (fRelPosLenSQR > fRadiusSQR)
362 fIntersectionTime = d - q;
363 else
364 fIntersectionTime = d + q;
365
366 if (out_pIntersectionDistance)
367 *out_pIntersectionDistance = fIntersectionTime;
368 if (out_pIntersection)
369 *out_pIntersection = vRayStartPos + vRayDirNormalized * fIntersectionTime;
370
371 return true;
372}
