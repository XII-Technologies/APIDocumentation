41 {
42  xiiBitflags<xiiGALPipelineResourceFlags> pipelineResourceFlags = xiiGALPipelineResourceFlags::None;
43 
44  switch (type)
45  {
46  case xiiGALShaderResourceType::ConstantBuffer:
47  pipelineResourceFlags |= xiiGALPipelineResourceFlags::NoDynamicBuffers | xiiGALPipelineResourceFlags::RuntimeArray;
48  break;
49  case xiiGALShaderResourceType::TextureSRV:
50  pipelineResourceFlags |= xiiGALPipelineResourceFlags::CombinedSampler | xiiGALPipelineResourceFlags::RuntimeArray;
51  break;
52  case xiiGALShaderResourceType::BufferSRV:
53  pipelineResourceFlags |= xiiGALPipelineResourceFlags::NoDynamicBuffers | xiiGALPipelineResourceFlags::Formattedbuffer | xiiGALPipelineResourceFlags::RuntimeArray;
54  break;
55  case xiiGALShaderResourceType::TextureUAV:
56  pipelineResourceFlags |= xiiGALPipelineResourceFlags::RuntimeArray;
57  break;
58  case xiiGALShaderResourceType::BufferUAV:
59  pipelineResourceFlags |= xiiGALPipelineResourceFlags::NoDynamicBuffers | xiiGALPipelineResourceFlags::Formattedbuffer | xiiGALPipelineResourceFlags::RuntimeArray;
60  break;
61  case xiiGALShaderResourceType::Sampler:
62  pipelineResourceFlags |= xiiGALPipelineResourceFlags::RuntimeArray;
63  break;
64  case xiiGALShaderResourceType::InputAttachment:
65  pipelineResourceFlags |= xiiGALPipelineResourceFlags::GeneralInputAttachment;
66  break;
67  case xiiGALShaderResourceType::AccelerationStructure:
68  pipelineResourceFlags |= xiiGALPipelineResourceFlags::RuntimeArray;
69  break;
70 
71  XII_DEFAULT_CASE_NOT_IMPLEMENTED;
72  }
73  return pipelineResourceFlags;
74 }
