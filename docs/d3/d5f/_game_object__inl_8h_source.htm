387XII_ALWAYS_INLINE void xiiGameObject::SetGlobalTransform(const xiiSimdTransform& transform)
388{
389 UpdateLastGlobalTransform();
390
391 m_pTransformationData->m_globalTransform = transform;
392
393 // xiiTransformTemplate<Type>::SetLocalTransform will produce NaNs in w components
394 // of pos and scale if scale.w is not set to 1 here. This only affects builds that
395 // use XII_SIMD_IMPLEMENTATION_FPU, e.g. arm atm.
396 m_pTransformationData->m_globalTransform.m_Scale.SetW(1.0f);
397 m_pTransformationData->UpdateLocalTransform();
398
399 if (IsStatic())
400 {
401 UpdateGlobalTransformAndBoundsRecursive();
402 }
403}
