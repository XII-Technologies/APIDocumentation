39 timeinfo.tm_mday = m_uiDay; /* day of the month - [1,31] */

41 timeinfo.tm_mon = m_uiMonth - 1; /* months since January - [0,11] */
42 timeinfo.tm_year = m_iYear - 1900; /* years since 1900 */
43 timeinfo.tm_isdst = 0; /* daylight savings time flag */
44 timeinfo.tm_zone = "GMT";
45
46 tm timeinfoCopy = timeinfo;
47
48 time_t iTimeStamp = mktime(&timeinfo);
49 // mktime may have 'patched' our time to be valid, we don't want that to count as a valid date.
50 if (iTimeStamp == (time_t)-1 || timeinfoCopy != timeinfo)
51 return xiiTimestamp::MakeInvalid();

54 // Subtract one hour if daylight saving time was activated by mktime.
