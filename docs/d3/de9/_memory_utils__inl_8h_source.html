40template <typename Destination, typename Source>
41XII_ALWAYS_INLINE void xiiMemoryUtils::CopyConstruct(Destination* pDestination, const Source& copy, size_t uiCount)
42{
43 if constexpr (xiiIsPodType<Destination>::value)
44 {
45 static_assert(std::is_same<Destination, Source>::value || (std::is_base_of<Destination, Source>::value == false && std::is_base_of<Source, Destination>::value == false),
46 "Can't copy POD types that are derived from each other. Are you certain any of these types should be POD?");
47
48 const Destination& copyConverted = copy;
49 for (size_t i = 0; i < uiCount; i++)
50 {
51 memcpy(pDestination + i, &copyConverted, sizeof(Destination));
52 }
53 }
54 else
55 {
56 XII_CHECK_CLASS(Destination);
57
58 for (size_t i = 0; i < uiCount; i++)
59 {
60 ::new (pDestination + i) Destination(copy); // Note that until now copy has not been converted to Destination. This allows for calling
61 // specialized constructors if available.
62 }
63 }
64}

275template <typename T>
276XII_ALWAYS_INLINE void xiiMemoryUtils::RelocateOverlapped(T* pDestination, T* pSource, size_t uiCount)
277{
278 if constexpr (xiiGetTypeClass<T>::value == 2) // mem-relocatable
279 {
280 if (pDestination < pSource)
281 {
282 size_t uiDestructCount = pSource - pDestination;
283 Destruct(pDestination, uiDestructCount);
284 }
285 else
286 {
287 size_t uiDestructCount = pDestination - pSource;
288 Destruct(pSource + uiCount, uiDestructCount);
289 }
290 memmove(pDestination, pSource, uiCount * sizeof(T));
291 }
292 else if constexpr (xiiGetTypeClass<T>::value == 1) // POD
293 {
294 memmove(pDestination, pSource, uiCount * sizeof(T));
295 }
296 else
297 {
298 XII_CHECK_CLASS(T);
299
300 if (pDestination == pSource)
301 return;
302
303 if (pDestination < pSource)
304 {
305 for (size_t i = 0; i < uiCount; i++)
306 {
307 pDestination[i] = std::move(pSource[i]);
308 }
309
310 size_t uiDestructCount = pSource - pDestination;
311 Destruct(pSource + uiCount - uiDestructCount, uiDestructCount);
312 }
313 else
314 {
315 for (size_t i = uiCount; i > 0; --i)
316 {
317 pDestination[i - 1] = std::move(pSource[i - 1]);
318 }
319
320 size_t uiDestructCount = pDestination - pSource;
321 Destruct(pSource, uiDestructCount);
322 }
323 }

326template <typename T>
327XII_ALWAYS_INLINE void xiiMemoryUtils::Prepend(T* pDestination, const T& source, size_t uiCount)
328{
329 if constexpr (xiiGetTypeClass<T>::value != 0) // POD or mem-relocatable
330 {
331 memmove(pDestination + 1, pDestination, uiCount * sizeof(T));
332 CopyConstruct(pDestination, source, 1);
333 }
334 else // class
335 {
336 XII_CHECK_CLASS(T);
337
338 if (uiCount > 0)
339 {
340 MoveConstruct(pDestination + uiCount, std::move(pDestination[uiCount - 1]));
341
342 for (size_t i = uiCount - 1; i > 0; --i)
343 {
344 pDestination[i] = std::move(pDestination[i - 1]);
345 }
346
347 *pDestination = source;
348 }
349 else
350 {
351 CopyConstruct(pDestination, source, 1);
352 }
353 }
