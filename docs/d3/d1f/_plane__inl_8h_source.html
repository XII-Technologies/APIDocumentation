85{

252template <typename Type>
253xiiResult xiiPlaneTemplate<Type>::FindSupportPoints(const xiiVec3Template<Type>* const pVertices, xiiInt32 iMaxVertices, xiiInt32& out_i1, xiiInt32& out_i2, xiiInt32& out_i3)
254{
255 const xiiVec3Template<Type> v1 = pVertices[0];
256
257 bool bFoundSecond = false;
258
259 xiiInt32 i = 1;
260 while (i < iMaxVertices)
261 {
262 if (pVertices[i].IsEqual(v1, 0.001f) == false)
263 {
264 bFoundSecond = true;
265 break;
266 }
267
268 ++i;
269 }
270
271 if (!bFoundSecond)
272 return XII_FAILURE;
273
274 const xiiVec3Template<Type> v2 = pVertices[i];
275
276 const xiiVec3Template<Type> vDir1 = (v1 - v2).GetNormalized();
277
278 out_i1 = 0;
279 out_i2 = i;
280
281 ++i;
282
283 while (i < iMaxVertices)
284 {
285 // check for inequality, then for non-collinearity
286 if ((pVertices[i].IsEqual(v2, 0.001f) == false) && (xiiMath::Abs((pVertices[i] - v2).GetNormalized().Dot(vDir1)) < (Type)0.999))
287 {
288 out_i3 = i;
289 return XII_SUCCESS;
290 }
291
292 ++i;
293 }
294
295 return XII_FAILURE;
296}

361template <typename Type>
362bool xiiPlaneTemplate<Type>::GetRayIntersection(const xiiVec3Template<Type>& vRayStartPos, const xiiVec3Template<Type>& vRayDir, Type* out_pIntersectionDistance, xiiVec3Template<Type>* out_pIntersection) const
363{
364 XII_ASSERT_DEBUG(vRayStartPos.IsValid(), "Ray start position must be valid.");
365 XII_ASSERT_DEBUG(vRayDir.IsValid(), "Ray direction must be valid.");
366
367 const Type fPlaneSide = GetDistanceTo(vRayStartPos);
368 const Type fCosAlpha = m_vNormal.Dot(vRayDir);
369
370 if (fCosAlpha == 0) // ray is orthogonal to plane
371 return false;
372
373 if (xiiMath::Sign(fPlaneSide) == xiiMath::Sign(fCosAlpha)) // ray points away from the plane
374 return false;
375
376 const Type fTime = -fPlaneSide / fCosAlpha;
377
378 if (out_pIntersectionDistance)
379 *out_pIntersectionDistance = fTime;
380
381 if (out_pIntersection)
382 *out_pIntersection = vRayStartPos + fTime * vRayDir;
383
384 return true;
385}
