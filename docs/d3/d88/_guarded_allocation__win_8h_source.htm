41  {
42  XII_ASSERT_DEV(xiiMath::IsPowerOf2((xiiUInt32)uiAlign), "Alignment must be power of two");
43  uiAlign = xiiMath::Max<size_t>(uiAlign, XII_ALIGNMENT_MINIMUM);
44 
45  size_t uiAlignedSize = xiiMemoryUtils::AlignSize(uiSize, uiAlign);
46  size_t uiTotalSize = uiAlignedSize + sizeof(AlloctionMetaData);
47 
48  // Align to full pages and add one page in front and one in back.
49  size_t uiPageSize = m_uiPageSize;
50  size_t uiFullPageSize = xiiMemoryUtils::AlignSize(uiTotalSize, uiPageSize);
51  void* pMemory = VirtualAlloc(nullptr, uiFullPageSize + 2 * uiPageSize, MEM_RESERVE, PAGE_NOACCESS);
52  XII_ASSERT_DEV(pMemory != nullptr, "Could not reserve memory pages. Error Code '{0}'", xiiArgErrorCode(::GetLastError()));
53 
54  // Add one page and commit the payload pages.
55  pMemory = xiiMemoryUtils::AddByteOffset(pMemory, uiPageSize);
56  void* ptr = VirtualAlloc(pMemory, uiFullPageSize, MEM_COMMIT, PAGE_READWRITE);
57  XII_ASSERT_DEV(ptr != nullptr, "Could not commit memory pages. Error Code '{0}'", xiiArgErrorCode(::GetLastError()));
58 
59  // Store information in meta data.
60  AlloctionMetaData* metaData = xiiMemoryUtils::AddByteOffset(static_cast<AlloctionMetaData*>(ptr), uiFullPageSize - uiTotalSize);
61  xiiMemoryUtils::Construct(metaData, 1);
62  metaData->m_uiSize = uiAlignedSize;
63 
64  // Finally add offset to the actual payload.
65  ptr = xiiMemoryUtils::AddByteOffset(metaData, sizeof(AlloctionMetaData));
66  return ptr;

81 
82  m_AllocationsToFreeLater.PopFront();
83  }
84 
85  // Retrieve info from meta data first.
86  AlloctionMetaData* metaData = xiiMemoryUtils::AddByteOffset(static_cast<AlloctionMetaData*>(pPtr), -((ptrdiff_t)sizeof(AlloctionMetaData)));
87  size_t uiAlignedSize = metaData->m_uiSize;

90 
91  // Decommit the pages but do not release the memory yet so use-after-free can be detected.
92  size_t uiPageSize = m_uiPageSize;
93  size_t uiTotalSize = uiAlignedSize + sizeof(AlloctionMetaData);
94  size_t uiFullPageSize = xiiMemoryUtils::AlignSize(uiTotalSize, uiPageSize);
95  pPtr = xiiMemoryUtils::AddByteOffset(pPtr, ((ptrdiff_t)uiAlignedSize) - uiFullPageSize);
96 
97  XII_VERIFY(::VirtualFree(pPtr, uiFullPageSize, MEM_DECOMMIT), "Could not decommit memory pages. Error Code '{0}'", xiiArgErrorCode(::GetLastError()));
98 
99  // Finally store the allocation so we can release it later.
100  void* pMemory = xiiMemoryUtils::AddByteOffset(pPtr, -((ptrdiff_t)uiPageSize));
101  m_AllocationsToFreeLater.PushBack(pMemory);
