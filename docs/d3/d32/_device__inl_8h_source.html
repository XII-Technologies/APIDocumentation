170template <typename HandleType>
171void xiiGALDevice::ReviveDestroyedObject(xiiUInt32 uiType, HandleType handle)
172{
173 xiiUInt32 uiHandle = handle.GetInternalID().m_Data;
174
175 for (xiiUInt32 i = 0; i < m_DestroyedObjects.GetCount(); ++i)
176 {
177 const auto& destroyedObject = m_DestroyedObjects[i];
178
179 if (destroyedObject.m_uiType == uiType && destroyedObject.m_uiHandle == uiHandle)
180 {
181 m_DestroyedObjects.RemoveAtAndCopy(i);
182 return;
183 }
184 }
