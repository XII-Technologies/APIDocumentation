81 {
82 xiiUInt8 uiTempBuffer[1024];
83
84 xiiUInt64 uiBytesSkipped = 0;
85
86 while (uiBytesSkipped < uiBytesToSkip)
87 {
88 xiiUInt64 uiBytesToRead = xiiMath::Min<xiiUInt64>(uiBytesToSkip - uiBytesSkipped, 1024);
89
90 xiiUInt64 uiBytesRead = ReadBytes(uiTempBuffer, uiBytesToRead);
91
92 uiBytesSkipped += uiBytesRead;
93
94 // Terminate early if the stream didn't read as many bytes as we requested (EOF for example)
95 if (uiBytesRead < uiBytesToRead)
96 break;
97 }
98
99 return uiBytesSkipped;
100 }
