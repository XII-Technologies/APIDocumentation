142XII_ALWAYS_INLINE void xiiSimdBBoxd::Transform(const xiiSimdMat4d& mMat)
143{
144 const xiiSimdVec4d center = GetCenter();
145 const xiiSimdVec4d halfExtents = center - m_Min;
146
147 const xiiSimdVec4d newCenter = mMat.TransformPosition(center);
148
149 xiiSimdVec4d newHalfExtents = mMat.m_col0.Abs() * halfExtents.x();
150 newHalfExtents += mMat.m_col1.Abs() * halfExtents.y();
151 newHalfExtents += mMat.m_col2.Abs() * halfExtents.z();
152
153 *this = xiiSimdBBoxd::MakeFromCenterAndHalfExtents(newCenter, newHalfExtents);
154}
